1//! Dependency collection for program extraction.
2//!
3//! Collects all transitive dependencies needed to extract a definition.
4
5use crate::kernel::{Context, Term};
6use std::collections::HashSet;
7
8/// Collect all dependencies of a definition.
9///
10/// Starting from an entry point, recursively collects all global names
11/// that are referenced, including inductives, constructors, and definitions.
12pub fn collect_dependencies(ctx: &Context, entry: &str) -> HashSet<String> {
13    let mut visited = HashSet::new();
14    let mut to_visit = vec![entry.to_string()];
15
16    while let Some(name) = to_visit.pop() {
17        if visited.contains(&name) {
18            continue;
19        }
20        visited.insert(name.clone());
21
22        // Get the term to analyze based on what kind of global it is
23        if let Some(body) = ctx.get_definition_body(&name) {
24            collect_globals(body, &mut to_visit);
25            // Also collect from the type
26            if let Some(ty) = ctx.get_definition_type(&name) {
27                collect_globals(ty, &mut to_visit);
28            }
29        }
30
31        if ctx.is_inductive(&name) {
32            // Add constructors and their types
33            for (ctor_name, ctor_ty) in ctx.get_constructors(&name) {
34                collect_globals(ctor_ty, &mut to_visit);
35                to_visit.push(ctor_name.to_string());
36            }
37        }
38
39        if ctx.is_constructor(&name) {
40            // Add the inductive type
41            if let Some(ind) = ctx.constructor_inductive(&name) {
42                to_visit.push(ind.to_string());
43            }
44        }
45    }
46
47    visited
48}
49
50/// Recursively collect all Global references from a term.
51fn collect_globals(term: &Term, deps: &mut Vec<String>) {
52    match term {
53        Term::Global(name) => deps.push(name.clone()),
54        Term::App(f, a) => {
55            collect_globals(f, deps);
56            collect_globals(a, deps);
57        }
58        Term::Lambda {
59            param_type, body, ..
60        } => {
61            collect_globals(param_type, deps);
62            collect_globals(body, deps);
63        }
64        Term::Pi {
65            param_type,
66            body_type,
67            ..
68        } => {
69            collect_globals(param_type, deps);
70            collect_globals(body_type, deps);
71        }
72        Term::Fix { body, .. } => collect_globals(body, deps),
73        Term::Match {
74            discriminant,
75            motive,
76            cases,
77        } => {
78            collect_globals(discriminant, deps);
79            collect_globals(motive, deps);
80            for case in cases {
81                collect_globals(case, deps);
82            }
83        }
84        // Base cases: no dependencies
85        Term::Sort(_) | Term::Var(_) | Term::Lit(_) | Term::Hole => {}
86    }
87}
