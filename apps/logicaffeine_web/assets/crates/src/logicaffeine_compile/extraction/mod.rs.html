1//! Program extraction from kernel terms to Rust.
2//!
3//! This module compiles verified kernel terms into executable Rust source code.
4//!
5//! # Extraction Rules
6//!
7//! | Kernel Term | Rust Output |
8//! |-------------|-------------|
9//! | Inductive | `enum Name { Ctor(Args...) }` |
10//! | Fix + Lambdas | `fn name(args) -> Ret { body }` |
11//! | Match | `match disc { Name::Ctor(vars) => body }` |
12//! | App | `func(arg)` |
13//!
14//! # Structural Honesty
15//!
16//! We extract types exactly as defined, with no optimization.
17//! For example, `Nat` becomes `enum Nat { Zero, Succ(Box<Nat>) }`.
18//! This guarantees that the extracted code is isomorphic to the verified logic.
19

31/// Extract a program rooted at the given entry point.
32///
33/// Returns Rust source code as a string.
34///
35/// # Arguments
36///
37/// * `ctx` - The kernel context containing definitions and inductives
38/// * `entry` - The name of the entry point to extract
39///
40/// # Errors
41///
42/// Returns an error if the entry point is not found or cannot be extracted.
43pub fn extract_program(ctx: &Context, entry: &str) -> Result<String, ExtractError> {
44    // Check if entry exists
45    if !ctx.is_inductive(entry) && !ctx.is_definition(entry) && !ctx.is_constructor(entry) {
46        return Err(ExtractError::NotFound(entry.to_string()));
47    }
48
49    // 1. Collect all dependencies
50    let deps = collect_dependencies(ctx, entry);
51
52    // 2. Topologically sort (inductives first, then definitions)
53    let sorted = topological_sort(ctx, &deps);
54
55    // 3. Generate code
56    let mut codegen = CodeGen::new(ctx);
57    for name in &sorted {
58        if ctx.is_inductive(name) {
59            codegen.emit_inductive(name)?;
60        } else if ctx.is_definition(name) {
61            codegen.emit_definition(name)?;
62        }
63        // Skip constructors (emitted with their inductive)
64    }
65
66    Ok(codegen.finish())
67}
68
69/// Topologically sort dependencies.
70///
71/// Inductives come first, then definitions in dependency order.
72fn topological_sort(ctx: &Context, deps: &HashSet<String>) -> Vec<String> {
73    let mut inductives = Vec::new();
74    let mut definitions = Vec::new();
75
76    for name in deps {
77        if ctx.is_inductive(name) {
78            inductives.push(name.clone());
79        } else if ctx.is_definition(name) {
80            definitions.push(name.clone());
81        }
82        // Skip constructors - they're emitted with their inductive
83    }
84
85    // Simple topological ordering: inductives first, then definitions
86    // A more sophisticated implementation would do proper dependency analysis
87    inductives.extend(definitions);
88    inductives
89}
