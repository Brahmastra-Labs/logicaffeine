1//! Diagnostic bridge for translating Rust errors to LOGOS.
2//!
3//! Translates Rust borrow checker errors into friendly LOGOS error messages
4//! using Socratic phrasing. Parses rustc JSON output and maps errors back
5//! to LOGOS source locations using the [`SourceMap`](crate::sourcemap::SourceMap).
6//!
7//! # Supported Error Codes
8//!
9//! | Rust Error | LOGOS Translation |
10//! |------------|-------------------|
11//! | E0382 | "Cannot use 'x' after giving it away" |
12//! | E0505 | "Cannot borrow 'x' while it's borrowed elsewhere" |
13//! | E0597 | "Reference 'x' cannot escape zone" |
14//!
15//! # Translation Flow
16//!
17//! ```text
18//! rustc --error-format=json
19//!           │
20//!           ▼
21//! ┌─────────────────────┐
22//! │ parse_rustc_json()  │ Parse JSON diagnostics
23//! └──────────┬──────────┘
24//!            ▼
25//! ┌─────────────────────┐
26//! │ translate_diagnostics│ Map to LOGOS source
27//! └──────────┬──────────┘
28//!            ▼
29//!    LogosError with friendly message
30

37/// A translated error message for LOGOS users.
38#[derive(Debug, Clone)]
39pub struct LogosError {
40    pub title: String,
41    pub explanation: String,
42    pub logos_span: Option<Span>,
43    pub suggestion: Option<String>,
44}
45
46impl std::fmt::Display for LogosError {
47    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
48        writeln!(f, "{}: {}", Style::bold_red("ownership error"), self.title)?;
49        writeln!(f)?;
50        writeln!(f, "{}", self.explanation)?;
51        if let Some(suggestion) = &self.suggestion {
52            writeln!(f)?;
53            writeln!(f, "{}: {}", Style::cyan("suggestion"), suggestion)?;
54        }
55        Ok(())
56    }
57}

99/// Parsed rustc output: either a diagnostic or artifact info.
100#[derive(Debug, Deserialize)]
101#[serde(tag = "reason")]
102#[serde(rename_all = "kebab-case")]
103pub enum RustcMessage {
104    CompilerMessage { message: RustcDiagnostic },
105    #[serde(other)]
106    Other,
107}

113/// Parse rustc stderr output from `cargo build --message-format=json`.
114pub fn parse_rustc_json(stderr: &str) -> Vec<RustcDiagnostic> {
115    let mut diagnostics = Vec::new();
116
117    for line in stderr.lines() {
118        // Skip empty lines and non-JSON output
119        if !line.starts_with('{') {
120            continue;
121        }
122
123        match serde_json::from_str::<RustcMessage>(line) {
124            Ok(RustcMessage::CompilerMessage { message }) => {
125                if message.level == "error" {
126                    diagnostics.push(message);
127                }
128            }
129            Ok(RustcMessage::Other) => {} // Ignore artifacts, build-finished, etc.
130            Err(_) => {} // Ignore malformed lines
131        }
132    }
133
134    diagnostics
135}

137/// Extract error code from a diagnostic.
138pub fn get_error_code(diag: &RustcDiagnostic) -> Option<&str> {
139    diag.code.as_ref().map(|c| c.code.as_str())
140}
141
142/// Extract the primary span from a diagnostic.
143pub fn get_primary_span(diag: &RustcDiagnostic) -> Option<&RustcSpan> {
144    diag.spans.iter().find(|s| s.is_primary)
145}

147/// Extract variable name from rustc error message.
148/// Example: "use of moved value: `data`" -> "data"
149fn extract_var_from_message(message: &str, prefix: &str, suffix: &str) -> Option<String> {
150    let start = message.find(prefix)?;
151    let after_prefix = &message[start + prefix.len()..];
152    let end = after_prefix.find(suffix)?;
153    Some(after_prefix[..end].to_string())
154}

160/// Translates rustc diagnostics into LOGOS error messages.
161pub struct DiagnosticBridge<'a> {
162    source_map: &'a SourceMap,
163    interner: &'a Interner,
164}
165
166impl<'a> DiagnosticBridge<'a> {
167    pub fn new(source_map: &'a SourceMap, interner: &'a Interner) -> Self {
168        Self { source_map, interner }
169    }
170
171    /// Translate a rustc diagnostic into a LOGOS error.
172    pub fn translate(&self, diag: &RustcDiagnostic) -> Option<LogosError> {
173        let code = get_error_code(diag)?;
174        let span = get_primary_span(diag);
175
176        match code {
177            "E0382" => self.translate_use_after_move(diag, span),
178            "E0505" => self.translate_move_while_borrowed(diag, span),
179            "E0597" => self.translate_lifetime_error(diag, span),
180            _ => self.translate_generic(diag, span),
181        }
182    }
183
184    /// E0382: "use of moved value: `x`"
185    /// LOGOS: "You already gave X away - you can't use it anymore"
186    fn translate_use_after_move(&self, diag: &RustcDiagnostic, span: Option<&RustcSpan>) -> Option<LogosError> {
187        let var_name = extract_var_from_message(&diag.message, "value: `", "`")
188            .or_else(|| extract_var_from_message(&diag.message, "value `", "`"))?;
189
190        let logos_span = span.and_then(|s| self.source_map.find_nearest_span(s.line_start));
191
192        // Look up variable origin if available
193        let (logos_name, role) = if let Some(origin) = self.source_map.get_var_origin(&var_name) {
194            (self.interner.resolve(origin.logos_name).to_string(), Some(origin.role))
195        } else {
196            (var_name.clone(), None)
197        };
198
199        let explanation = match role {
200            Some(OwnershipRole::GiveObject) => format!(
201                "You gave '{}' away with a Give statement, so you can't use it anymore.\n\
202                In LOGOS, 'Give X to Y' transfers ownership - X moves to Y and leaves your hands.\n\
203                This is like handing someone a physical object: once given, you no longer have it.",
204                logos_name
205            ),
206            Some(OwnershipRole::LetBinding) | None => format!(
207                "The value '{}' was moved somewhere else and can't be used again.\n\
208                Check if you used 'Give' or passed it to a function that took ownership.",
209                logos_name
210            ),
211            _ => format!(
212                "The value '{}' has been moved and is no longer available.",
213                logos_name
214            ),
215        };
216
217        let suggestion = Some(format!(
218            "If you need to use '{}' after giving it away, either:\n\
219             1. Use 'Show {} to Y' instead (this borrows, keeping ownership)\n\
220             2. Use 'a copy of {}' before the Give",
221            logos_name, logos_name, logos_name
222        ));
223
224        Some(LogosError {
225            title: format!("Cannot use '{}' after giving it away", logos_name),
226            explanation,
227            logos_span,
228            suggestion,
229        })
230    }
231
232    /// E0505: "cannot move out of `x` because it is borrowed"
233    /// LOGOS: "You're trying to give X away while someone is still looking at it"
234    fn translate_move_while_borrowed(&self, diag: &RustcDiagnostic, span: Option<&RustcSpan>) -> Option<LogosError> {
235        let var_name = extract_var_from_message(&diag.message, "out of `", "`")
236            .or_else(|| extract_var_from_message(&diag.message, "move out of `", "`"))?;
237
238        let logos_span = span.and_then(|s| self.source_map.find_nearest_span(s.line_start));
239
240        let logos_name = if let Some(origin) = self.source_map.get_var_origin(&var_name) {
241            self.interner.resolve(origin.logos_name).to_string()
242        } else {
243            var_name.clone()
244        };
245
246        let explanation = format!(
247            "You showed '{}' to someone (creating a temporary view),\n\
248            but then tried to give it away before they finished looking.\n\
249            In LOGOS, 'Show' creates a promise that the data won't change or disappear\n\
250            while being viewed. You can't break that promise by giving it away.",
251            logos_name
252        );
253
254        let suggestion = Some(format!(
255            "Make sure all 'Show' usages of '{}' complete before any 'Give'.\n\
256            Alternatively, give away a copy: 'Give a copy of {} to Y'",
257            logos_name, logos_name
258        ));
259
260        Some(LogosError {
261            title: format!("Cannot give '{}' while it's being shown", logos_name),
262            explanation,
263            logos_span,
264            suggestion,
265        })
266    }

268    /// E0597: "borrowed value does not live long enough"
269    /// LOGOS: "You can't take a reference outside its zone" (Hotel California)
270    fn translate_lifetime_error(&self, diag: &RustcDiagnostic, span: Option<&RustcSpan>) -> Option<LogosError> {
271        let logos_span = span.and_then(|s| self.source_map.find_nearest_span(s.line_start));
272
273        // Check if this is zone-related by looking at the message and children
274        let is_zone_related = diag.message.contains("borrowed")
275            || diag.children.iter().any(|c| c.message.contains("dropped"));
276
277        let explanation = if is_zone_related {
278            "A value created inside a Zone cannot be referenced from outside.\n\
279            Zones are memory arenas - when the Zone ends, everything inside it is released.\n\
280            This is the 'Hotel California' rule: data can check in (be created),\n\
281            but references can't check out (escape the Zone).".to_string()
282        } else {
283            "A borrowed reference is being used after the original value has gone away.\n\
284            References are temporary views - they can't outlive what they're viewing.".to_string()
285        };
286
287        let suggestion = Some(
288            "If you need the data after the Zone ends, either:\n\
289             1. Move the data out with 'Give' before the Zone closes\n\
290             2. Copy the data: 'Let result be a copy of zone_data'\n\
291             3. Restructure so the computation completes inside the Zone".to_string()
292        );
293
294        Some(LogosError {
295            title: "Reference cannot outlive its data".to_string(),
296            explanation,
297            logos_span,
298            suggestion,
299        })
300    }
301
302    /// Fallback for other errors - provide the raw message with context.
303    fn translate_generic(&self, diag: &RustcDiagnostic, span: Option<&RustcSpan>) -> Option<LogosError> {
304        let logos_span = span.and_then(|s| self.source_map.find_nearest_span(s.line_start));
305
306        // Try to extract any variable name
307        let var_hint = if let Some(start) = diag.message.find('`') {
308            if let Some(end) = diag.message[start + 1..].find('`') {
309                Some(&diag.message[start + 1..start + 1 + end])
310            } else {
311                None
312            }
313        } else {
314            None
315        };
316
317        let explanation = if let Some(var) = var_hint {
318            format!(
319                "The Rust compiler reported an error involving '{}':\n{}",
320                var, diag.message
321            )
322        } else {
323            format!("The Rust compiler reported an error:\n{}", diag.message)
324        };
325
326        Some(LogosError {
327            title: "Compilation error".to_string(),
328            explanation,
329            logos_span,
330            suggestion: None,
331        })
332    }
333}

335/// Attempt to translate all diagnostics and return the first translated error.
336pub fn translate_diagnostics(
337    diagnostics: &[RustcDiagnostic],
338    source_map: &SourceMap,
339    interner: &Interner,
340) -> Option<LogosError> {
341    let bridge = DiagnosticBridge::new(source_map, interner);
342
343    for diag in diagnostics {
344        if let Some(error) = bridge.translate(diag) {
345            return Some(error);
346        }
347    }
348
349    None
350}
