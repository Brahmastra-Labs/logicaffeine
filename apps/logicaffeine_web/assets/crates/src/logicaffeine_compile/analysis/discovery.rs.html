1//! Discovery extensions with module imports.
2//!
3//! Provides [`discover_with_imports`] which uses the [`Loader`](crate::loader::Loader)
4//! for multi-file projects with namespace support.
5//!
6//! # Multi-File Type Discovery
7//!
8//! ```text
9//! main.md                           geometry.md
10//! ┌────────────────────────┐        ┌────────────────────────┐
11//! │ ## Abstract            │        │ ## Definition          │
12//! │ [Geometry](geometry.md)│───────▶│ A Point has x: Int,    │
13//! └────────────────────────┘        │   y: Int.              │
14//!                                   └────────────────────────┘
15//!         ↓
16//! TypeRegistry contains:
17//!   - Geometry::Point { x: Int, y: Int }
18//! ```
19
20use std::path::Path;
21use crate::loader::Loader;
22use crate::{Interner, Lexer, mwe};
23use super::{TypeRegistry, TypeDef, DiscoveryPass, scan_dependencies};

25/// Recursive discovery with module imports.
26///
27/// Scans a LOGOS source file for:
28/// 1. Dependencies declared in the Abstract (Markdown links)
29/// 2. Type definitions in `## Definition` blocks
30///
31/// # Namespace Prefixing
32///
33/// Dependencies are loaded recursively, and their types are merged into
34/// the registry with namespace prefixes (e.g., `Geometry::Point`).
35///
36/// # Arguments
37///
38/// * `file_path` - Path to the source file
39/// * `source` - Source code content
40/// * `loader` - Module loader for resolving imports
41/// * `interner` - Symbol interner
42///
43/// # Example
44///
45/// ```ignore
46/// let mut loader = Loader::new(&root_path);
47/// let registry = discover_with_imports(
48///     Path::new("main.md"),
49///     source,
50///     &mut loader,
51///     &mut interner
52/// )?;
53/// // registry now contains types from main.md and all imports
54/// ```
55pub fn discover_with_imports(
56    file_path: &Path,
57    source: &str,
58    loader: &mut Loader,
59    interner: &mut Interner,
60) -> Result<TypeRegistry, String> {
61    let mut registry = TypeRegistry::with_primitives(interner);
62
63    // 1. Scan for dependencies in the abstract
64    let deps = scan_dependencies(source);
65
66    // 2. For each dependency, recursively discover types
67    for dep in deps {
68        let module_source = loader.resolve(file_path, &dep.uri)?;
69        let dep_content = module_source.content.clone();
70        let dep_path = module_source.path.clone();
71
72        // Recursively discover types in the dependency
73        let dep_registry = discover_with_imports(
74            &dep_path,
75            &dep_content,
76            loader,
77            interner
78        )?;
79
80        // Merge with namespace prefix
81        merge_registry(&mut registry, &dep.alias, dep_registry, interner);
82    }
83
84    // 3. Scan local definitions using existing DiscoveryPass
85    let mut lexer = Lexer::new(source, interner);
86    let tokens = lexer.tokenize();
87    let mwe_trie = mwe::build_mwe_trie();
88    let tokens = mwe::apply_mwe_pipeline(tokens, &mwe_trie, interner);
89
90    let mut discovery = DiscoveryPass::new(&tokens, interner);
91    let local_registry = discovery.run();
92
93    // Merge local types (without namespace prefix)
94    for (sym, def) in local_registry.iter_types() {
95        // Skip primitives (already in registry)
96        let name = interner.resolve(*sym);
97        if !["Int", "Nat", "Text", "Bool", "Real", "Unit"].contains(&name) {
98            registry.register(*sym, def.clone());
99        }
100    }
101
102    Ok(registry)
103}

105/// Merges types from a dependency registry into the main registry with namespace prefix.
106fn merge_registry(
107    main: &mut TypeRegistry,
108    namespace: &str,
109    dep: TypeRegistry,
110    interner: &mut Interner,
111) {
112    for (sym, def) in dep.iter_types() {
113        // Get original name
114        let orig_name = interner.resolve(*sym);
115
116        // Skip primitives
117        if ["Int", "Nat", "Text", "Bool", "Real", "Unit"].contains(&orig_name) {
118            continue;
119        }
120
121        // Create namespaced name: "Geometry::Point"
122        let namespaced = format!("{}::{}", namespace, orig_name);
123        let namespaced_sym = interner.intern(&namespaced);
124
125        // Clone the typedef with the new namespaced symbol
126        let namespaced_def = match def {
127            TypeDef::Struct { fields, generics, is_portable, is_shared } => TypeDef::Struct {
128                fields: fields.clone(),
129                generics: generics.clone(),
130                is_portable: *is_portable,
131                is_shared: *is_shared,
132            },
133            TypeDef::Enum { variants, generics, is_portable, is_shared } => TypeDef::Enum {
134                variants: variants.clone(),
135                generics: generics.clone(),
136                is_portable: *is_portable,
137                is_shared: *is_shared,
138            },
139            TypeDef::Alias { target } => TypeDef::Alias {
140                target: target.clone(),
141            },
142            TypeDef::Generic { param_count } => TypeDef::Generic {
143                param_count: *param_count,
144            },
145            TypeDef::Primitive => TypeDef::Primitive,
146        };
147
148        main.register(namespaced_sym, namespaced_def);
149    }
150}
