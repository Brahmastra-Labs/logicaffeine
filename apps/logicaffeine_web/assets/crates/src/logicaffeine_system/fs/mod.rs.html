1//! Virtual File System Abstraction
2//!
3//! Provides platform-agnostic async file operations through the [`Vfs`] trait.
4//! This enables the same code to work on native platforms and in the browser.
5//!
6//! # Platform Implementations
7//!
8//! - **Native** ([`NativeVfs`]): Uses `tokio::fs` with atomic write operations
9//! - **WASM** ([`OpfsVfs`]): Uses the browser's Origin Private File System API
10//!
11//! # Features
12//!
13//! Requires the `persistence` feature.
14//!
15//! # Example
16//!
17//! ```rust,ignore
18//! use logicaffeine_system::fs::{Vfs, NativeVfs};
19//! use std::sync::Arc;
20//!
21//! let vfs: Arc<dyn Vfs + Send + Sync> = Arc::new(NativeVfs::new("/data"));
22//!
23//! // Write and read files
24//! vfs.write("config.json", b"{}").await?;
25//! let data = vfs.read("config.json").await?;
26//!
27//! // Atomic append for journaling
28//! vfs.append("log.txt", b"entry\n").await?;
29//! ```

43/// Error type for VFS operations
44#[derive(Debug)]
45pub enum VfsError {
46    NotFound(String),
47    PermissionDenied(String),
48    AlreadyExists(String),
49    IoError(io::Error),
50    SerializationError(String),
51    JournalCorrupted(String),
52}
53
54impl std::fmt::Display for VfsError {
55    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
56        match self {
57            VfsError::NotFound(s) => write!(f, "Not found: {}", s),
58            VfsError::PermissionDenied(s) => write!(f, "Permission denied: {}", s),
59            VfsError::AlreadyExists(s) => write!(f, "Already exists: {}", s),
60            VfsError::IoError(e) => write!(f, "IO error: {}", e),
61            VfsError::SerializationError(s) => write!(f, "Serialization error: {}", s),
62            VfsError::JournalCorrupted(s) => write!(f, "Journal corrupted: {}", s),
63        }
64    }
65}
66
67impl std::error::Error for VfsError {}
68
69impl From<io::Error> for VfsError {
70    fn from(e: io::Error) -> Self {
71        match e.kind() {
72            io::ErrorKind::NotFound => VfsError::NotFound(e.to_string()),
73            io::ErrorKind::PermissionDenied => VfsError::PermissionDenied(e.to_string()),
74            io::ErrorKind::AlreadyExists => VfsError::AlreadyExists(e.to_string()),
75            _ => VfsError::IoError(e),
76        }
77    }
78}

91/// Virtual File System trait for platform-agnostic file operations.
92///
93/// On native platforms, requires Send+Sync for thread-safe access.
94/// On WASM, these bounds are relaxed since JS is single-threaded.
95#[cfg(not(target_arch = "wasm32"))]
96#[async_trait]
97pub trait Vfs: Send + Sync {
98    /// Read entire file contents as bytes.
99    async fn read(&self, path: &str) -> VfsResult<Vec<u8>>;
100
101    /// Read file contents as UTF-8 string.
102    async fn read_to_string(&self, path: &str) -> VfsResult<String>;
103
104    /// Write bytes to file (atomic on native, best-effort on WASM).
105    async fn write(&self, path: &str, contents: &[u8]) -> VfsResult<()>;
106
107    /// Append bytes to file (atomic append semantics).
108    async fn append(&self, path: &str, contents: &[u8]) -> VfsResult<()>;
109
110    /// Check if file exists.
111    async fn exists(&self, path: &str) -> VfsResult<bool>;
112
113    /// Delete a file.
114    async fn remove(&self, path: &str) -> VfsResult<()>;
115
116    /// Create directory and all parent directories.
117    async fn create_dir_all(&self, path: &str) -> VfsResult<()>;
118
119    /// Atomically rename a file (for journal compaction).
120    async fn rename(&self, from: &str, to: &str) -> VfsResult<()>;
121
122    /// List entries in a directory.
123    async fn list_dir(&self, path: &str) -> VfsResult<Vec<DirEntry>>;
124}

158/// Native filesystem VFS using tokio::fs.
159#[cfg(not(target_arch = "wasm32"))]
160pub struct NativeVfs {
161    /// Base directory for all operations (sandbox root).
162    base_dir: PathBuf,
163}
164
165#[cfg(not(target_arch = "wasm32"))]
166impl NativeVfs {
167    /// Create a new NativeVfs rooted at the given directory.
168    pub fn new<P: Into<PathBuf>>(base_dir: P) -> Self {
169        Self {
170            base_dir: base_dir.into(),
171        }
172    }
173
174    /// Resolve a virtual path to an absolute filesystem path.
175    fn resolve(&self, path: &str) -> PathBuf {
176        // Security: Prevent path traversal attacks
177        let clean = path.trim_start_matches('/').trim_start_matches("../");
178        self.base_dir.join(clean)
179    }
180}
181
182#[cfg(not(target_arch = "wasm32"))]
183#[async_trait]
184impl Vfs for NativeVfs {
185    async fn read(&self, path: &str) -> VfsResult<Vec<u8>> {
186        let full_path = self.resolve(path);
187        tokio::fs::read(&full_path).await.map_err(VfsError::from)
188    }
189
190    async fn read_to_string(&self, path: &str) -> VfsResult<String> {
191        let full_path = self.resolve(path);
192        tokio::fs::read_to_string(&full_path).await.map_err(VfsError::from)
193    }
194
195    async fn write(&self, path: &str, contents: &[u8]) -> VfsResult<()> {
196        let full_path = self.resolve(path);
197
198        // Ensure parent directory exists
199        if let Some(parent) = full_path.parent() {
200            tokio::fs::create_dir_all(parent).await?;
201        }
202
203        // Atomic write: write to temp file, then rename
204        let temp_path = full_path.with_extension("tmp");
205        tokio::fs::write(&temp_path, contents).await?;
206        tokio::fs::rename(&temp_path, &full_path).await?;
207
208        Ok(())
209    }
210
211    async fn append(&self, path: &str, contents: &[u8]) -> VfsResult<()> {
212        use tokio::io::AsyncWriteExt;
213
214        let full_path = self.resolve(path);
215
216        // Ensure parent directory exists
217        if let Some(parent) = full_path.parent() {
218            tokio::fs::create_dir_all(parent).await?;
219        }
220
221        let mut file = tokio::fs::OpenOptions::new()
222            .create(true)
223            .append(true)
224            .open(&full_path)
225            .await?;
226
227        file.write_all(contents).await?;
228        file.sync_all().await?;
229
230        Ok(())
231    }
232
233    async fn exists(&self, path: &str) -> VfsResult<bool> {
234        let full_path = self.resolve(path);
235        Ok(full_path.exists())
236    }
237
238    async fn remove(&self, path: &str) -> VfsResult<()> {
239        let full_path = self.resolve(path);
240        tokio::fs::remove_file(&full_path).await.map_err(VfsError::from)
241    }
242
243    async fn create_dir_all(&self, path: &str) -> VfsResult<()> {
244        let full_path = self.resolve(path);
245        tokio::fs::create_dir_all(&full_path).await.map_err(VfsError::from)
246    }
247
248    async fn rename(&self, from: &str, to: &str) -> VfsResult<()> {
249        let from_path = self.resolve(from);
250        let to_path = self.resolve(to);
251        tokio::fs::rename(&from_path, &to_path).await.map_err(VfsError::from)
252    }
253
254    async fn list_dir(&self, path: &str) -> VfsResult<Vec<DirEntry>> {
255        let full_path = self.resolve(path);
256        let mut entries = Vec::new();
257        let mut read_dir = tokio::fs::read_dir(&full_path).await.map_err(VfsError::from)?;
258
259        while let Some(entry) = read_dir.next_entry().await.map_err(VfsError::from)? {
260            let metadata = entry.metadata().await.map_err(VfsError::from)?;
261            entries.push(DirEntry {
262                name: entry.file_name().to_string_lossy().into_owned(),
263                is_directory: metadata.is_dir(),
264            });
265        }
266
267        // Sort entries: directories first, then alphabetically
268        entries.sort_by(|a, b| {
269            match (a.is_directory, b.is_directory) {
270                (true, false) => std::cmp::Ordering::Less,
271                (false, true) => std::cmp::Ordering::Greater,
272                _ => a.name.to_lowercase().cmp(&b.name.to_lowercase()),
273            }
274        });
275
276        Ok(entries)
277    }
278}

340    #[tokio::test]
341    async fn test_native_vfs_list_dir() {
342        let temp = TempDir::new().unwrap();
343        let vfs = NativeVfs::new(temp.path());
344
345        // Create files and directories
346        vfs.write("file1.txt", b"content1").await.unwrap();
347        vfs.write("file2.txt", b"content2").await.unwrap();
348        vfs.write("subdir/nested.txt", b"nested").await.unwrap();
349
350        // List root directory
351        let entries = vfs.list_dir("").await.unwrap();
352
353        // Should have 3 entries: subdir (dir), file1.txt, file2.txt
354        assert_eq!(entries.len(), 3);
355
356        // Directory should come first
357        assert_eq!(entries[0].name, "subdir");
358        assert!(entries[0].is_directory);
359
360        // Files should be alphabetically sorted
361        assert_eq!(entries[1].name, "file1.txt");
362        assert!(!entries[1].is_directory);
363        assert_eq!(entries[2].name, "file2.txt");
364        assert!(!entries[2].is_directory);
365    }
