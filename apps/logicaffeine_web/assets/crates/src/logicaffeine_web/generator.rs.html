1//! Exercise generation from templates.
2//!
3//! Transforms exercise configurations from the curriculum into concrete
4//! challenges by filling template slots with words from the lexicon.
5//!
6//! # Template Syntax
7//!
8//! Templates use curly braces for slots: `{SlotType}` or `{SlotType:Modifier}`.
9//!
10//! ## Slot Types
11//!
12//! | Slot | Description | Example Output |
13//! |------|-------------|----------------|
14//! | `{ProperName}` | Random proper noun | "Alice", "Bob" |
15//! | `{Noun}` | Random common noun (singular) | "cat", "dog" |
16//! | `{Noun:Plural}` | Pluralized common noun | "cats", "dogs" |
17//! | `{Verb}` | Base form verb | "run", "eat" |
18//! | `{Verb:Past}` | Past tense | "ran", "ate" |
19//! | `{Verb:Present3s}` | Third person singular | "runs", "eats" |
20//! | `{Verb:Gerund}` | Present participle | "running", "eating" |
21//! | `{Adjective}` | Intersective adjective | "tall", "red" |
22//!
23//! ## Constraints
24//!
25//! Constraints filter word selection by semantic features. They're specified
26//! in the exercise config's `constraints` map:
27//!
28//! ```json
29//! {
30//!   "Verb": ["Intransitive"],
31//!   "Noun": ["Animate"]
32//! }
33//! ```
34//!
35//! ## Example
36//!
37//! Template: `"{ProperName} {Verb:Present3s}."`
38//! With constraint `Verb: ["Intransitive"]`
39//! Output: `"Alice runs."`
40//!
41//! # Usage
42//!
43//! ```ignore
44//! use logicaffeine_web::generator::Generator;
45//! use rand::SeedableRng;
46//!
47//! let generator = Generator::new();
48//! let mut rng = rand::rngs::StdRng::seed_from_u64(42);
49//!
50//! if let Some(challenge) = generator.generate(&exercise_config, &mut rng) {
51//!     println!("Sentence: {}", challenge.sentence);
52//! }
53//! ```

62/// Exercise generator that fills templates with lexicon words.
63///
64/// Holds a reference to the lexicon index for word selection.
65/// Stateless; create one instance and reuse for all exercise generation.
66pub struct Generator {
67    lexicon: LexiconIndex,
68}
69
70/// A generated exercise instance ready for display and grading.
71///
72/// Contains the filled template, expected answer(s), and optional hints.
73#[derive(Debug, Clone)]
74pub struct Challenge {
75    /// Unique identifier linking back to the exercise config.
76    pub exercise_id: String,
77    /// Instructions shown to the user (e.g., "Translate this sentence").
78    pub prompt: String,
79    /// The filled template sentence for the user to translate.
80    pub sentence: String,
81    /// The expected answer format and content.
82    pub answer: AnswerType,
83    /// Optional hint revealed on request.
84    pub hint: Option<String>,
85    /// Detailed explanation shown after answering.
86    pub explanation: Option<String>,
87}
88
89/// Expected answer format for a challenge.
90#[derive(Debug, Clone)]
91pub enum AnswerType {
92    /// User enters free-form FOL expression, compared against golden answer.
93    FreeForm {
94        /// The correct FOL translation, pre-compiled from the sentence.
95        golden_logic: String,
96    },
97    /// User selects from multiple predefined options.
98    MultipleChoice {
99        /// List of answer choices to display.
100        options: Vec<String>,
101        /// Zero-based index of the correct option.
102        correct_index: usize,
103    },
104    /// Sentence has multiple valid readings; user must provide all.
105    Ambiguity {
106        /// All valid FOL translations for the ambiguous sentence.
107        readings: Vec<String>,
108    },
109}

111impl Generator {
112    /// Creates a new generator with a fresh lexicon index.
113    pub fn new() -> Self {
114        Self {
115            lexicon: LexiconIndex::new(),
116        }
117    }
118
119    /// Generates a challenge from an exercise configuration.
120    ///
121    /// Returns `None` if:
122    /// - The template cannot be filled (missing word categories)
123    /// - The generated sentence fails to compile (invalid grammar)
124    /// - Required fields are missing from the exercise config
125    pub fn generate(&self, exercise: &ExerciseConfig, rng: &mut impl Rng) -> Option<Challenge> {
126        match exercise.exercise_type {
127            ExerciseType::Translation => self.generate_translation(exercise, rng),
128            ExerciseType::MultipleChoice => self.generate_multiple_choice(exercise, rng),
129            ExerciseType::Ambiguity => self.generate_ambiguity(exercise, rng),
130        }
131    }
132
133    fn generate_translation(&self, exercise: &ExerciseConfig, rng: &mut impl Rng) -> Option<Challenge> {
134        let template = exercise.template.as_ref()?;
135        let sentence = self.fill_template(template, &exercise.constraints, rng)?;
136
137        let golden_logic = compile(&sentence).ok()?;
138
139        Some(Challenge {
140            exercise_id: exercise.id.clone(),
141            prompt: exercise.prompt.clone(),
142            sentence,
143            answer: AnswerType::FreeForm { golden_logic },
144            hint: exercise.hint.clone(),
145            explanation: exercise.explanation.clone(),
146        })
147    }
148
149    fn generate_multiple_choice(&self, exercise: &ExerciseConfig, rng: &mut impl Rng) -> Option<Challenge> {
150        let options = exercise.options.clone()?;
151        let correct_index = exercise.correct?;
152
153        let sentence = if let Some(template) = &exercise.template {
154            self.fill_template(template, &exercise.constraints, rng)?
155        } else {
156            exercise.prompt.clone()
157        };
158
159        Some(Challenge {
160            exercise_id: exercise.id.clone(),
161            prompt: exercise.prompt.clone(),
162            sentence,
163            answer: AnswerType::MultipleChoice { options, correct_index },
164            hint: exercise.hint.clone(),
165            explanation: exercise.explanation.clone(),
166        })
167    }
168
169    fn generate_ambiguity(&self, exercise: &ExerciseConfig, rng: &mut impl Rng) -> Option<Challenge> {
170        let template = exercise.template.as_ref()?;
171        let sentence = self.fill_template(template, &exercise.constraints, rng)?;
172
173        let readings = compile_all_scopes(&sentence).ok()?;
174
175        Some(Challenge {
176            exercise_id: exercise.id.clone(),
177            prompt: exercise.prompt.clone(),
178            sentence,
179            answer: AnswerType::Ambiguity { readings },
180            hint: exercise.hint.clone(),
181            explanation: exercise.explanation.clone(),
182        })
183    }
184
185    fn fill_template(&self, template: &str, constraints: &HashMap<String, Vec<String>>, rng: &mut impl Rng) -> Option<String> {
186        let mut result = template.to_string();
187        let mut used_names: HashMap<String, String> = HashMap::new();
188
189        while let Some(start) = result.find('{') {
190            let end = result[start..].find('}')? + start;
191            let slot = &result[start + 1..end];
192
193            let (slot_type, modifier) = if let Some(colon_pos) = slot.find(':') {
194                (&slot[..colon_pos], Some(&slot[colon_pos + 1..]))
195            } else {
196                (slot, None)
197            };
198
199            let slot_constraints = constraints.get(slot_type).map(|v| v.as_slice()).unwrap_or(&[]);
200            let word = self.fill_slot(slot_type, slot_constraints, modifier, &mut used_names, rng)?;
201
202            result = format!("{}{}{}", &result[..start], word, &result[end + 1..]);
203        }
204
205        Some(result)
206    }
207
208    fn fill_slot(
209        &self,
210        slot_type: &str,
211        constraints: &[String],
212        modifier: Option<&str>,
213        used_names: &mut HashMap<String, String>,
214        rng: &mut impl Rng,
215    ) -> Option<String> {
216        match slot_type {
217            "ProperName" => {
218                let key = format!("ProperName_{}", used_names.len());
219                if let Some(existing) = used_names.get(&key) {
220                    return Some(existing.clone());
221                }
222
223                let proper_nouns = self.lexicon.proper_nouns();
224                let available: Vec<_> = proper_nouns
225                    .iter()
226                    .filter(|n| !used_names.values().any(|v| v == &n.lemma))
227                    .copied()
228                    .collect();
229
230                let entry = if !available.is_empty() {
231                    available.choose(rng)?
232                } else {
233                    proper_nouns.choose(rng)?
234                };
235                let name = entry.lemma.clone();
236                used_names.insert(key, name.clone());
237                Some(name)
238            }
239            "Noun" => {
240                let nouns = if constraints.is_empty() {
241                    self.lexicon.common_nouns()
242                } else {
243                    let mut filtered = Vec::new();
244                    for constraint in constraints {
245                        filtered.extend(self.lexicon.nouns_with_feature(constraint));
246                    }
247                    filtered
248                };
249
250                let entry = nouns.choose(rng)?;
251                let word = entry.lemma.to_lowercase();
252
253                match modifier {
254                    Some("Plural") => Some(pluralize(entry)),
255                    _ => Some(word),
256                }
257            }
258            "Verb" => {
259                let verbs = if constraints.contains(&"Intransitive".to_string()) {
260                    self.lexicon.intransitive_verbs()
261                } else if constraints.contains(&"Transitive".to_string()) {
262                    self.lexicon.transitive_verbs()
263                } else {
264                    let mut result = Vec::new();
265                    for constraint in constraints {
266                        result.extend(self.lexicon.verbs_with_feature(constraint));
267                    }
268                    if result.is_empty() {
269                        self.lexicon.intransitive_verbs()
270                    } else {
271                        result
272                    }
273                };
274
275                let entry = verbs.choose(rng)?;
276
277                match modifier {
278                    Some("Past") => Some(past_tense(entry)),
279                    Some("Gerund") => Some(gerund(entry)),
280                    Some("Present3s") => Some(present_3s(entry)),
281                    _ => Some(entry.lemma.to_lowercase()),
282                }
283            }
284            "Adjective" => {
285                let adjectives = if constraints.contains(&"Intersective".to_string()) {
286                    self.lexicon.intersective_adjectives()
287                } else if constraints.is_empty() {
288                    self.lexicon.intersective_adjectives()
289                } else {
290                    let mut result = Vec::new();
291                    for constraint in constraints {
292                        result.extend(self.lexicon.adjectives_with_feature(constraint));
293                    }
294                    result
295                };
296
297                let entry = adjectives.choose(rng)?;
298                Some(entry.lemma.to_lowercase())
299            }
300            _ => Some("thing".to_string()),
301        }
302    }
303}

438    #[test]
439    fn test_all_exercises_across_all_modules() {
440        let engine = ContentEngine::new();
441        let generator = Generator::new();
442
443        let mut total_exercises = 0;
444        let mut successful = 0;
445        let mut failed_exercises = Vec::new();
446
447        for era in engine.eras() {
448            for module in &era.modules {
449                if let Some(m) = engine.get_module(&era.meta.id, &module.meta.id) {
450                    for (i, exercise) in m.exercises.iter().enumerate() {
451                        total_exercises += 1;
452                        let mut rng = StdRng::seed_from_u64(42 + i as u64);
453
454                        let challenge = generator.generate(exercise, &mut rng);
455                        if challenge.is_some() {
456                            successful += 1;
457                        } else {
458                            failed_exercises.push(format!("{}/{}/{}", era.meta.id, module.meta.id, exercise.id));
459                        }
460                    }
461                }
462            }
463        }
464
465        println!("Total exercises: {}", total_exercises);
466        println!("Successful: {}", successful);
467        println!("Failed: {}", failed_exercises.len());
468
469        if !failed_exercises.is_empty() {
470            println!("\nFailed exercises:");
471            for ex in &failed_exercises {
472                println!("  - {}", ex);
473            }
474        }
475
476        // Allow some failures for now, but ensure most work
477        let success_rate = successful as f64 / total_exercises as f64;
478        assert!(success_rate >= 0.8, "At least 80% of exercises should generate (got {:.1}%)", success_rate * 100.0);
479    }
