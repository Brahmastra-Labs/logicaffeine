1//! Responsive design system and mobile styling utilities.
2//!
3//! Centralizes all mobile/responsive concerns to ensure consistent behavior
4//! across device sizes. Provides breakpoints, touch target standards, and
5//! reusable CSS for mobile-first components.
6//!
7//! # Breakpoint Strategy
8//!
9//! Mobile-first approach with these breakpoints:
10//! - `XS` (480px): Small phones
11//! - `SM` (640px): Large phones, small tablets
12//! - `MD` (768px): Tablets (primary mobile breakpoint)
13//! - `LG` (1024px): Small laptops
14//! - `XL` (1280px): Desktops
15//!
16//! # Touch Target Standards
17//!
18//! Follows WCAG 2.5 guidelines:
19//! - Minimum: 44x44px
20//! - Comfortable: 48x48px
21//! - Large (primary actions): 56x56px
22//!
23//! # Usage
24//!
25//! Include base styles in your root component:
26//!
27//! ```ignore
28//! use crate::ui::responsive::{MOBILE_BASE_STYLES, all_mobile_styles};
29//!
30//! rsx! {
31//!     // Option 1: Just the base utilities
32//!     style { "{MOBILE_BASE_STYLES}" }
33//!
34//!     // Option 2: All mobile styles including tabs, panels, buttons
35//!     style { "{all_mobile_styles()}" }
36//! }
37//! ```
38//!
39//! # CSS Classes
40//!
41//! | Class | Description |
42//! |-------|-------------|
43//! | `.desktop-only` | Hidden on mobile (â‰¤768px) |
44//! | `.mobile-only` | Hidden on desktop (>768px) |
45//! | `.touch-target` | Minimum 44x44px touch area |
46//! | `.safe-top` | Respects notch/status bar |
47//! | `.safe-bottom` | Respects home indicator |
48//! | `.scroll-smooth` | iOS momentum scrolling |

512/// All mobile styles combined - include this for a complete mobile solution
513pub fn all_mobile_styles() -> String {
514    format!(
515        "{}\n{}\n{}\n{}\n{}\n{}",
516        MOBILE_BASE_STYLES,
517        MOBILE_TAB_BAR_STYLES,
518        MOBILE_PANEL_STYLES,
519        MOBILE_BUTTON_STYLES,
520        MOBILE_INPUT_STYLES,
521        MOBILE_RESIZER_STYLES,
522    )
523}

525/// Generate a complete mobile-ready style block for a page
526/// This combines the base mobile utilities with any page-specific styles
527pub fn with_mobile_styles(page_styles: &str) -> String {
528    format!("{}\n{}", MOBILE_BASE_STYLES, page_styles)
