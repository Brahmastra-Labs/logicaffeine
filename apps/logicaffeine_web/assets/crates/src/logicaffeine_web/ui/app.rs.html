385/// Root application component.
386///
387/// Sets up global context providers for license and registry authentication,
388/// triggers license revalidation on mount, and renders the router.
389///
390/// # Context Providers
391///
392/// - [`LicenseState`] - Manages subscription validation
393/// - [`RegistryAuthState`] - Manages GitHub OAuth for package registry
394pub fn App() -> Element {
395    let license_state = use_context_provider(LicenseState::new);
396    let _registry_auth = use_context_provider(RegistryAuthState::new);
397
398    use_effect(move || {
399        let mut license_state = license_state.clone();
400        spawn(async move {
401            if license_state.has_license() && license_state.needs_revalidation() {
402                license_state.validate().await;
403            }
404        });
405    });
406
407    rsx! {
408        style { "{GLOBAL_STYLE}" }
409        Router::<Route> {}
410    }
411}
