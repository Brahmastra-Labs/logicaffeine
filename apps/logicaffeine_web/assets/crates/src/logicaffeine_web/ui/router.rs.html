1//! Application routing and navigation.
2//!
3//! Defines all routes for the Logicaffeine web application using the Dioxus
4//! router. Each route maps to a page component in [`crate::ui::pages`].
5//!
6//! # Routes
7//!
8//! | Path | Component | Description |
9//! |------|-----------|-------------|
10//! | `/` | [`Landing`] | Marketing homepage |
11//! | `/learn` | [`Learn`] | Main learning interface with curriculum |
12//! | `/studio` | [`Studio`] | Playground for experimentation |
13//! | `/profile` | [`Profile`] | User settings and progress |
14//! | `/pricing` | [`Pricing`] | Subscription plans |
15//! | `/guide` | [`Guide`] | Documentation and tutorials |
16//! | `/registry` | [`Registry`] | Package browser |
17//! | `/workspace/:subject` | [`Workspace`] | Subject-specific workspace |
18//!
19//! # Navigation
20//!
21//! Use the Dioxus `Link` component with `Route` variants:
22//!
23//! ```ignore
24//! use crate::ui::router::Route;
25//!
26//! rsx! {
27//!     Link { to: Route::Learn {}, "Start Learning" }
28//!     Link { to: Route::Studio {}, "Open Studio" }
29//! }
30//! ```
31
32use dioxus::prelude::*;
33use crate::ui::pages::{Landing, Learn, Pricing, Privacy, Profile, Roadmap, Success, Terms, Workspace, Studio, Guide, Crates};
34use crate::ui::pages::registry::{Registry, PackageDetail};
35
36/// Application routes.
37///
38/// This enum is derived with `Routable` to generate route matching logic.
39/// Each variant corresponds to a URL pattern and its associated page component.
40#[derive(Clone, Routable, Debug, PartialEq)]
41pub enum Route {
42    /// Marketing homepage at `/`.
43    #[route("/")]
44    Landing {},
45
46    /// Subscription plans and pricing at `/pricing`.
47    #[route("/pricing")]
48    Pricing {},
49
50    /// Privacy policy at `/privacy`.
51    #[route("/privacy")]
52    Privacy {},
53
54    /// Terms of service at `/terms`.
55    #[route("/terms")]
56    Terms {},
57
58    /// Product roadmap at `/roadmap`.
59    #[route("/roadmap")]
60    Roadmap {},
61
62    /// Documentation and tutorials at `/guide`.
63    #[route("/guide")]
64    Guide {},
65
66    /// Crate documentation at `/crates`.
67    #[route("/crates")]
68    Crates {},
69
70    /// Post-checkout success page at `/success`.
71    #[route("/success")]
72    Success {},
73
74    /// Playground for experimentation at `/studio`.
75    #[route("/studio")]
76    Studio {},
77
78    /// Main learning interface at `/learn`.
79    ///
80    /// All learning happens here - curriculum browsing, exercises, and review.
81    #[route("/learn")]
82    Learn {},
83
84    /// User profile and settings at `/profile`.
85    #[route("/profile")]
86    Profile {},
87
88    /// Subject-specific workspace at `/workspace/:subject`.
89    #[route("/workspace/:subject")]
90    Workspace {
91        /// The workspace subject identifier (e.g., "logic", "proofs").
92        subject: String,
93    },
94
95    /// Package registry browser at `/registry`.
96    #[route("/registry")]
97    Registry {},
98
99    /// Package detail page at `/registry/package/:name`.
100    #[route("/registry/package/:name")]
101    PackageDetail {
102        /// The package name to display.
103        name: String,
104    },
105
106    /// Catch-all for unknown routes, renders 404 page.
107    #[route("/:..route")]
108    NotFound {
109        /// Path segments that didn't match any route.
110        route: Vec<String>,
111    },
112}
113
114#[component]
115fn NotFound(route: Vec<String>) -> Element {
116    rsx! {
117        div {
118            style: "min-height: 100vh; display: flex; flex-direction: column; align-items: center; justify-content: center; background: linear-gradient(135deg, #1a1a2e 0%, #16213e 50%, #0f3460 100%); color: #e8e8e8;",
119            h1 { style: "font-size: 48px; margin-bottom: 16px;", "404" }
120            p { style: "color: #888; margin-bottom: 24px;", "Page not found: /{route.join(\"/\")}" }
121            Link {
122                to: Route::Landing {},
123                style: "padding: 12px 24px; background: linear-gradient(135deg, #667eea, #764ba2); border-radius: 8px; color: white; text-decoration: none;",
124                "Go Home"
125            }
126        }
127    }
128}
