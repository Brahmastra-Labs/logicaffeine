262pub fn Crates() -> Element {
263    rsx! {
264        style { "{CRATES_STYLE}" }
265
266        div { class: "crates-page",
267            MainNav {
268                active: ActivePage::Crates,
269                subtitle: Some("Crate Documentation"),
270            }
271
272            header { class: "crates-hero",
273                div { class: "crates-hero-badge",
274                    div { class: "dot" }
275                    span { "Rustdoc" }
276                }
277                h1 { "Crate Documentation" }
278                p {
279                    "Explore the API documentation for all LOGICAFFEINE crates. Each crate is documented with examples and detailed type information."
280                }
281            }
282
283            main { class: "crates-content",
284                section { class: "crates-section",
285                    h2 { class: "crates-section-title", "Core Crates" }
286                    div { class: "crates-grid",
287                        for crate_info in CORE_CRATES.iter() {
288                            CrateCard { info: crate_info }
289                        }
290                    }
291                }
292
293                section { class: "crates-section",
294                    h2 { class: "crates-section-title", "Application Crates" }
295                    div { class: "crates-grid",
296                        for crate_info in APP_CRATES.iter() {
297                            CrateCard { info: crate_info }
298                        }
299                    }
300                }
301            }
302        }
303    }
304}

306#[component]
307fn CrateCard(info: &'static CrateInfo) -> Element {
308    let doc_url = format!("/assets/crates/{}/index.html", info.name);
309
310    rsx! {
311        a {
312            class: "crate-card",
313            href: "{doc_url}",
314            div { class: "crate-header",
315                h3 { class: "crate-name",
316                    "{info.name}"
317                }
318                span {
319                    class: if info.is_core { "crate-badge core" } else { "crate-badge app" },
320                    if info.is_core { "Core" } else { "App" }
321                }
322            }
323            p { class: "crate-description", "{info.description}" }
324            div { class: "crate-arrow",
325                "View docs"
326                svg {
327                    xmlns: "http://www.w3.org/2000/svg",
328                    fill: "none",
329                    view_box: "0 0 24 24",
330                    stroke: "currentColor",
331                    stroke_width: "2",
332                    path {
333                        stroke_linecap: "round",
334                        stroke_linejoin: "round",
335                        d: "M13 7l5 5m0 0l-5 5m5-5H6"
336                    }
337                }
338            }
339        }
340    }
