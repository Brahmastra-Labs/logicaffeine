208const STRIPE_CUSTOMER_PORTAL: &str = "https://billing.stripe.com/p/login/8x200l3VN98D7qa1SMe3e00";

210fn get_session_id_from_url() -> Option<String> {
211    let window = web_sys::window()?;
212    let location = window.location();
213    let search = location.search().ok()?;
214
215    let params = web_sys::UrlSearchParams::new_with_str(&search).ok()?;
216    params.get("session_id")
217}
218
219fn save_license_to_storage(license_key: &str, plan: &str) {
220    if let Some(window) = web_sys::window() {
221        if let Ok(Some(storage)) = window.local_storage() {
222            let _ = storage.set_item("logos_license_key", license_key);
223            let _ = storage.set_item("logos_license_plan", plan);
224            let timestamp = js_sys::Date::now().to_string();
225            let _ = storage.set_item("logos_license_validated_at", &timestamp);
226        }
227    }
228}
229
230fn copy_to_clipboard(text: &str) {
231    if let Some(window) = web_sys::window() {
232        let clipboard = window.navigator().clipboard();
233        let _ = clipboard.write_text(text);
234    }
235}

237#[derive(Clone, PartialEq)]
238enum LicenseStatus {
239    Loading,
240    Success { subscription_id: String, plan: String },
241    Error(String),
242    NoSession,
243}
244
245async fn fetch_license_from_session(session_id: String) -> LicenseStatus {
246    use gloo_net::http::Request;
247
248    let body = serde_json::json!({ "sessionId": session_id });
249
250    let response = Request::post(LICENSE_API_URL)
251        .header("Content-Type", "application/json")
252        .body(body.to_string())
253        .unwrap()
254        .send()
255        .await;
256
257    match response {
258        Ok(resp) => {
259            if resp.ok() {
260                match resp.json::<serde_json::Value>().await {
261                    Ok(data) => {
262                        let subscription_id = data["subscriptionId"]
263                            .as_str()
264                            .unwrap_or("")
265                            .to_string();
266                        let plan = data["plan"]
267                            .as_str()
268                            .unwrap_or("unknown")
269                            .to_string();
270                        LicenseStatus::Success { subscription_id, plan }
271                    }
272                    Err(_) => LicenseStatus::Error("Failed to parse response".to_string()),
273                }
274            } else {
275                LicenseStatus::Error("License lookup failed".to_string())
276            }
277        }
278        Err(e) => LicenseStatus::Error(format!("Network error: {}", e)),
279    }
280}

282#[component]
283pub fn Success() -> Element {
284    let mut license_status = use_signal(|| LicenseStatus::Loading);
285    let mut copied = use_signal(|| false);
286    let mut saved = use_signal(|| false);
287    let license_state = use_context::<LicenseState>();
288
289    use_effect(move || {
290        let mut license_state = license_state.clone();
291        spawn(async move {
292            if let Some(session_id) = get_session_id_from_url() {
293                let result = fetch_license_from_session(session_id).await;
294                if let LicenseStatus::Success { ref subscription_id, ref plan } = result {
295                    save_license_to_storage(subscription_id, plan);
296                    license_state.set_license(
297                        subscription_id.clone(),
298                        LicensePlan::from_str(plan),
299                    );
300                    saved.set(true);
301                }
302                license_status.set(result);
303            } else {
304                license_status.set(LicenseStatus::NoSession);
305            }
306        });
307    });
308
309    let on_copy = move |_| {
310        if let LicenseStatus::Success { ref subscription_id, .. } = *license_status.read() {
311            copy_to_clipboard(subscription_id);
312            copied.set(true);
313        }
314    };
315
316    let (has_license, license_key) = match &*license_status.read() {
317        LicenseStatus::Success { subscription_id, .. } => (true, subscription_id.clone()),
318        _ => (false, String::new()),
319    };
320
321    let is_loading = matches!(*license_status.read(), LicenseStatus::Loading);
322
323    rsx! {
324        style { "{SUCCESS_STYLE}" }
325
326        MainNav { active: ActivePage::Pricing, subtitle: Some("Payment Complete"), show_nav_links: false }
327
328        div { class: "success-container",
329            div { class: "success-icon", "✓" }
330
331            h1 { class: "success-title", "Thank You!" }
332
333            p { class: "success-message",
334                "Your payment was successful. Welcome to logicaffeine! "
335                "You now have access to use LOGOS for commercial purposes."
336            }
337
338            if is_loading {
339                div { class: "loading-spinner" }
340                p { class: "success-message", "Retrieving your license..." }
341            }
342
343            match &*license_status.read() {
344                LicenseStatus::Error(msg) => rsx! {
345                    div { class: "error-message", "{msg}" }
346                },
347                LicenseStatus::NoSession => rsx! {
348                    div { class: "error-message", "No checkout session found. Please try again." }
349                },
350                _ => rsx! {}
351            }
352
353            if has_license {
354                div { class: "license-box",
355                    div { class: "license-label", "Your License Key" }
356                    div { class: "license-key", "{license_key}" }
357                    button {
358                        class: "copy-btn",
359                        onclick: on_copy,
360                        if *copied.read() { "Copied!" } else { "Copy to Clipboard" }
361                    }
362                    if *saved.read() {
363                        div { class: "license-saved",
364                            "✓ License saved to your browser"
365                        }
366                    }
367                }
368            }
369
370            div { class: "success-actions",
371                Link {
372                    class: "btn-primary",
373                    to: Route::Studio {},
374                    "Open Studio"
375                }
376
377                a {
378                    class: "btn-secondary",
379                    href: STRIPE_CUSTOMER_PORTAL,
380                    target: "_blank",
381                    "Manage Subscription"
382                }
383            }
384
385            div { class: "success-note",
386                p {
387                    "Save your license key somewhere safe. "
388                    "Need help? Contact us at "
389                    a { href: "mailto:tristen@brahmastra-labs.com", "tristen@brahmastra-labs.com" }
390                    "."
391                }
392            }
393        }
394    }
