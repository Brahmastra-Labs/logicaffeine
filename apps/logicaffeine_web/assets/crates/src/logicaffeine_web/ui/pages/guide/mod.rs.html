255#[component]
256pub fn Guide() -> Element {
257    let mut active_section = use_signal(|| "introduction".to_string());
258
259    // Build section info for sidebar
260    let sections_info: Vec<SectionInfo> = SECTIONS.iter().map(|s| SectionInfo {
261        id: s.id.to_string(),
262        number: s.number,
263        title: s.title.to_string(),
264        part: s.part.to_string(),
265    }).collect();
266
267    // Collect all section IDs for intersection observer
268    #[allow(unused_variables)]
269    let section_ids: Vec<String> = SECTIONS.iter().map(|s| s.id.to_string()).collect();
270
271    // Set up scroll tracking with IntersectionObserver
272    #[cfg(target_arch = "wasm32")]
273    {
274        use wasm_bindgen::prelude::*;
275        use wasm_bindgen::JsCast;
276
277        let section_ids_for_effect = section_ids.clone();
278
279        use_effect(move || {
280            let window = match web_sys::window() {
281                Some(w) => w,
282                None => return,
283            };
284            let document = match window.document() {
285                Some(d) => d,
286                None => return,
287            };
288
289            // Use RefCell to allow mutation from within Fn closure
290            use std::cell::RefCell;
291            use std::rc::Rc;
292
293            let active_section_clone = Rc::new(RefCell::new(active_section.clone()));
294            let active_section_for_closure = active_section_clone.clone();
295
296            let callback = Closure::<dyn Fn(js_sys::Array, web_sys::IntersectionObserver)>::new(
297                move |entries: js_sys::Array, _observer: web_sys::IntersectionObserver| {
298                    // Simple approach: when a section crosses the threshold line,
299                    // it becomes active
300                    for i in 0..entries.length() {
301                        if let Ok(entry) = entries.get(i).dyn_into::<web_sys::IntersectionObserverEntry>() {
302                            if entry.is_intersecting() {
303                                let target = entry.target();
304                                let id = target.id();
305                                if !id.is_empty() {
306                                    active_section_for_closure.borrow_mut().set(id);
307                                }
308                            }
309                        }
310                    }
311                },
312            );
313
314            // Create IntersectionObserver options
315            let mut options = web_sys::IntersectionObserverInit::new();
316            // Root margin creates a thin "tripwire" near the top of the screen
317            options.root_margin("-100px 0px -90% 0px");
318            let thresholds = js_sys::Array::new();
319            thresholds.push(&JsValue::from(0.0));
320            options.threshold(&thresholds);
321
322            // Create the observer
323            let observer = match web_sys::IntersectionObserver::new_with_options(
324                callback.as_ref().unchecked_ref(),
325                &options,
326            ) {
327                Ok(obs) => obs,
328                Err(_) => return,
329            };
330
331            // Observe all sections
332            for section_id in &section_ids_for_effect {
333                if let Some(element) = document.get_element_by_id(section_id) {
334                    observer.observe(&element);
335                }
336            }
337
338            // Keep callback alive
339            callback.forget();
340        });
341    }
342
343    // Track current part for dividers
344    let mut current_part = String::new();
345
346    rsx! {
347        style { "{GUIDE_STYLE}" }
348
349        div { class: "guide-page",
350            // Navigation
351            MainNav {
352                active: ActivePage::Guide,
353                subtitle: Some("Programmer's Guide"),
354            }
355
356            // Hero
357            header { class: "guide-hero",
358                div { class: "guide-hero-badge",
359                    div { class: "dot" }
360                    span { "Interactive Guide" }
361                }
362                h1 { "LOGOS Language Guide" }
363                p {
364                    "Write English. Get Logic. Run Code. A comprehensive guide to programming in LOGOS, from basics to advanced features."
365                }
366            }
367
368            // Main layout
369            div { class: "guide-layout",
370                // Sidebar
371                GuideSidebar {
372                    sections: sections_info,
373                    active_section: active_section.read().clone(),
374                    on_section_click: move |id: String| {
375                        active_section.set(id);
376                    },
377                }
378
379                // Content
380                main { class: "guide-content",
381                    for section in SECTIONS.iter() {
382                        {
383                            // Check if we need a part divider
384                            let show_divider = section.part != current_part && section.number > 1;
385                            current_part = section.part.to_string();
386
387                            rsx! {
388                                // Part divider
389                                if show_divider {
390                                    div { class: "guide-part-divider",
391                                        h2 { "{section.part}" }
392                                    }
393                                }
394
395                                // Section
396                                section {
397                                    id: "{section.id}",
398                                    class: "guide-section",
399
400                                    h2 {
401                                        span { class: "section-number", "{section.number}." }
402                                        "{section.title}"
403                                    }
404
405                                    // Render content as HTML
406                                    div {
407                                        dangerous_inner_html: render_markdown(section.content)
408                                    }
409
410                                    // Render code examples
411                                    if !section.examples.is_empty() {
412                                        div { class: "guide-examples",
413                                            for example in section.examples.iter() {
414                                                GuideCodeBlock {
415                                                    id: example.id.to_string(),
416                                                    label: example.label.to_string(),
417                                                    mode: example.mode,
418                                                    initial_code: example.code.to_string(),
419                                                }
420                                            }
421                                        }
422                                    }
423                                }
424                            }
425                        }
426                    }
427                }
428            }
429        }
430    }
431}

433/// Simple markdown to HTML converter
434/// Handles: headers, paragraphs, lists, tables, inline code, bold
435fn render_markdown(content: &str) -> String {
436    let mut html = String::new();
437    let mut in_list = false;
438    let mut in_table = false;
439    let mut in_table_header = false;
440
441    for line in content.lines() {
442        let trimmed = line.trim();
443
444        // Skip empty lines
445        if trimmed.is_empty() {
446            if in_list {
447                html.push_str("</ul>");
448                in_list = false;
449            }
450            if in_table {
451                html.push_str("</tbody></table>");
452                in_table = false;
453            }
454            continue;
455        }
456
457        // Headers
458        if trimmed.starts_with("### ") {
459            if in_list { html.push_str("</ul>"); in_list = false; }
460            if in_table { html.push_str("</tbody></table>"); in_table = false; }
461            html.push_str(&format!("<h3>{}</h3>", inline_markdown(&trimmed[4..])));
462            continue;
463        }
464
465        // Table row
466        if trimmed.starts_with('|') && trimmed.ends_with('|') {
467            // Check if this is a separator row (|---|---|)
468            if trimmed.contains("---") {
469                in_table_header = false;
470                continue;
471            }
472
473            if !in_table {
474                html.push_str("<table><thead>");
475                in_table = true;
476                in_table_header = true;
477            }
478
479            let cells: Vec<&str> = trimmed[1..trimmed.len()-1]
480                .split('|')
481                .map(|s| s.trim())
482                .collect();
483
484            if in_table_header {
485                html.push_str("<tr>");
486                for cell in &cells {
487                    html.push_str(&format!("<th>{}</th>", inline_markdown(cell)));
488                }
489                html.push_str("</tr></thead><tbody>");
490            } else {
491                html.push_str("<tr>");
492                for cell in &cells {
493                    html.push_str(&format!("<td>{}</td>", inline_markdown(cell)));
494                }
495                html.push_str("</tr>");
496            }
497            continue;
498        }
499
500        // Close table if not a table row
501        if in_table && !trimmed.starts_with('|') {
502            html.push_str("</tbody></table>");
503            in_table = false;
504        }
505
506        // List items
507        if trimmed.starts_with("- ") || trimmed.starts_with("* ") {
508            if !in_list {
509                html.push_str("<ul>");
510                in_list = true;
511            }
512            html.push_str(&format!("<li>{}</li>", inline_markdown(&trimmed[2..])));
513            continue;
514        }
515
516        // Numbered list
517        if trimmed.chars().next().map_or(false, |c| c.is_ascii_digit()) {
518            if let Some(dot_pos) = trimmed.find(". ") {
519                if !in_list {
520                    html.push_str("<ul>");
521                    in_list = true;
522                }
523                html.push_str(&format!("<li>{}</li>", inline_markdown(&trimmed[dot_pos + 2..])));
524                continue;
525            }
526        }
527
528        // Close list if not a list item
529        if in_list {
530            html.push_str("</ul>");
531            in_list = false;
532        }
533
534        // Paragraph
535        html.push_str(&format!("<p>{}</p>", inline_markdown(trimmed)));
536    }
537
538    // Close any open tags
539    if in_list {
540        html.push_str("</ul>");
541    }
542    if in_table {
543        html.push_str("</tbody></table>");
544    }
545
546    html
547}

549/// Process inline markdown: **bold**, `code`, [links]
550fn inline_markdown(text: &str) -> String {
551    let mut result = text.to_string();
552
553    // Escape HTML entities
554    result = result.replace('&', "&amp;");
555    result = result.replace('<', "&lt;");
556    result = result.replace('>', "&gt;");
557
558    // Bold: **text**
559    while let Some(start) = result.find("**") {
560        if let Some(end) = result[start + 2..].find("**") {
561            let before = &result[..start];
562            let inner = &result[start + 2..start + 2 + end];
563            let after = &result[start + 2 + end + 2..];
564            result = format!("{}<strong>{}</strong>{}", before, inner, after);
565        } else {
566            break;
567        }
568    }
569
570    // Inline code: `code`
571    while let Some(start) = result.find('`') {
572        if let Some(end) = result[start + 1..].find('`') {
573            let before = &result[..start];
574            let inner = &result[start + 1..start + 1 + end];
575            let after = &result[start + 1 + end + 1..];
576            result = format!("{}<code>{}</code>{}", before, inner, after);
577        } else {
578            break;
579        }
580    }
581
582    result
