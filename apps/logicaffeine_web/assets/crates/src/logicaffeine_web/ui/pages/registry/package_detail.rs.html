291#[component]
292pub fn PackageDetail(name: String) -> Element {
293    let mut details = use_signal(|| None::<PackageDetails>);
294    let mut error = use_signal(|| None::<String>);
295    let mut is_loading = use_signal(|| true);
296    let mut active_tab = use_signal(|| DetailTab::Readme);
297
298    // Fetch package details
299    use_effect({
300        let name = name.clone();
301        move || {
302            let name = name.clone();
303            spawn(async move {
304                is_loading.set(true);
305                match fetch_package_details(&name).await {
306                    Ok(d) => details.set(Some(d)),
307                    Err(e) => error.set(Some(e)),
308                }
309                is_loading.set(false);
310            });
311        }
312    });
313
314    rsx! {
315        style { "{DETAIL_STYLE}" }
316
317        MainNav { active: ActivePage::Registry, subtitle: Some("Package Details") }
318
319        div { class: "detail-container",
320            if *is_loading.read() {
321                div { class: "loading-spinner", "Loading package..." }
322            } else if let Some(err) = error.read().as_ref() {
323                div { class: "error-message",
324                    p { "Error: {err}" }
325                }
326            } else if let Some(pkg) = details.read().as_ref() {
327                // Header
328                header { class: "detail-header",
329                    div { class: "package-title",
330                        h1 { "{pkg.name}" }
331                        if pkg.verified {
332                            span { class: "verified-badge", "Official" }
333                        }
334                    }
335                    div { class: "package-meta",
336                        span { "by {pkg.owner}" }
337                        if let Some(repo) = &pkg.repository {
338                            a { href: "{repo}", target: "_blank", "Repository" }
339                        }
340                        if let Some(license) = &pkg.license {
341                            span { "{license}" }
342                        }
343                    }
344                }
345
346                // Content
347                div { class: "detail-content",
348                    main { class: "main-content",
349                        nav { class: "tab-nav",
350                            button {
351                                class: if *active_tab.read() == DetailTab::Readme { "tab-btn active" } else { "tab-btn" },
352                                onclick: move |_| active_tab.set(DetailTab::Readme),
353                                "README"
354                            }
355                            button {
356                                class: if *active_tab.read() == DetailTab::Versions { "tab-btn active" } else { "tab-btn" },
357                                onclick: move |_| active_tab.set(DetailTab::Versions),
358                                "Versions ({pkg.versions.len()})"
359                            }
360                        }
361
362                        match *active_tab.read() {
363                            DetailTab::Readme => rsx! {
364                                div { class: "readme-content",
365                                    if let Some(readme) = &pkg.readme {
366                                        // Note: In production, render markdown properly
367                                        pre { "{readme}" }
368                                    } else {
369                                        p { "No README available." }
370                                    }
371                                }
372                            },
373                            DetailTab::Versions => rsx! {
374                                div { class: "versions-list",
375                                    for version in pkg.versions.iter() {
376                                        div { class: "version-item",
377                                            span { class: "version-name",
378                                                "v{version.version}"
379                                                if version.yanked {
380                                                    span { class: "yanked-badge", "yanked" }
381                                                }
382                                            }
383                                            div { class: "version-meta",
384                                                span { "{format_size(version.size)}" }
385                                                span { "{version.published_at}" }
386                                            }
387                                        }
388                                    }
389                                }
390                            },
391                        }
392                    }
393
394                    aside { class: "sidebar",
395                        div { class: "sidebar-section",
396                            h3 { "Install" }
397                            code { class: "install-cmd", "largo add {pkg.name}" }
398                        }
399
400                        div { class: "sidebar-section",
401                            h3 { "Details" }
402                            div { class: "meta-item",
403                                span { class: "meta-label", "Downloads" }
404                                span { class: "meta-value", "{pkg.downloads}" }
405                            }
406                            if !pkg.versions.is_empty() {
407                                div { class: "meta-item",
408                                    span { class: "meta-label", "Latest" }
409                                    span { class: "meta-value", "v{pkg.versions[0].version}" }
410                                }
411                            }
412                            if let Some(homepage) = &pkg.homepage {
413                                div { class: "meta-item",
414                                    span { class: "meta-label", "Homepage" }
415                                    span { class: "meta-value",
416                                        a { href: "{homepage}", target: "_blank", "Link" }
417                                    }
418                                }
419                            }
420                        }
421
422                        if !pkg.keywords.is_empty() {
423                            div { class: "sidebar-section",
424                                h3 { "Keywords" }
425                                div { class: "keywords-list",
426                                    for keyword in pkg.keywords.iter() {
427                                        span { class: "keyword-tag", "{keyword}" }
428                                    }
429                                }
430                            }
431                        }
432                    }
433                }
434            }
435        }
436    }
437}

439fn format_size(bytes: u64) -> String {
440    if bytes < 1024 {
441        format!("{} B", bytes)
442    } else if bytes < 1024 * 1024 {
443        format!("{:.1} KB", bytes as f64 / 1024.0)
444    } else {
445        format!("{:.1} MB", bytes as f64 / (1024.0 * 1024.0))
446    }
447}

449async fn fetch_package_details(name: &str) -> Result<PackageDetails, String> {
450    #[cfg(target_arch = "wasm32")]
451    {
452        use gloo_net::http::Request;
453
454        let url = format!("{}/packages/{}", REGISTRY_API_URL, name);
455
456        let response = Request::get(&url)
457            .send()
458            .await
459            .map_err(|e| e.to_string())?;
460
461        if !response.ok() {
462            return Err("Package not found".to_string());
463        }
464
465        response.json().await.map_err(|e| e.to_string())
466    }
467
468    #[cfg(not(target_arch = "wasm32"))]
469    {
470        Err("Not available in non-WASM builds".to_string())
471    }
