330/// Props for VocabReference
331#[derive(Props, Clone, PartialEq)]
332pub struct VocabReferenceProps {
333    /// Whether to show the panel initially
334    #[props(default = false)]
335    pub initial_open: bool,
336}
337
338/// Vocabulary Reference floating panel
339#[component]
340pub fn VocabReference(props: VocabReferenceProps) -> Element {
341    let mut is_open = use_signal(|| props.initial_open);
342    let mut search_query = use_signal(|| String::new());
343    let mut show_attention = use_signal(|| false);
344
345    // Trigger attention animation after 10 seconds if not opened
346    use_effect(move || {
347        spawn(async move {
348            // Wait 10 seconds
349            gloo_timers::future::TimeoutFuture::new(10_000).await;
350            // Only show attention if panel hasn't been opened
351            if !*is_open.peek() {
352                show_attention.set(true);
353            }
354        });
355    });
356
357    let symbols = get_symbols();
358    let vocab_terms = get_vocab_terms();
359
360    // Filter based on search
361    let query = search_query.read().to_lowercase();
362    let filtered_symbols: Vec<_> = if query.is_empty() {
363        symbols.iter().collect()
364    } else {
365        symbols.iter().filter(|s| {
366            s.symbol.to_lowercase().contains(&query) ||
367            s.name.to_lowercase().contains(&query) ||
368            s.meaning.to_lowercase().contains(&query)
369        }).collect()
370    };
371
372    let filtered_terms: Vec<_> = if query.is_empty() {
373        vocab_terms.iter().collect()
374    } else {
375        vocab_terms.iter().filter(|t| {
376            t.term.to_lowercase().contains(&query) ||
377            t.definition.to_lowercase().contains(&query)
378        }).collect()
379    };
380
381    // Group symbols by category
382    let quantifiers: Vec<_> = filtered_symbols.iter().filter(|s| s.category == "quantifier").collect();
383    let connectives: Vec<_> = filtered_symbols.iter().filter(|s| s.category == "connective").collect();
384    let predicates: Vec<_> = filtered_symbols.iter().filter(|s| s.category == "predicate").collect();
385
386    rsx! {
387        style { "{VOCAB_REFERENCE_STYLE}" }
388
389        // Toggle button
390        button {
391            class: {
392                let open = *is_open.read();
393                let attention = *show_attention.read();
394                if open {
395                    "vocab-reference-toggle active"
396                } else if attention {
397                    "vocab-reference-toggle attention"
398                } else {
399                    "vocab-reference-toggle"
400                }
401            },
402            onclick: move |_| {
403                let current = *is_open.read();
404                is_open.set(!current);
405                // Stop attention animation once clicked
406                show_attention.set(false);
407            },
408            title: "Symbol & Vocabulary Reference",
409            if *is_open.read() { "Ã—" } else { "ðŸ“–" }
410        }
411
412        // Panel
413        if *is_open.read() {
414            div { class: "vocab-reference-panel",
415                // Header
416                div { class: "vocab-panel-header",
417                    div { class: "vocab-panel-title",
418                        "ðŸ“– Reference"
419                    }
420                    button {
421                        class: "vocab-panel-close",
422                        onclick: move |_| is_open.set(false),
423                        "Ã—"
424                    }
425                }
426
427                // Content
428                div { class: "vocab-panel-content",
429                    // Search
430                    div { class: "vocab-search",
431                        input {
432                            class: "vocab-search-input",
433                            r#type: "text",
434                            placeholder: "Search symbols or terms...",
435                            value: "{search_query}",
436                            oninput: move |e| search_query.set(e.value()),
437                        }
438                    }
439
440                    // Quantifiers
441                    if !quantifiers.is_empty() {
442                        div { class: "vocab-section",
443                            div { class: "vocab-section-title quantifiers", "Quantifiers" }
444                            div { class: "vocab-items",
445                                for sym in quantifiers {
446                                    div { class: "vocab-item",
447                                        span { class: "vocab-symbol quantifier", "{sym.symbol}" }
448                                        div { class: "vocab-info",
449                                            div { class: "vocab-name", "{sym.name}" }
450                                            div { class: "vocab-meaning", "{sym.meaning}" }
451                                        }
452                                    }
453                                }
454                            }
455                        }
456                    }
457
458                    // Connectives
459                    if !connectives.is_empty() {
460                        div { class: "vocab-section",
461                            div { class: "vocab-section-title connectives", "Connectives" }
462                            div { class: "vocab-items",
463                                for sym in connectives {
464                                    div { class: "vocab-item",
465                                        span { class: "vocab-symbol connective", "{sym.symbol}" }
466                                        div { class: "vocab-info",
467                                            div { class: "vocab-name", "{sym.name}" }
468                                            div { class: "vocab-meaning", "{sym.meaning}" }
469                                        }
470                                    }
471                                }
472                            }
473                        }
474                    }
475
476                    // Predicates
477                    if !predicates.is_empty() {
478                        div { class: "vocab-section",
479                            div { class: "vocab-section-title predicates", "Predicates & Relations" }
480                            div { class: "vocab-items",
481                                for sym in predicates {
482                                    div { class: "vocab-item",
483                                        span { class: "vocab-symbol predicate", "{sym.symbol}" }
484                                        div { class: "vocab-info",
485                                            div { class: "vocab-name", "{sym.name}" }
486                                            div { class: "vocab-meaning", "{sym.meaning}" }
487                                        }
488                                    }
489                                }
490                            }
491                        }
492                    }
493
494                    // Vocabulary Terms
495                    if !filtered_terms.is_empty() {
496                        div { class: "vocab-section",
497                            div { class: "vocab-section-title terms", "Key Terms" }
498                            for term in filtered_terms {
499                                div { class: "vocab-term-item",
500                                    div { class: "vocab-term-name", "{term.term}" }
501                                    div { class: "vocab-term-def", "{term.definition}" }
502                                }
503                            }
504                        }
505                    }
506                }
507            }
508        }
509    }
