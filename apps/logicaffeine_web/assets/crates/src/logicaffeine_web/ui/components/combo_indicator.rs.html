3//! Displays the current combo count with animated flames and multiplier.

7//!

87#[component]
88pub fn ComboIndicator(combo: u32, multiplier: f64, is_new_record: bool) -> Element {
89    let mut show_record = use_signal(|| false);
90    let mut last_record_combo = use_signal(|| 0u32);
91
92    if is_new_record && combo > last_record_combo() {
93        show_record.set(true);
94        last_record_combo.set(combo);
95
96        spawn(async move {
97            gloo_timers::future::TimeoutFuture::new(1500).await;
98            show_record.set(false);
99        });
100    }
101
102    if combo == 0 {
103        return rsx! {};
104    }
105
106    let flame_count = (combo.min(5)) as usize;
107
108    rsx! {
109        style { "{COMBO_STYLE}" }
110        div { class: "combo-wrapper",
111            if show_record() {
112                div { class: "combo-record", "NEW RECORD!" }
113            }
114            div { class: "combo-indicator",
115                div { class: "combo-flames",
116                    for _ in 0..flame_count {
117                        span { class: "flame", "ðŸ”¥" }
118                    }
119                }
120                span { class: "combo-count", "{combo}x" }
121                span { class: "combo-multiplier", "({multiplier:.1}x)" }
122            }
123        }
124    }
125}
