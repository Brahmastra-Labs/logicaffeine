1//! Multi-Word Expression (MWE) processing
2//!
3//! Post-tokenization pipeline that collapses multi-token sequences
4//! into single semantic units (e.g., "fire engine" -> FireEngine).
5
6use std::collections::HashMap;
7use crate::token::{Token, TokenType};
8use crate::lexicon::{VerbClass, Time, Aspect};
9use logicaffeine_base::Interner;
10
11#[derive(Debug, Clone)]
12pub struct MweTarget {
13    pub lemma: &'static str,
14    pub pos: &'static str,
15    pub class: Option<VerbClass>,
16}
17
18#[derive(Default, Debug)]
19pub struct MweTrie {
20    pub children: HashMap<String, MweTrie>,
21    pub target: Option<MweTarget>,
22}
23
24impl MweTrie {
25    pub fn insert(&mut self, pattern: &[&str], target: MweTarget) {
26        if pattern.is_empty() {
27            self.target = Some(target);
28            return;
29        }
30        self.children
31            .entry(pattern[0].to_lowercase())
32            .or_default()
33            .insert(&pattern[1..], target);
34    }
35}
36
37/// Apply MWE collapsing to a token stream.
38/// Matches on lemmas (not raw strings) to handle morphological variants.
39pub fn apply_mwe_pipeline(
40    tokens: Vec<Token>,
41    trie: &MweTrie,
42    interner: &mut Interner,
43) -> Vec<Token> {
44    let mut result = Vec::new();
45    let mut i = 0;
46
47    while i < tokens.len() {
48        if let Some((match_len, target)) = find_longest_match(&tokens[i..], trie, interner) {
49            let merged = create_merged_token(&tokens[i], target, interner);
50            result.push(merged);
51            i += match_len;
52        } else {
53            result.push(tokens[i].clone());
54            i += 1;
55        }
56    }
57    result
58}

60/// Extract lemma from a token for MWE matching.
61/// Uses lowercase for case-insensitive matching.
62fn get_lemma(token: &Token, interner: &Interner) -> String {
63    match &token.kind {
64        TokenType::Verb { lemma, .. } => interner.resolve(*lemma).to_lowercase(),
65        TokenType::Noun(sym) => interner.resolve(*sym).to_lowercase(),
66        TokenType::Adjective(sym) => interner.resolve(*sym).to_lowercase(),
67        TokenType::NonIntersectiveAdjective(sym) => interner.resolve(*sym).to_lowercase(),
68        TokenType::Preposition(sym) => interner.resolve(*sym).to_lowercase(),
69        TokenType::Particle(sym) => interner.resolve(*sym).to_lowercase(),
70        TokenType::Article(_) => interner.resolve(token.lexeme).to_lowercase(),
71        _ => interner.resolve(token.lexeme).to_lowercase(),
72    }
73}
74
75/// Find the longest MWE match starting at the beginning of the token slice.
76fn find_longest_match<'a>(
77    tokens: &[Token],
78    trie: &'a MweTrie,
79    interner: &Interner,
80) -> Option<(usize, &'a MweTarget)> {
81    let mut node = trie;
82    let mut best: Option<(usize, &MweTarget)> = None;
83
84    for (i, token) in tokens.iter().enumerate() {
85        let lemma = get_lemma(token, interner);
86        if let Some(child) = node.children.get(&lemma) {
87            node = child;
88            if let Some(target) = &node.target {
89                best = Some((i + 1, target));
90            }
91        } else {
92            break;
93        }
94    }
95    best
96}
97
98/// Create a merged token from the MWE target, inheriting tense from the head token.
99fn create_merged_token(head: &Token, target: &MweTarget, interner: &mut Interner) -> Token {
100    let lemma_sym = interner.intern(target.lemma);
101
102    let kind = match target.pos {
103        "Noun" => TokenType::Noun(lemma_sym),
104        "Verb" => {
105            let (time, aspect) = match &head.kind {
106                TokenType::Verb { time, aspect, .. } => (*time, *aspect),
107                _ => (Time::Present, Aspect::Simple),
108            };
109            TokenType::Verb {
110                lemma: lemma_sym,
111                time,
112                aspect,
113                class: target.class.unwrap_or(VerbClass::Activity),
114            }
115        }
116        "Preposition" => TokenType::Preposition(lemma_sym),
117        "Conjunction" => TokenType::And,
118        "Quantifier" => TokenType::NoOne,
119        _ => TokenType::Noun(lemma_sym),
120    };
121
122    Token {
123        kind,
124        lexeme: lemma_sym,
125        span: head.span,
126    }
127}
