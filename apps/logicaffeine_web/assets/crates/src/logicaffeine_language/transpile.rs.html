1//! Transpilation from AST to first-order logic notation.
2//!
3//! This module converts the internal AST representation to human-readable
4//! logical formulas in various output formats.
5//!
6//! ## Output Formats
7//!
8//! | Format | Example | Use Case |
9//! |--------|---------|----------|
10//! | **Unicode** | ∀x(Cat(x) → Sleeps(x)) | Terminal display |
11//! | **LaTeX** | `\forall x(Cat(x) \to Sleeps(x))` | Academic papers |
12//! | **ASCII** | `Ax(Cat(x) -> Sleeps(x))` | Plain text |
13//! | **Kripke** | `□(P) @w0` | Modal logic diagrams |
14//!
15//! ## Neo-Davidsonian Events
16//!
17//! Verb semantics use event variables following Davidson's event semantics:
18//!
19//! ```text
20//! "John loves Mary"
21//! → ∃e(Love(e) ∧ Agent(e, john) ∧ Theme(e, mary))
22//! ```
23//!
24//! This representation enables reasoning about event modification,
25//! aspect, and temporal relations.

37/// Collect event variables from NeoEvents with suppress_existential=true
38/// Returns unique event variables (coordinated weather verbs share the same var)
39fn collect_suppress_existential_events<'a>(expr: &LogicExpr<'a>) -> Vec<Symbol> {
40    let mut events = Vec::new();
41    collect_suppress_existential_events_inner(expr, &mut events);
42    // Deduplicate - coordinated weather verbs share the same event variable
43    // Symbol is Copy so we can use a simple O(n^2) dedup
44    let mut unique = Vec::new();
45    for e in events {
46        if !unique.iter().any(|x| *x == e) {
47            unique.push(e);
48        }
49    }
50    unique
51}
52
53fn collect_suppress_existential_events_inner<'a>(expr: &LogicExpr<'a>, events: &mut Vec<Symbol>) {
54    match expr {
55        LogicExpr::NeoEvent(data) => {
56            if data.suppress_existential {
57                events.push(data.event_var);
58            }
59        }
60        LogicExpr::BinaryOp { left, right, .. } => {
61            collect_suppress_existential_events_inner(left, events);
62            collect_suppress_existential_events_inner(right, events);
63        }
64        LogicExpr::UnaryOp { operand, .. } => {
65            collect_suppress_existential_events_inner(operand, events);
66        }
67        LogicExpr::Temporal { body, .. } => {
68            collect_suppress_existential_events_inner(body, events);
69        }
70        LogicExpr::Aspectual { body, .. } => {
71            collect_suppress_existential_events_inner(body, events);
72        }
73        LogicExpr::Modal { operand, .. } => {
74            collect_suppress_existential_events_inner(operand, events);
75        }
76        _ => {}
77    }
78}

80pub fn capitalize_first(s: &str) -> String {
81    let mut chars = s.chars();
82    match chars.next() {
83        None => String::new(),
84        Some(c) => c.to_uppercase().collect::<String>() + chars.as_str(),
85    }
86}
87
88fn write_capitalized<W: Write>(w: &mut W, s: &str) -> std::fmt::Result {
89    let mut chars = s.chars();
90    match chars.next() {
91        None => Ok(()),
92        Some(c) => {
93            for uc in c.to_uppercase() {
94                write!(w, "{}", uc)?;
95            }
96            write!(w, "{}", chars.as_str())
97        }
98    }
99}

115        registry: &mut SymbolRegistry,
116        interner: &Interner,

121    pub fn write_to_full<W: Write>(
122        &self,
123        w: &mut W,
124        registry: &mut SymbolRegistry,
125        interner: &Interner,
126    ) -> std::fmt::Result {
127        self.write_to_inner(w, registry, interner, true)

129
130    /// Write term preserving original case (for code generation)
131    pub fn write_to_raw<W: Write>(
132        &self,
133        w: &mut W,
134        interner: &Interner,
135    ) -> std::fmt::Result {
136        match self {
137            Term::Constant(name) | Term::Variable(name) => {
138                write!(w, "{}", interner.resolve(*name))
139            }
140            Term::Function(name, args) => {
141                write!(w, "{}(", interner.resolve(*name))?;
142                for (i, arg) in args.iter().enumerate() {
143                    if i > 0 {
144                        write!(w, ", ")?;
145                    }
146                    arg.write_to_raw(w, interner)?;
147                }
148                write!(w, ")")
149            }
150            Term::Group(members) => {
151                write!(w, "(")?;
152                for (i, m) in members.iter().enumerate() {
153                    if i > 0 {
154                        write!(w, ", ")?;
155                    }
156                    m.write_to_raw(w, interner)?;
157                }
158                write!(w, ")")
159            }
160            Term::Possessed { possessor, possessed } => {
161                possessor.write_to_raw(w, interner)?;
162                write!(w, ".{}", interner.resolve(*possessed))
163            }
164            Term::Value { kind, .. } => match kind {
165                NumberKind::Integer(n) => write!(w, "{}", n),
166                NumberKind::Real(f) => write!(w, "{}", f),
167                NumberKind::Symbolic(s) => write!(w, "{}", interner.resolve(*s)),
168            }
169            Term::Sigma(predicate) => write!(w, "σ({})", interner.resolve(*predicate)),
170            Term::Intension(predicate) => write!(w, "^{}", interner.resolve(*predicate)),
171            Term::Proposition(expr) => write!(w, "[proposition]"),
172        }
173    }
174
175    fn write_to_inner<W: Write>(
176        &self,
177        w: &mut W,
178        registry: &mut SymbolRegistry,
179        interner: &Interner,
180        use_full_names: bool,
181    ) -> std::fmt::Result {
182        match self {
183            Term::Constant(name) => {
184                if use_full_names {
185                    write!(w, "{}", registry.get_symbol_full(*name, interner))
186                } else {
187                    write!(w, "{}", registry.get_symbol(*name, interner))
188                }
189            }
190            Term::Variable(name) => write!(w, "{}", interner.resolve(*name)),
191            Term::Function(name, args) => {
192                let fn_name = if use_full_names {
193                    registry.get_symbol_full(*name, interner)
194                } else {
195                    registry.get_symbol(*name, interner)
196                };
197                write!(w, "{}(", fn_name)?;
198                for (i, arg) in args.iter().enumerate() {
199                    if i > 0 {
200                        write!(w, ", ")?;
201                    }
202                    arg.write_to_inner(w, registry, interner, use_full_names)?;
203                }
204                write!(w, ")")
205            }
206            Term::Group(members) => {
207                for (i, m) in members.iter().enumerate() {
208                    if i > 0 {
209                        write!(w, " ⊕ ")?;
210                    }
211                    m.write_to_inner(w, registry, interner, use_full_names)?;
212                }
213                Ok(())
214            }
215            Term::Possessed { possessor, possessed } => {
216                let poss_name = if use_full_names {
217                    registry.get_symbol_full(*possessed, interner)
218                } else {
219                    registry.get_symbol(*possessed, interner)
220                };
221                write!(w, "Poss(")?;
222                possessor.write_to_inner(w, registry, interner, use_full_names)?;
223                write!(w, ", {})", poss_name)
224            }
225            Term::Sigma(predicate) => {
226                let pred_name = if use_full_names {
227                    registry.get_symbol_full(*predicate, interner)
228                } else {
229                    registry.get_symbol(*predicate, interner)
230                };
231                write!(w, "σ{}", pred_name)
232            }
233            Term::Intension(predicate) => {
234                // Use full word for intensional terms, not abbreviated symbol
235                let word = interner.resolve(*predicate);
236                let capitalized = word.chars().next()
237                    .map(|c| c.to_uppercase().collect::<String>() + &word[1..])
238                    .unwrap_or_default();
239                write!(w, "^{}", capitalized)
240            }
241            Term::Proposition(expr) => {
242                write!(w, "[")?;
243                expr.write_logic(w, registry, interner, &UnicodeFormatter)?;
244                write!(w, "]")
245            }
246            Term::Value { kind, unit, dimension: _ } => {
247                use crate::ast::NumberKind;
248                match kind {
249                    NumberKind::Real(r) => write!(w, "{}", r)?,
250                    NumberKind::Integer(i) => write!(w, "{}", i)?,
251                    NumberKind::Symbolic(s) => write!(w, "{}", interner.resolve(*s))?,
252                }
253                if let Some(u) = unit {
254                    write!(w, " {}", interner.resolve(*u))?;
255                }
256                Ok(())
257            }
258        }
259    }

261    pub fn transpile(&self, registry: &mut SymbolRegistry, interner: &Interner) -> String {
262        let mut buf = String::new();
263        let _ = self.write_to(&mut buf, registry, interner);
264        buf
265    }
266}

268/// Extract top-level conjuncts from a discourse (sentences combined with AND).
269/// Returns a vector of individual sentence expressions.
270fn collect_discourse_conjuncts<'a>(expr: &'a LogicExpr<'a>) -> Vec<&'a LogicExpr<'a>> {
271    let mut conjuncts = Vec::new();
272    collect_discourse_conjuncts_inner(expr, &mut conjuncts);
273    conjuncts
274}
275
276fn collect_discourse_conjuncts_inner<'a>(expr: &'a LogicExpr<'a>, conjuncts: &mut Vec<&'a LogicExpr<'a>>) {
277    match expr {
278        LogicExpr::BinaryOp { left, op: TokenType::And, right } => {
279            // Recursively collect from both sides
280            collect_discourse_conjuncts_inner(left, conjuncts);
281            collect_discourse_conjuncts_inner(right, conjuncts);
282        }
283        _ => {
284            // This is a leaf sentence (not a top-level conjunction)
285            conjuncts.push(expr);
286        }
287    }
288}

290impl<'a> LogicExpr<'a> {
291    /// Transpile a discourse (multiple sentences) as numbered formulas.
292    /// If the expression is a top-level conjunction of sentences, formats as:
293    /// ```text
294    /// 1) formula1
295    /// 2) formula2
296    /// 3) formula3
297    /// ```
298    /// If it's a single sentence, just returns the formula without numbering.
299    pub fn transpile_discourse(
300        &self,
301        registry: &mut SymbolRegistry,
302        interner: &Interner,
303        format: OutputFormat,
304    ) -> String {
305        let conjuncts = collect_discourse_conjuncts(self);
306
307        if conjuncts.len() <= 1 {
308            // Single sentence - no numbering needed
309            return self.transpile(registry, interner, format);
310        }
311
312        // Multiple sentences - format as numbered list
313        let mut result = String::new();
314        for (i, conjunct) in conjuncts.iter().enumerate() {
315            if i > 0 {
316                result.push('\n');
317            }
318            let formula = conjunct.transpile(registry, interner, format);
319            result.push_str(&format!("{}) {}", i + 1, formula));
320        }
321        result
322    }

324    pub fn write_logic<W: Write, F: LogicFormatter>(
325        &self,
326        w: &mut W,
327        registry: &mut SymbolRegistry,
328        interner: &Interner,
329        fmt: &F,
330    ) -> std::fmt::Result {
331        match self {
332            LogicExpr::Predicate { name, args, world } => {
333                let pred_name = if fmt.use_full_names() {
334                    registry.get_symbol_full(*name, interner)
335                } else {
336                    registry.get_symbol(*name, interner)
337                };
338
339                // If formatter wants world arguments and we have one, append it
340                if fmt.include_world_arguments() {
341                    if let Some(w_sym) = world {
342                        // Build extended args with world variable appended
343                        let mut extended: Vec<Term> = args.to_vec();
344                        extended.push(Term::Variable(*w_sym));
345                        return fmt.write_predicate(w, &pred_name, &extended, registry, interner);
346                    }
347                }
348                fmt.write_predicate(w, &pred_name, args, registry, interner)
349            }
350
351            LogicExpr::Identity { left, right } => {
352                if fmt.wrap_identity() {
353                    write!(w, "(")?;
354                }
355                if fmt.preserve_case() {
356                    left.write_to_raw(w, interner)?;
357                } else if fmt.use_full_names() {
358                    left.write_to_full(w, registry, interner)?;
359                } else {
360                    left.write_to(w, registry, interner)?;
361                }
362                write!(w, "{}", fmt.identity())?;
363                if fmt.preserve_case() {
364                    right.write_to_raw(w, interner)?;
365                } else if fmt.use_full_names() {
366                    right.write_to_full(w, registry, interner)?;
367                } else {
368                    right.write_to(w, registry, interner)?;
369                }
370                if fmt.wrap_identity() {
371                    write!(w, ")")?;
372                }
373                Ok(())
374            }
375
376            LogicExpr::Metaphor { tenor, vehicle } => {
377                write!(w, "Metaphor(")?;
378                tenor.write_to(w, registry, interner)?;
379                write!(w, ", ")?;
380                vehicle.write_to(w, registry, interner)?;
381                write!(w, ")")
382            }
383
384            LogicExpr::Quantifier { kind, variable, body, .. } => {
385                let var_str = interner.resolve(*variable);
386
387                // In SimpleFOL mode, skip event quantifiers (variables named "e" or starting with "e" followed by digits)
388                if fmt.use_simple_events() && (var_str == "e" || var_str.starts_with("e") && var_str[1..].chars().all(|c| c.is_ascii_digit())) {
389                    return body.write_logic(w, registry, interner, fmt);
390                }
391
392                let mut body_buf = String::new();
393                body.write_logic(&mut body_buf, registry, interner, fmt)?;
394                write!(w, "{}", fmt.quantifier(kind, var_str, &body_buf))
395            }
396
397            LogicExpr::Categorical(data) => {
398                let s = if fmt.use_full_names() {
399                    fmt.sanitize(&data.subject.to_symbol_full(registry, interner))
400                } else {
401                    fmt.sanitize(&data.subject.to_symbol(registry, interner))
402                };
403                let p = if fmt.use_full_names() {
404                    fmt.sanitize(&data.predicate.to_symbol_full(registry, interner))
405                } else {
406                    fmt.sanitize(&data.predicate.to_symbol(registry, interner))
407                };
408                match (&data.quantifier, data.copula_negative) {
409                    (TokenType::All, false) => write!(w, "{} {} is {}", fmt.categorical_all(), s, p),
410                    (TokenType::No, false) => write!(w, "{} {} is {}", fmt.categorical_no(), s, p),
411                    (TokenType::Some, false) => write!(w, "{} {} is {}", fmt.categorical_some(), s, p),
412                    (TokenType::Some, true) => write!(w, "{} {} is {} {}", fmt.categorical_some(), s, fmt.categorical_not(), p),
413                    (TokenType::All, true) => write!(w, "{} {} is {} {}", fmt.categorical_some(), s, fmt.categorical_not(), p),
414                    _ => write!(w, "Invalid Syllogism"),
415                }
416            }
417
418            LogicExpr::Relation(data) => {
419                let s = if fmt.use_full_names() {
420                    data.subject.to_symbol_full(registry, interner)
421                } else {
422                    data.subject.to_symbol(registry, interner)
423                };
424                let v = if fmt.use_full_names() {
425                    fmt.sanitize(&registry.get_symbol_full(data.verb, interner))
426                } else {
427                    fmt.sanitize(&registry.get_symbol(data.verb, interner))
428                };
429                let o = if fmt.use_full_names() {
430                    data.object.to_symbol_full(registry, interner)
431                } else {
432                    data.object.to_symbol(registry, interner)
433                };
434                write!(w, "{}({}, {})", v, s, o)
435            }
436
437            LogicExpr::Modal { vector, operand } => {
438                let mut o = String::new();
439                operand.write_logic(&mut o, registry, interner, fmt)?;
440                write!(w, "{}", fmt.modal(vector.domain, vector.force, &o))
441            }
442
443            LogicExpr::BinaryOp { left, op, right } => {
444                let mut l = String::new();
445                let mut r = String::new();
446                left.write_logic(&mut l, registry, interner, fmt)?;
447                right.write_logic(&mut r, registry, interner, fmt)?;
448
449                // For conditionals (If), check if there are suppress_existential events
450                // that need universal quantification (DRS semantics for generic conditionals)
451                if matches!(op, TokenType::If) {
452                    let events = collect_suppress_existential_events(self);
453                    if !events.is_empty() {
454                        // Wrap with universal quantifiers for each event variable
455                        let mut result = fmt.binary_op(op, &l, &r);
456                        for event_var in events.into_iter().rev() {
457                            let var_str = interner.resolve(event_var);
458                            result = fmt.quantifier(&QuantifierKind::Universal, var_str, &result);
459                        }
460                        return write!(w, "{}", result);
461                    }
462                }
463
464                write!(w, "{}", fmt.binary_op(op, &l, &r))
465            }
466
467            LogicExpr::UnaryOp { op, operand } => {
468                let mut o = String::new();
469                operand.write_logic(&mut o, registry, interner, fmt)?;
470                write!(w, "{}", fmt.unary_op(op, &o))
471            }
472
473            LogicExpr::Temporal { operator, body } => {
474                let mut inner = String::new();
475                body.write_logic(&mut inner, registry, interner, fmt)?;
476                write!(w, "{}", fmt.temporal(operator, &inner))
477            }
478
479            LogicExpr::Aspectual { operator, body } => {
480                let mut inner = String::new();
481                body.write_logic(&mut inner, registry, interner, fmt)?;
482                write!(w, "{}", fmt.aspectual(operator, &inner))
483            }
484
485            LogicExpr::Voice { operator, body } => {
486                let mut inner = String::new();
487                body.write_logic(&mut inner, registry, interner, fmt)?;
488                write!(w, "{}", fmt.voice(operator, &inner))
489            }
490
491            LogicExpr::Question { wh_variable, body } => {
492                let mut body_str = String::new();
493                body.write_logic(&mut body_str, registry, interner, fmt)?;
494                write!(w, "{}", fmt.lambda(interner.resolve(*wh_variable), &body_str))
495            }
496
497            LogicExpr::YesNoQuestion { body } => {
498                write!(w, "?")?;
499                body.write_logic(w, registry, interner, fmt)
500            }
501
502            LogicExpr::Atom(s) => {
503                let name = if fmt.preserve_case() {
504                    interner.resolve(*s).to_string()
505                } else if fmt.use_full_names() {
506                    registry.get_symbol_full(*s, interner)
507                } else {
508                    registry.get_symbol(*s, interner)
509                };
510                write!(w, "{}", fmt.sanitize(&name))
511            }
512
513            LogicExpr::Lambda { variable, body } => {
514                let mut b = String::new();
515                body.write_logic(&mut b, registry, interner, fmt)?;
516                write!(w, "{}", fmt.lambda(interner.resolve(*variable), &b))
517            }
518
519            LogicExpr::App { function, argument } => {
520                write!(w, "(")?;
521                function.write_logic(w, registry, interner, fmt)?;
522                write!(w, ")(")?;
523                argument.write_logic(w, registry, interner, fmt)?;
524                write!(w, ")")
525            }
526
527            LogicExpr::Intensional { operator, content } => {
528                write!(w, "{}[", fmt.sanitize(&registry.get_symbol(*operator, interner)))?;
529                content.write_logic(w, registry, interner, fmt)?;
530                write!(w, "]")
531            }
532
533            LogicExpr::Event { predicate, adverbs } => {
534                let mut pred_str = String::new();
535                predicate.write_logic(&mut pred_str, registry, interner, fmt)?;
536                let adverb_preds: Vec<String> = adverbs
537                    .iter()
538                    .map(|a| format!("{}(e)", fmt.sanitize(&registry.get_symbol(*a, interner))))
539                    .collect();
540                write!(w, "{}", fmt.event_quantifier(&pred_str, &adverb_preds))
541            }
542
543            LogicExpr::NeoEvent(data) => {
544                use crate::ast::{QuantifierKind, ThematicRole};
545
546                if fmt.use_simple_events() {
547                    write!(w, "{}", registry.get_symbol_full(data.verb, interner))?;
548                    write!(w, "(")?;
549                    let mut first = true;
550                    for (role, term) in data.roles.iter() {
551                        // Include core thematic roles in SimpleFOL output
552                        if matches!(role, ThematicRole::Agent | ThematicRole::Patient | ThematicRole::Theme | ThematicRole::Goal | ThematicRole::Location) {
553                            if !first {
554                                write!(w, ", ")?;
555                            }
556                            first = false;
557                            term.write_to_full(w, registry, interner)?;
558                        }
559                    }
560                    write!(w, ")")
561                } else {
562                    let e = interner.resolve(data.event_var);
563                    let mut body = String::new();
564
565                    // Get world argument suffix if Kripke format
566                    let world_suffix = if fmt.include_world_arguments() {
567                        data.world.map(|w| format!(", {}", interner.resolve(w))).unwrap_or_default()
568                    } else {
569                        String::new()
570                    };
571
572                    write_capitalized(&mut body, interner.resolve(data.verb))?;
573                    write!(body, "({}{})", e, world_suffix)?;
574                    for (role, term) in data.roles.iter() {
575                        let role_str = match role {
576                            ThematicRole::Agent => "Agent",
577                            ThematicRole::Patient => "Patient",
578                            ThematicRole::Theme => "Theme",
579                            ThematicRole::Recipient => "Recipient",
580                            ThematicRole::Goal => "Goal",
581                            ThematicRole::Source => "Source",
582                            ThematicRole::Instrument => "Instrument",
583                            ThematicRole::Location => "Location",
584                            ThematicRole::Time => "Time",
585                            ThematicRole::Manner => "Manner",
586                        };
587                        write!(body, " {} {}({}, ", fmt.and(), role_str, e)?;
588                        if fmt.use_full_names() {
589                            term.write_to_full(&mut body, registry, interner)?;
590                        } else {
591                            term.write_to(&mut body, registry, interner)?;
592                        }
593                        write!(body, "{})", world_suffix)?;
594                    }
595                    for mod_sym in data.modifiers.iter() {
596                        write!(body, " {} ", fmt.and())?;
597                        write_capitalized(&mut body, interner.resolve(*mod_sym))?;
598                        write!(body, "({}{})", e, world_suffix)?;
599                    }
600                    if data.suppress_existential {
601                        // Event var will be bound by outer ∀ from DRS (generic conditionals)
602                        write!(w, "{}", body)
603                    } else {
604                        // Normal case: emit ∃e(...)
605                        write!(w, "{}", fmt.quantifier(&QuantifierKind::Existential, e, &body))
606                    }
607                }
608            }
609
610            LogicExpr::Imperative { action } => {
611                write!(w, "!")?;
612                action.write_logic(w, registry, interner, fmt)
613            }
614
615            LogicExpr::SpeechAct { performer, act_type, content } => {
616                write!(w, "SpeechAct({}, {}, ", interner.resolve(*act_type), fmt.sanitize(&registry.get_symbol(*performer, interner)))?;
617                content.write_logic(w, registry, interner, fmt)?;
618                write!(w, ")")
619            }
620
621            LogicExpr::Counterfactual { antecedent, consequent } => {
622                let mut a = String::new();
623                let mut c = String::new();
624                antecedent.write_logic(&mut a, registry, interner, fmt)?;
625                consequent.write_logic(&mut c, registry, interner, fmt)?;
626                write!(w, "{}", fmt.counterfactual(&a, &c))
627            }
628
629            LogicExpr::Causal { effect, cause } => {
630                write!(w, "Cause(")?;
631                cause.write_logic(w, registry, interner, fmt)?;
632                write!(w, ", ")?;
633                effect.write_logic(w, registry, interner, fmt)?;
634                write!(w, ")")
635            }
636
637            LogicExpr::Comparative { adjective, subject, object, difference } => {
638                let adj = interner.resolve(*adjective);
639                let mut subj_buf = String::new();
640                if fmt.preserve_case() {
641                    subject.write_to_raw(&mut subj_buf, interner)?;
642                } else {
643                    subject.write_to(&mut subj_buf, registry, interner)?;
644                }
645                let mut obj_buf = String::new();
646                if fmt.preserve_case() {
647                    object.write_to_raw(&mut obj_buf, interner)?;
648                } else {
649                    object.write_to(&mut obj_buf, registry, interner)?;
650                }
651                let diff_str = if let Some(diff) = difference {
652                    let mut diff_buf = String::new();
653                    if fmt.preserve_case() {
654                        diff.write_to_raw(&mut diff_buf, interner)?;
655                    } else {
656                        diff.write_to(&mut diff_buf, registry, interner)?;
657                    }
658                    Some(diff_buf)
659                } else {
660                    None
661                };
662                fmt.write_comparative(w, adj, &subj_buf, &obj_buf, diff_str.as_deref())
663            }
664
665            LogicExpr::Superlative { adjective, subject, domain } => {
666                let mut s = String::new();
667                subject.write_to(&mut s, registry, interner)?;
668                let mut d = String::new();
669                write_capitalized(&mut d, interner.resolve(*domain))?;
670                let comp = format!("{}er", interner.resolve(*adjective));
671                write!(w, "{}", fmt.superlative(&comp, &d, &s))
672            }
673
674            LogicExpr::Scopal { operator, body } => {
675                write!(w, "{}(", interner.resolve(*operator))?;
676                body.write_logic(w, registry, interner, fmt)?;
677                write!(w, ")")
678            }
679
680            LogicExpr::TemporalAnchor { anchor, body } => {
681                write!(w, "{}(", interner.resolve(*anchor))?;
682                body.write_logic(w, registry, interner, fmt)?;
683                write!(w, ")")
684            }
685
686            LogicExpr::Control { verb, subject, object, infinitive } => {
687                write!(w, "{}(", fmt.sanitize(&registry.get_symbol(*verb, interner)))?;
688                subject.write_to(w, registry, interner)?;
689                if let Some(obj) = object {
690                    write!(w, ", ")?;
691                    obj.write_to(w, registry, interner)?;
692                }
693                write!(w, ", ")?;
694                infinitive.write_logic(w, registry, interner, fmt)?;
695                write!(w, ")")
696            }
697
698            LogicExpr::Presupposition { assertion, presupposition } => {
699                assertion.write_logic(w, registry, interner, fmt)?;
700                write!(w, " [Presup: ")?;
701                presupposition.write_logic(w, registry, interner, fmt)?;
702                write!(w, "]")
703            }
704
705            LogicExpr::Focus { kind, focused, scope } => {
706                use crate::token::FocusKind;
707                let prefix = match kind {
708                    FocusKind::Only => "Only",
709                    FocusKind::Even => "Even",
710                    FocusKind::Just => "Just",
711                };
712                write!(w, "{}(", prefix)?;
713                focused.write_to(w, registry, interner)?;
714                write!(w, ", ")?;
715                scope.write_logic(w, registry, interner, fmt)?;
716                write!(w, ")")
717            }
718
719            LogicExpr::Distributive { predicate } => {
720                write!(w, "*")?;
721                predicate.write_logic(w, registry, interner, fmt)
722            }
723
724            LogicExpr::GroupQuantifier { group_var, count, member_var, restriction, body } => {
725                let g = interner.resolve(*group_var);
726                let x = interner.resolve(*member_var);
727
728                // ∃g(Group(g) ∧ Count(g,n) ∧ ∀x(Member(x,g) → restriction) ∧ body)
729                write!(w, "{}{}(Group({}) {} Count({}, {}) {} {}{}(Member({}, {}) {} ",
730                    fmt.existential(), g, g,
731                    fmt.and(), g, count,
732                    fmt.and(), fmt.universal(), x, x, g, fmt.implies())?;
733
734                restriction.write_logic(w, registry, interner, fmt)?;
735
736                write!(w, ") {} ", fmt.and())?;
737
738                body.write_logic(w, registry, interner, fmt)?;
739
740                write!(w, ")")
741            }
742        }
743    }
744
745    pub fn transpile_with<F: LogicFormatter>(
746        &self,
747        registry: &mut SymbolRegistry,
748        interner: &Interner,
749        fmt: &F,
750    ) -> String {
751        let mut buf = String::new();
752        let _ = self.write_logic(&mut buf, registry, interner, fmt);
753        buf
754    }
755
756    pub fn transpile(
757        &self,
758        registry: &mut SymbolRegistry,
759        interner: &Interner,
760        format: OutputFormat,
761    ) -> String {
762        match format {
763            OutputFormat::Unicode => self.transpile_with(registry, interner, &UnicodeFormatter),
764            OutputFormat::LaTeX => self.transpile_with(registry, interner, &LatexFormatter),
765            OutputFormat::SimpleFOL => self.transpile_with(registry, interner, &SimpleFOLFormatter),
766            OutputFormat::Kripke => self.transpile_with(registry, interner, &KripkeFormatter),
767        }
768    }
769
770    pub fn transpile_ctx<F: LogicFormatter>(
771        &self,
772        ctx: &mut TranspileContext<'_>,
773        fmt: &F,
774    ) -> String {
775        self.transpile_with(ctx.registry, ctx.interner, fmt)
776    }
777
778    pub fn transpile_ctx_unicode(&self, ctx: &mut TranspileContext<'_>) -> String {
779        self.transpile_ctx(ctx, &UnicodeFormatter)
780    }
781
782    pub fn transpile_ctx_latex(&self, ctx: &mut TranspileContext<'_>) -> String {
783        self.transpile_ctx(ctx, &LatexFormatter)
784    }
