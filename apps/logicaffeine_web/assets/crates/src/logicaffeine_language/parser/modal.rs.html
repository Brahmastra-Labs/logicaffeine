1//! Modal verb parsing with Kripke semantics support.
2//!
3//! This module handles modal auxiliaries (can, could, may, might, must, should, would)
4//! and their semantic interpretation using modal vectors that encode:
5//!
6//! - **Domain**: Alethic (possibility/necessity) vs Deontic (permission/obligation)
7//! - **Flavor**: Root (circumstantial) vs Epistemic (knowledge-based)
8//! - **Force**: Possibility (◇) vs Necessity (□)
9//!
10//! # Modal Vector Examples
11//!
12//! | Modal | Default Reading | Alternative Reading |
13//! |-------|-----------------|---------------------|
14//! | can   | Ability (Root)  | Permission (Deontic) |
15//! | may   | Permission (Deontic) | Possibility (Epistemic) |
16//! | must  | Necessity (Root) | Obligation (Deontic) |
17//! | might | Possibility (Epistemic) | - |
18//!
19//! The module also handles aspect chains (perfect "have", progressive "be -ing").
20

23use super::{ParseResult, Parser};
24use crate::ast::{AspectOperator, LogicExpr, ModalDomain, ModalFlavor, ModalVector, NeoEventData, ThematicRole, VoiceOperator, Term};
25use crate::drs::TimeRelation;

31pub trait ModalParsing<'a, 'ctx, 'int> {
32    fn parse_modal(&mut self) -> ParseResult<&'a LogicExpr<'a>>;
33    fn parse_aspect_chain(&mut self, subject_symbol: Symbol) -> ParseResult<&'a LogicExpr<'a>>;
34    fn parse_aspect_chain_with_term(&mut self, subject_term: Term<'a>) -> ParseResult<&'a LogicExpr<'a>>;
35    fn token_to_vector(&self, token: &TokenType) -> ModalVector;
36}
37
38impl<'a, 'ctx, 'int> ModalParsing<'a, 'ctx, 'int> for Parser<'a, 'ctx, 'int> {
39    fn parse_modal(&mut self) -> ParseResult<&'a LogicExpr<'a>> {
40        use crate::drs::BoxType;
41
42        let vector = self.token_to_vector(&self.previous().kind.clone());
43
44        // Enter modal box in parser's DRS (not world_state - that's swapped at sentence boundaries)
45        self.drs.enter_box(BoxType::ModalScope);
46
47        if self.check(&TokenType::That) {
48            self.advance();
49        }
50
51        let content = self.parse_sentence()?;
52
53        // Exit modal box
54        self.drs.exit_box();
55
56        Ok(self.ctx.exprs.alloc(LogicExpr::Modal {
57            vector,
58            operand: content,
59        }))
60    }
61
62    fn parse_aspect_chain(&mut self, subject_symbol: Symbol) -> ParseResult<&'a LogicExpr<'a>> {
63        self.parse_aspect_chain_with_term(Term::Constant(subject_symbol))
64    }
65
66    fn parse_aspect_chain_with_term(&mut self, subject_term: Term<'a>) -> ParseResult<&'a LogicExpr<'a>> {
67        let mut has_modal = false;
68        let mut modal_vector = None;
69        let mut has_negation = false;
70        let mut has_perfect = false;
71        let mut has_passive = false;
72        let mut has_progressive = false;
73
74        if self.check(&TokenType::Would) || self.check(&TokenType::Could)
75            || self.check(&TokenType::Must) || self.check(&TokenType::Can)
76            || self.check(&TokenType::Should) || self.check(&TokenType::May)
77            || self.check(&TokenType::Cannot) || self.check(&TokenType::Might) {
78            let modal_token = self.peek().kind.clone();
79            self.advance();
80            has_modal = true;
81            let vector = self.token_to_vector(&modal_token);
82            modal_vector = Some(vector.clone());
83            // Enter modal box in DRS so any new referents are marked as hypothetical
84            // This ensures "A wolf might enter" puts the wolf in a modal scope
85            self.drs.enter_box(crate::drs::BoxType::ModalScope);
86            // Also set modal context on WorldState for cross-sentence tracking
87            // This is used by end_sentence() to mark telescope candidates as modal-sourced
88            let is_epistemic = matches!(vector.flavor, crate::ast::ModalFlavor::Epistemic);
89            self.world_state.enter_modal_context(is_epistemic, vector.force);
90        }
91
92        if self.check(&TokenType::Not) {
93            self.advance();
94            has_negation = true;
95        }
96
97        // Check for "be able to" periphrastic modal (= can)
98        // This creates a nested modal: "might be able to fly" → ◇◇Fly(x)
99        let mut nested_modal_vector = None;
100        if self.check_content_word() {
101            let word = self.interner.resolve(self.peek().lexeme).to_lowercase();
102            if word == "be" {
103                // Look ahead for "able to"
104                if let Some(next1) = self.tokens.get(self.current + 1) {
105                    let next1_word = self.interner.resolve(next1.lexeme).to_lowercase();
106                    if next1_word == "able" {
107                        if let Some(next2) = self.tokens.get(self.current + 2) {
108                            if matches!(next2.kind, TokenType::To) {
109                                // Consume "be able to" - it's a modal meaning "can" (ability)
110                                self.advance(); // consume "be"
111                                self.advance(); // consume "able"
112                                self.advance(); // consume "to"
113                                nested_modal_vector = Some(ModalVector {
114                                    domain: ModalDomain::Alethic,
115                                    force: 0.5, // ability = possibility
116                                    flavor: ModalFlavor::Root, // "be able to" = Root modal (ability)
117                                });
118                            }
119                        }
120                    }
121                }
122            }
123        }
124
125        if self.check_content_word() {
126            let word = self.interner.resolve(self.peek().lexeme).to_lowercase();
127            if word == "have" || word == "has" || word == "had" {
128                self.advance();
129                has_perfect = true;
130            }
131        }
132
133        if self.check(&TokenType::Had) {
134            self.advance();
135            has_perfect = true;
136            // "had" = past perfect: R < S (past reference time)
137            let r_var = self.world_state.next_reference_time();
138            self.world_state.add_time_constraint(r_var, TimeRelation::Precedes, "S".to_string());
139        }
140
141        if self.check_content_word() {
142            let word = self.interner.resolve(self.peek().lexeme).to_lowercase();
143            if word == "been" {
144                self.advance();
145
146                if self.check_verb() {
147                    match &self.peek().kind {
148                        TokenType::Verb { aspect: Aspect::Progressive, .. } => {
149                            has_progressive = true;
150                        }
151                        TokenType::Verb { .. } => {
152                            let next_word = self.interner.resolve(self.peek().lexeme);
153                            if next_word.ends_with("ing") {
154                                has_progressive = true;
155                            } else {
156                                has_passive = true;
157                            }
158                        }
159                        _ => {
160                            has_passive = true;
161                        }
162                    }
163                }
164            }
165        }
166
167        if self.check_content_word() {
168            let word = self.interner.resolve(self.peek().lexeme).to_lowercase();
169            if word == "being" {
170                self.advance();
171                has_progressive = true;
172            }
173        }
174
175        let verb = if self.check_verb() {
176            self.consume_verb()
177        } else if self.check_content_word() {
178            self.consume_content_word()?
179        } else {
180            return Err(ParseError {
181                kind: ParseErrorKind::ExpectedContentWord { found: self.peek().kind.clone() },
182                span: self.peek().span.clone(),
183            });
184        };
185
186        let subject_role = if has_passive {
187            ThematicRole::Theme
188        } else {
189            ThematicRole::Agent
190        };
191        let mut roles: Vec<(ThematicRole, Term<'a>)> = vec![(subject_role, subject_term)];
192
193        if has_passive && self.check_preposition() {
194            if let TokenType::Preposition(sym) = self.peek().kind {
195                if self.interner.resolve(sym) == "by" {
196                    self.advance();
197                    let agent_np = self.parse_noun_phrase(true)?;
198                    let agent_term = self.noun_phrase_to_term(&agent_np);
199                    roles.push((ThematicRole::Agent, agent_term));
200                }
201            }
202        } else if !has_passive && (self.check_content_word() || self.check_article()) {
203            let obj_np = self.parse_noun_phrase(false)?;
204            let obj_term = self.noun_phrase_to_term(&obj_np);
205            roles.push((ThematicRole::Theme, obj_term));
206        }
207
208        let event_var = self.get_event_var();
209        let mut modifiers: Vec<Symbol> = Vec::new();
210        if let Some(pending) = self.pending_time {
211            match pending {
212                Time::Past => modifiers.push(self.interner.intern("Past")),
213                Time::Future => modifiers.push(self.interner.intern("Future")),
214                _ => {}
215            }
216        }
217        let suppress_existential = self.drs.in_conditional_antecedent();
218        let base_pred = self.ctx.exprs.alloc(LogicExpr::NeoEvent(Box::new(NeoEventData {
219            event_var,
220            verb,
221            roles: self.ctx.roles.alloc_slice(roles.clone()),
222            modifiers: self.ctx.syms.alloc_slice(modifiers.clone()),
223            suppress_existential,
224            world: None,
225        })));
226
227        // Capture template for ellipsis reconstruction
228        self.capture_event_template(verb, &roles, &modifiers);
229
230        let mut result: &'a LogicExpr<'a> = base_pred;
231
232        if has_progressive {
233            result = self.ctx.aspectual(AspectOperator::Progressive, result);
234        }
235
236        if has_passive {
237            result = self.ctx.voice(VoiceOperator::Passive, result);
238        }
239
240        if has_perfect {
241            result = self.ctx.aspectual(AspectOperator::Perfect, result);
242
243            // Check pending_time to set up reference time for tense
244            if let Some(pending) = self.pending_time.take() {
245                match pending {
246                    Time::Future => {
247                        // Future perfect: S < R
248                        let r_var = self.world_state.next_reference_time();
249                        self.world_state.add_time_constraint("S".to_string(), TimeRelation::Precedes, r_var);
250                    }
251                    Time::Past => {
252                        // Past perfect fallback (if not already set by "had")
253                        if self.world_state.current_reference_time() == "S" {
254                            let r_var = self.world_state.next_reference_time();
255                            self.world_state.add_time_constraint(r_var, TimeRelation::Precedes, "S".to_string());
256                        }
257                    }
258                    _ => {}
259                }
260            }
261
262            // Perfect: E < R (event before reference)
263            let e_var = format!("e{}", self.world_state.event_history().len().max(1));
264            let r_var = self.world_state.current_reference_time();
265            self.world_state.add_time_constraint(e_var, TimeRelation::Precedes, r_var);
266        }
267
268        if has_negation {
269            result = self.ctx.exprs.alloc(LogicExpr::UnaryOp {
270                op: TokenType::Not,
271                operand: result,
272            });
273        }
274
275        // Apply nested modal first (from "be able to" = ability)
276        if let Some(vector) = nested_modal_vector {
277            result = self.ctx.modal(vector, result);
278        }
279
280        // Then apply outer modal (e.g., "might")
281        if has_modal {
282            // Exit modal box in DRS (matches enter_box above)
283            self.drs.exit_box();
284            // Note: We do NOT exit_modal_context() here because we want the modal flag
285            // to persist until end_sentence() so telescope candidates are marked as modal.
286            // The modal context is cleared by end_sentence() → prior_modal_context.take()
287            if let Some(vector) = modal_vector {
288                result = self.ctx.modal(vector, result);
289            }
290        }
291
292        Ok(result)
293    }

295    fn token_to_vector(&self, token: &TokenType) -> ModalVector {
296        use crate::ast::ModalFlavor;
297        use super::ModalPreference;
298
299        match token {
300            // Root modals → Narrow Scope (De Re)
301            // These attach the modal to the predicate inside the quantifier
302            TokenType::Must => ModalVector {
303                domain: ModalDomain::Alethic,
304                force: 1.0,
305                flavor: ModalFlavor::Root,
306            },
307            TokenType::Cannot => ModalVector {
308                domain: ModalDomain::Alethic,
309                force: 0.0,
310                flavor: ModalFlavor::Root,
311            },
312
313            // Polysemous modal: CAN
314            // Default: Ability (Alethic, Root/Narrow)
315            // Deontic: Permission (Deontic, Root/Narrow)
316            TokenType::Can => {
317                match self.modal_preference {
318                    ModalPreference::Deontic => {
319                        // Permission: "You can go" (Deontic, Narrow Scope)
320                        ModalVector {
321                            domain: ModalDomain::Deontic,
322                            force: 0.5,
323                            flavor: ModalFlavor::Root,
324                        }
325                    }
326                    _ => {
327                        // Ability: "Birds can fly" (Alethic, Narrow Scope)
328                        ModalVector {
329                            domain: ModalDomain::Alethic,
330                            force: 0.5,
331                            flavor: ModalFlavor::Root,
332                        }
333                    }
334                }
335            },
336
337            // Polysemous modal: COULD
338            // Default: Past Ability (Alethic, Root/Narrow)
339            // Epistemic: Conditional Possibility (Alethic, Epistemic/Wide)
340            TokenType::Could => {
341                match self.modal_preference {
342                    ModalPreference::Epistemic => {
343                        // Conditional Possibility: "It could rain" (Alethic, Wide Scope)
344                        ModalVector {
345                            domain: ModalDomain::Alethic,
346                            force: 0.5,
347                            flavor: ModalFlavor::Epistemic,
348                        }
349                    }
350                    _ => {
351                        // Past Ability: "She could swim" (Alethic, Narrow Scope)
352                        ModalVector {
353                            domain: ModalDomain::Alethic,
354                            force: 0.5,
355                            flavor: ModalFlavor::Root,
356                        }
357                    }
358                }
359            },
360
361            TokenType::Would => ModalVector {
362                domain: ModalDomain::Alethic,
363                force: 0.5,
364                flavor: ModalFlavor::Root,
365            },
366            TokenType::Shall => ModalVector {
367                domain: ModalDomain::Deontic,
368                force: 0.9,
369                flavor: ModalFlavor::Root,
370            },
371            TokenType::Should => ModalVector {
372                domain: ModalDomain::Deontic,
373                force: 0.6,
374                flavor: ModalFlavor::Root,
375            },
376
377            // Epistemic modals → Wide Scope (De Dicto)
378            // These wrap the entire quantifier in the modal
379            TokenType::Might => ModalVector {
380                domain: ModalDomain::Alethic,
381                force: 0.3,
382                flavor: ModalFlavor::Epistemic,
383            },
384
385            // Polysemous modal: MAY
386            // Default: Permission (Deontic, Root/Narrow)
387            // Epistemic: Possibility (Alethic, Epistemic/Wide)
388            TokenType::May => {
389                match self.modal_preference {
390                    ModalPreference::Epistemic => {
391                        // Possibility: "It may rain" (Alethic, Wide Scope)
392                        ModalVector {
393                            domain: ModalDomain::Alethic,
394                            force: 0.5,
395                            flavor: ModalFlavor::Epistemic,
396                        }
397                    }
398                    _ => {
399                        // Permission: "Students may leave" (Deontic, Narrow Scope)
400                        ModalVector {
401                            domain: ModalDomain::Deontic,
402                            force: 0.5,
403                            flavor: ModalFlavor::Root,
404                        }
405                    }
406                }
407            },
408
409            _ => panic!("Unknown modal token: {:?}", token),
410        }
411    }
