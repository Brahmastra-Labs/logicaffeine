1//! Backward chaining proof engine.
2//!
3//! "The machine that crawls backward from the Conclusion to the Axioms."
4//!
5//! This module implements the core proof search algorithm. It takes inference
6//! rules and *hunts* for proofs using backward chaining and unification.
7//!
8//! ## Backward Chaining Strategy
9//!
10//! 1. Start with the goal we want to prove
11//! 2. Find rules whose conclusions unify with our goal
12//! 3. Recursively prove the premises of those rules
13//! 4. Build the derivation tree as we succeed
14//!
15//! ## Example
16//!
17//! ```text
18//! Goal: Mortal(socrates)
19//!
20//! Knowledge Base:
21//!   - Human(socrates)
22//!   - ∀x(Human(x) → Mortal(x))
23//!
24//! Search:
25//!   1. Goal matches conclusion of ∀x(Human(x) → Mortal(x)) with x=socrates
26//!   2. New subgoal: Human(socrates)
27//!   3. Human(socrates) matches knowledge base fact
28//!   4. Build derivation tree: ModusPonens(UniversalInst, PremiseMatch)
29//! ```

60/// Convert a ProofTerm to a ProofExpr for reduction.
61///
62/// Terms embed into expressions as atoms or constructors.
63fn term_to_expr(term: &ProofTerm) -> ProofExpr {
64    match term {
65        ProofTerm::Constant(s) => ProofExpr::Atom(s.clone()),
66        ProofTerm::Variable(s) => ProofExpr::Atom(s.clone()),
67        ProofTerm::BoundVarRef(s) => ProofExpr::Atom(s.clone()),
68        ProofTerm::Function(name, args) => {
69            // Check if this is a known constructor
70            if matches!(name.as_str(), "Zero" | "Succ" | "Nil" | "Cons") {
71                ProofExpr::Ctor {
72                    name: name.clone(),
73                    args: args.iter().map(term_to_expr).collect(),
74                }
75            } else {
76                // Otherwise it's a predicate/function
77                ProofExpr::Predicate {
78                    name: name.clone(),
79                    args: args.clone(),
80                    world: None,
81                }
82            }
83        }
84        ProofTerm::Group(terms) => {
85            // Groups become nested predicates or just the single element
86            if terms.len() == 1 {
87                term_to_expr(&terms[0])
88            } else {
89                // Multi-term groups - convert to predicate
90                ProofExpr::Predicate {
91                    name: "Group".into(),
92                    args: terms.clone(),
93                    world: None,
94                }
95            }
96        }
97    }
98}

100/// Check if two expressions are structurally equal.
101///
102/// This is syntactic equality after normalization - no unification needed.
103fn exprs_structurally_equal(left: &ProofExpr, right: &ProofExpr) -> bool {
104    match (left, right) {
105        (ProofExpr::Atom(a), ProofExpr::Atom(b)) => a == b,
106
107        (ProofExpr::Ctor { name: n1, args: a1 }, ProofExpr::Ctor { name: n2, args: a2 }) => {
108            n1 == n2 && a1.len() == a2.len() && a1.iter().zip(a2).all(|(x, y)| exprs_structurally_equal(x, y))
109        }
110
111        (
112            ProofExpr::Predicate { name: n1, args: a1, .. },
113            ProofExpr::Predicate { name: n2, args: a2, .. },
114        ) => n1 == n2 && a1.len() == a2.len() && a1.iter().zip(a2).all(|(x, y)| terms_structurally_equal(x, y)),
115
116        (ProofExpr::Identity(l1, r1), ProofExpr::Identity(l2, r2)) => {
117            terms_structurally_equal(l1, l2) && terms_structurally_equal(r1, r2)
118        }
119
120        (ProofExpr::And(l1, r1), ProofExpr::And(l2, r2))
121        | (ProofExpr::Or(l1, r1), ProofExpr::Or(l2, r2))
122        | (ProofExpr::Implies(l1, r1), ProofExpr::Implies(l2, r2))
123        | (ProofExpr::Iff(l1, r1), ProofExpr::Iff(l2, r2)) => {
124            exprs_structurally_equal(l1, l2) && exprs_structurally_equal(r1, r2)
125        }
126
127        (ProofExpr::Not(a), ProofExpr::Not(b)) => exprs_structurally_equal(a, b),
128
129        (
130            ProofExpr::ForAll { variable: v1, body: b1 },
131            ProofExpr::ForAll { variable: v2, body: b2 },
132        )
133        | (
134            ProofExpr::Exists { variable: v1, body: b1 },
135            ProofExpr::Exists { variable: v2, body: b2 },
136        ) => v1 == v2 && exprs_structurally_equal(b1, b2),
137
138        (
139            ProofExpr::Lambda { variable: v1, body: b1 },
140            ProofExpr::Lambda { variable: v2, body: b2 },
141        ) => v1 == v2 && exprs_structurally_equal(b1, b2),
142
143        (ProofExpr::App(f1, a1), ProofExpr::App(f2, a2)) => {
144            exprs_structurally_equal(f1, f2) && exprs_structurally_equal(a1, a2)
145        }
146
147        (
148            ProofExpr::TypedVar { name: n1, typename: t1 },
149            ProofExpr::TypedVar { name: n2, typename: t2 },
150        ) => n1 == n2 && t1 == t2,
151
152        (
153            ProofExpr::Fixpoint { name: n1, body: b1 },
154            ProofExpr::Fixpoint { name: n2, body: b2 },
155        ) => n1 == n2 && exprs_structurally_equal(b1, b2),
156
157        _ => false,
158    }
159}

161/// Check if two terms are structurally equal.
162fn terms_structurally_equal(left: &ProofTerm, right: &ProofTerm) -> bool {
163    match (left, right) {
164        (ProofTerm::Constant(a), ProofTerm::Constant(b)) => a == b,
165        (ProofTerm::Variable(a), ProofTerm::Variable(b)) => a == b,
166        (ProofTerm::BoundVarRef(a), ProofTerm::BoundVarRef(b)) => a == b,
167        (ProofTerm::Function(n1, a1), ProofTerm::Function(n2, a2)) => {
168            n1 == n2 && a1.len() == a2.len() && a1.iter().zip(a2).all(|(x, y)| terms_structurally_equal(x, y))
169        }
170        (ProofTerm::Group(a1), ProofTerm::Group(a2)) => {
171            a1.len() == a2.len() && a1.iter().zip(a2).all(|(x, y)| terms_structurally_equal(x, y))
172        }
173        _ => false,
174    }
175}

197    /// Add an axiom/fact/rule to the knowledge base.
198    ///
199    /// Event semantics are automatically abstracted to simple predicates for efficient proof search.
200    pub fn add_axiom(&mut self, expr: ProofExpr) {
201        // Pre-process: abstract event semantics to simple predicates
202        let abstracted = self.abstract_all_events(&expr);
203        // Simplify definite description conjunctions (e.g., butler(butler) ∧ P → P)
204        let simplified = self.simplify_definite_description_conjunction(&abstracted);
205        self.knowledge_base.push(simplified);
206    }

208    /// Attempt to prove a goal.
209    ///
210    /// Returns a derivation tree if successful, explaining how the proof was constructed.
211    /// Event semantics in the goal are automatically abstracted (but De Morgan is not applied
212    /// to preserve goal pattern matching for reductio strategies).
213    pub fn prove(&mut self, goal: ProofExpr) -> ProofResult<DerivationTree> {
214        // Pre-process: unify definite descriptions across all axioms
215        // This handles Russell's theory of definite descriptions, where multiple
216        // "the X" references should refer to the same entity.
217        self.unify_definite_descriptions();
218
219        // Pre-process: abstract event semantics in the goal
220        // Use abstract_events_only which doesn't apply De Morgan (to preserve ¬∃ pattern)
221        let abstracted_goal = self.abstract_events_only(&goal);
222        // Simplify definite description conjunctions
223        let normalized_goal = self.simplify_definite_description_conjunction(&abstracted_goal);
224        self.prove_goal(ProofGoal::new(normalized_goal), 0)
225    }

227    /// Unify definite descriptions across axioms.
228    ///
229    /// When multiple axioms contain the same definite description pattern
230    /// (e.g., "the barber" creates `∃x ((barber(x) ∧ ∀y (barber(y) → y=x)) ∧ P(x))`),
231    /// this function:
232    /// 1. Identifies all axioms with the same defining predicate
233    /// 2. Extracts the properties attributed to the definite description
234    /// 3. Replaces them with a unified Skolem constant and extracted properties
235    fn unify_definite_descriptions(&mut self) {
236        // Collect definite descriptions by their defining predicate
237        let mut definite_descs: std::collections::HashMap<String, Vec<(usize, String, ProofExpr)>> = std::collections::HashMap::new();
238
239        for (idx, axiom) in self.knowledge_base.iter().enumerate() {
240            if let Some((pred_name, var_name, property)) = self.extract_definite_description(axiom) {
241                definite_descs.entry(pred_name).or_default().push((idx, var_name, property));
242            }
243        }
244
245        // For each group of definite descriptions with the same predicate
246        for (pred_name, descs) in definite_descs {
247            if descs.is_empty() {
248                continue;
249            }
250
251            // Create a unified Skolem constant for this definite description
252            let skolem_name = format!("the_{}", pred_name);
253            let skolem_const = ProofTerm::Constant(skolem_name.clone());
254
255            // Add the defining property: pred(skolem)
256            let defining_fact = ProofExpr::Predicate {
257                name: pred_name.clone(),
258                args: vec![skolem_const.clone()],
259                world: None,
260            };
261            self.knowledge_base.push(defining_fact);
262
263            // CRITICAL: Add uniqueness constraint: ∀y (pred(y) → y = skolem)
264            // This is essential for proofs that assume ∃x pred(x) - they need to
265            // unify their Skolem constant with our unified constant.
266            let uniqueness = ProofExpr::ForAll {
267                variable: "_u".to_string(),
268                body: Box::new(ProofExpr::Implies(
269                    Box::new(ProofExpr::Predicate {
270                        name: pred_name.clone(),
271                        args: vec![ProofTerm::Variable("_u".to_string())],
272                        world: None,
273                    }),
274                    Box::new(ProofExpr::Identity(
275                        ProofTerm::Variable("_u".to_string()),
276                        skolem_const.clone(),
277                    )),
278                )),
279            };
280            self.knowledge_base.push(uniqueness);
281
282            // Replace axioms with the extracted properties
283            let mut indices_to_remove: Vec<usize> = Vec::new();
284            for (idx, var_name, property) in descs {
285                // Substitute the original variable with the Skolem constant
286                let substituted = self.substitute_term_in_expr(
287                    &property,
288                    &ProofTerm::Variable(var_name),
289                    &skolem_const,
290                );
291                // Normalize the property (especially for ∀x ¬(P ∧ Q) → ∀x (P → ¬Q))
292                let normalized = self.normalize_for_proof(&substituted);
293                self.knowledge_base.push(normalized);
294                indices_to_remove.push(idx);
295            }
296
297            // Remove the original existential axioms (in reverse order to preserve indices)
298            indices_to_remove.sort_unstable_by(|a, b| b.cmp(a));
299            for idx in indices_to_remove {
300                self.knowledge_base.remove(idx);
301            }
302        }
303    }

305    /// Normalize an expression for proof search.
306    ///
307    /// Applies transformations like: ∀x ¬(P ∧ Q) → ∀x (P → ¬Q)
308    fn normalize_for_proof(&self, expr: &ProofExpr) -> ProofExpr {
309        match expr {
310            ProofExpr::ForAll { variable, body } => {
311                // Check for pattern: ∀x ¬(P ∧ Q) → ∀x (P → ¬Q)
312                if let ProofExpr::Not(inner) = body.as_ref() {
313                    if let ProofExpr::And(left, right) = inner.as_ref() {
314                        return ProofExpr::ForAll {
315                            variable: variable.clone(),
316                            body: Box::new(ProofExpr::Implies(
317                                Box::new(self.normalize_for_proof(left)),
318                                Box::new(ProofExpr::Not(Box::new(self.normalize_for_proof(right)))),
319                            )),
320                        };
321                    }
322                }
323                ProofExpr::ForAll {
324                    variable: variable.clone(),
325                    body: Box::new(self.normalize_for_proof(body)),
326                }
327            }
328            ProofExpr::And(left, right) => ProofExpr::And(
329                Box::new(self.normalize_for_proof(left)),
330                Box::new(self.normalize_for_proof(right)),
331            ),
332            ProofExpr::Or(left, right) => ProofExpr::Or(
333                Box::new(self.normalize_for_proof(left)),
334                Box::new(self.normalize_for_proof(right)),
335            ),
336            ProofExpr::Implies(left, right) => ProofExpr::Implies(
337                Box::new(self.normalize_for_proof(left)),
338                Box::new(self.normalize_for_proof(right)),
339            ),
340            ProofExpr::Not(inner) => ProofExpr::Not(Box::new(self.normalize_for_proof(inner))),
341            ProofExpr::Exists { variable, body } => ProofExpr::Exists {
342                variable: variable.clone(),
343                body: Box::new(self.normalize_for_proof(body)),
344            },
345            other => other.clone(),
346        }
347    }

349    /// Extract a definite description from an axiom.
350    ///
351    /// Pattern: ∃x ((P(x) ∧ ∀y (P(y) → y = x)) ∧ Q(x))
352    /// Returns: Some((predicate_name, variable_name, Q(x)))
353    fn extract_definite_description(&self, expr: &ProofExpr) -> Option<(String, String, ProofExpr)> {
354        // Match: ∃x (body)
355        let (var, body) = match expr {
356            ProofExpr::Exists { variable, body } => (variable.clone(), body.as_ref()),
357            _ => return None,
358        };
359
360        // Match: (defining_part ∧ property)
361        let (defining_part, property) = match body {
362            ProofExpr::And(left, right) => (left.as_ref(), right.as_ref().clone()),
363            _ => return None,
364        };
365
366        // Match defining_part: (P(x) ∧ ∀y (P(y) → y = x))
367        let (type_pred, uniqueness) = match defining_part {
368            ProofExpr::And(left, right) => (left.as_ref(), right.as_ref()),
369            _ => return None,
370        };
371
372        // Extract predicate name from P(x)
373        let pred_name = match type_pred {
374            ProofExpr::Predicate { name, args, .. } if args.len() == 1 => {
375                // Verify the arg is our variable
376                if let ProofTerm::Variable(v) = &args[0] {
377                    if v == &var {
378                        name.clone()
379                    } else {
380                        return None;
381                    }
382                } else {
383                    return None;
384                }
385            }
386            _ => return None,
387        };
388
389        // Verify uniqueness constraint: ∀y (P(y) → y = x)
390        match uniqueness {
391            ProofExpr::ForAll { variable: _, body: inner_body } => {
392                match inner_body.as_ref() {
393                    ProofExpr::Implies(ante, cons) => {
394                        // Verify antecedent is P(y)
395                        if let ProofExpr::Predicate { name, .. } = ante.as_ref() {
396                            if name != &pred_name {
397                                return None;
398                            }
399                        } else {
400                            return None;
401                        }
402                        // Verify consequent is an identity (y = x)
403                        if !matches!(cons.as_ref(), ProofExpr::Identity(_, _)) {
404                            return None;
405                        }
406                    }
407                    _ => return None,
408                }
409            }
410            _ => return None,
411        }
412
413        Some((pred_name, var, property))
414    }

416    /// Internal proof search with depth tracking.
417    fn prove_goal(&mut self, goal: ProofGoal, depth: usize) -> ProofResult<DerivationTree> {
418        // Check depth limit
419        if depth > self.max_depth {
420            return Err(ProofError::DepthExceeded);
421        }
422
423        // PRIORITY: Check for inductive goals FIRST
424        // Goals with TypedVar (e.g., n:Nat) require structural induction,
425        // not direct unification which would incorrectly ground the variable.
426        if let Some((_, typename)) = self.find_typed_var(&goal.target) {
427            // For known inductive types, require induction to succeed
428            // Falling back to direct matching would incorrectly unify the TypedVar
429            let is_known_inductive = matches!(typename.as_str(), "Nat" | "List");
430
431            if let Some(tree) = self.try_structural_induction(&goal, depth)? {
432                return Ok(tree);
433            }
434
435            // For known inductive types, if induction fails, the proof fails
436            // (don't allow incorrect direct unification)
437            if is_known_inductive {
438                return Err(ProofError::NoProofFound);
439            }
440            // For unknown types, fall through to other strategies
441        }
442
443        // Strategy 0: Reflexivity by computation
444        // Try to prove a = b by normalizing both sides
445        if let Some(tree) = self.try_reflexivity(&goal)? {
446            return Ok(tree);
447        }
448
449        // Strategy 1: Direct fact matching
450        if let Some(tree) = self.try_match_fact(&goal)? {
451            return Ok(tree);
452        }
453
454        // Strategy 2: Introduction rules (structural decomposition)
455        if let Some(tree) = self.try_intro_rules(&goal, depth)? {
456            return Ok(tree);
457        }
458
459        // Strategy 3: Backward chaining on implications
460        if let Some(tree) = self.try_backward_chain(&goal, depth)? {
461            return Ok(tree);
462        }
463
464        // Strategy 3b: Modus Tollens (from P → Q and ¬Q, derive ¬P)
465        if let Some(tree) = self.try_modus_tollens(&goal, depth)? {
466            return Ok(tree);
467        }
468
469        // Strategy 4: Universal instantiation
470        if let Some(tree) = self.try_universal_inst(&goal, depth)? {
471            return Ok(tree);
472        }
473
474        // Strategy 5: Existential introduction
475        if let Some(tree) = self.try_existential_intro(&goal, depth)? {
476            return Ok(tree);
477        }
478
479        // Strategy 5b: Disjunction elimination (disjunctive syllogism)
480        if let Some(tree) = self.try_disjunction_elimination(&goal, depth)? {
481            return Ok(tree);
482        }
483
484        // Strategy 5c: Proof by contradiction (reductio ad absurdum)
485        // For negation goals, assume the positive and derive contradiction
486        if let Some(tree) = self.try_reductio_ad_absurdum(&goal, depth)? {
487            return Ok(tree);
488        }
489
490        // Strategy 5d: Existential elimination from premises
491        // Extract witnesses from ∃x P(x) premises and add to context
492        if let Some(tree) = self.try_existential_elimination(&goal, depth)? {
493            return Ok(tree);
494        }
495
496        // Strategy 6: Equality rewriting (Leibniz's Law)
497        if let Some(tree) = self.try_equality_rewrite(&goal, depth)? {
498            return Ok(tree);
499        }
500
501        // Strategy 7: Oracle fallback (Z3)
502        #[cfg(feature = "verification")]
503        if let Some(tree) = self.try_oracle_fallback(&goal)? {
504            return Ok(tree);
505        }
506
507        // No proof found
508        Err(ProofError::NoProofFound)
509    }

515    /// Try to prove an identity a = b by normalizing both sides.
516    ///
517    /// If both sides reduce to structurally identical expressions,
518    /// the proof is by reflexivity (a = a).
519    fn try_reflexivity(&self, goal: &ProofGoal) -> ProofResult<Option<DerivationTree>> {
520        if let ProofExpr::Identity(left_term, right_term) = &goal.target {
521            // Convert terms to expressions for reduction
522            let left_expr = term_to_expr(left_term);
523            let right_expr = term_to_expr(right_term);
524
525            // Normalize both sides using full reduction (beta + iota + fix)
526            let left_normal = beta_reduce(&left_expr);
527            let right_normal = beta_reduce(&right_expr);
528
529            // Check structural equality after normalization
530            if exprs_structurally_equal(&left_normal, &right_normal) {
531                return Ok(Some(DerivationTree::leaf(
532                    goal.target.clone(),
533                    InferenceRule::Reflexivity,
534                )));
535            }
536        }
537        Ok(None)
538    }

544    /// Try to match the goal directly against a fact in the knowledge base.
545    fn try_match_fact(&self, goal: &ProofGoal) -> ProofResult<Option<DerivationTree>> {
546        // Also check local context
547        for fact in goal.context.iter().chain(self.knowledge_base.iter()) {
548            if let Ok(subst) = unify_exprs(&goal.target, fact) {
549                return Ok(Some(
550                    DerivationTree::leaf(goal.target.clone(), InferenceRule::PremiseMatch)
551                        .with_substitution(subst),
552                ));
553            }
554        }
555        Ok(None)
556    }

562    /// Try introduction rules based on the goal's structure.
563    fn try_intro_rules(
564        &mut self,
565        goal: &ProofGoal,
566        depth: usize,
567    ) -> ProofResult<Option<DerivationTree>> {
568        match &goal.target {
569            // Conjunction Introduction: To prove A ∧ B, prove A and prove B
570            ProofExpr::And(left, right) => {
571                let left_goal = ProofGoal::with_context((**left).clone(), goal.context.clone());
572                let right_goal = ProofGoal::with_context((**right).clone(), goal.context.clone());
573
574                // Try to prove both sides
575                if let (Ok(left_proof), Ok(right_proof)) = (
576                    self.prove_goal(left_goal, depth + 1),
577                    self.prove_goal(right_goal, depth + 1),
578                ) {
579                    return Ok(Some(DerivationTree::new(
580                        goal.target.clone(),
581                        InferenceRule::ConjunctionIntro,
582                        vec![left_proof, right_proof],
583                    )));
584                }
585            }
586
587            // Disjunction Introduction: To prove A ∨ B, prove A or prove B
588            ProofExpr::Or(left, right) => {
589                // Try left side first
590                let left_goal = ProofGoal::with_context((**left).clone(), goal.context.clone());
591                if let Ok(left_proof) = self.prove_goal(left_goal, depth + 1) {
592                    return Ok(Some(DerivationTree::new(
593                        goal.target.clone(),
594                        InferenceRule::DisjunctionIntro,
595                        vec![left_proof],
596                    )));
597                }
598
599                // Try right side
600                let right_goal = ProofGoal::with_context((**right).clone(), goal.context.clone());
601                if let Ok(right_proof) = self.prove_goal(right_goal, depth + 1) {
602                    return Ok(Some(DerivationTree::new(
603                        goal.target.clone(),
604                        InferenceRule::DisjunctionIntro,
605                        vec![right_proof],
606                    )));
607                }
608            }
609
610            // Double Negation: To prove ¬¬P, prove P
611            ProofExpr::Not(inner) => {
612                if let ProofExpr::Not(core) = &**inner {
613                    let core_goal = ProofGoal::with_context((**core).clone(), goal.context.clone());
614                    if let Ok(core_proof) = self.prove_goal(core_goal, depth + 1) {
615                        return Ok(Some(DerivationTree::new(
616                            goal.target.clone(),
617                            InferenceRule::DoubleNegation,
618                            vec![core_proof],
619                        )));
620                    }
621                }
622            }
623
624            _ => {}
625        }
626
627        Ok(None)
628    }

634    /// Try backward chaining: find P → Goal in KB, then prove P.
635    fn try_backward_chain(
636        &mut self,
637        goal: &ProofGoal,
638        depth: usize,
639    ) -> ProofResult<Option<DerivationTree>> {
640        // Collect implications from KB (we need to clone to avoid borrow issues)
641        let implications: Vec<(usize, ProofExpr)> = self
642            .knowledge_base
643            .iter()
644            .enumerate()
645            .filter_map(|(idx, expr)| {
646                if let ProofExpr::Implies(_, _) = expr {
647                    Some((idx, expr.clone()))
648                } else {
649                    None
650                }
651            })
652            .collect();
653
654        for (idx, impl_expr) in implications {
655            if let ProofExpr::Implies(antecedent, consequent) = &impl_expr {
656                // Rename variables to avoid capture
657                let renamed = self.rename_variables(&impl_expr);
658                if let ProofExpr::Implies(ant, con) = renamed {
659                    // Try to unify the consequent with our goal
660                    if let Ok(subst) = unify_exprs(&goal.target, &con) {
661                        // Apply substitution to the antecedent
662                        let new_antecedent = apply_subst_to_expr(&ant, &subst);
663
664                        // Try to prove the antecedent
665                        let ant_goal =
666                            ProofGoal::with_context(new_antecedent, goal.context.clone());
667
668                        if let Ok(ant_proof) = self.prove_goal(ant_goal, depth + 1) {
669                            // Success! Build the modus ponens tree
670                            let impl_leaf = DerivationTree::leaf(
671                                impl_expr.clone(),
672                                InferenceRule::PremiseMatch,
673                            );
674
675                            return Ok(Some(
676                                DerivationTree::new(
677                                    goal.target.clone(),
678                                    InferenceRule::ModusPonens,
679                                    vec![impl_leaf, ant_proof],
680                                )
681                                .with_substitution(subst),
682                            ));
683                        }
684                    }
685                }
686            }
687        }
688
689        Ok(None)
690    }

694    // =========================================================================
695
696    /// Try Modus Tollens: from P → Q and ¬Q, derive ¬P.
697    ///
698    /// If the goal is ¬P:
699    /// 1. Look for implications P → Q in the KB
700    /// 2. Check if ¬Q is known (in KB or context) OR can be proved
701    /// 3. If so, derive ¬P by Modus Tollens
702    fn try_modus_tollens(
703        &mut self,
704        goal: &ProofGoal,
705        depth: usize,
706    ) -> ProofResult<Option<DerivationTree>> {
707        // Modus Tollens only applies when goal is a negation: ¬P
708        let inner_goal = match &goal.target {
709            ProofExpr::Not(inner) => (**inner).clone(),
710            _ => return Ok(None),
711        };
712
713        // Collect all implications from KB, including those inside ForAll
714        let implications: Vec<ProofExpr> = self
715            .knowledge_base
716            .iter()
717            .chain(goal.context.iter())
718            .flat_map(|expr| {
719                match expr {
720                    ProofExpr::Implies(_, _) => vec![expr.clone()],
721                    ProofExpr::ForAll { body, .. } => {
722                        // Extract implications from inside universal quantifiers
723                        if let ProofExpr::Implies(_, _) = body.as_ref() {
724                            vec![*body.clone()]
725                        } else {
726                            vec![]
727                        }
728                    }
729                    _ => vec![],
730                }
731            })
732            .collect();
733
734        // Collect all negations from KB and context (for direct matching)
735        let negations: Vec<ProofExpr> = self
736            .knowledge_base
737            .iter()
738            .chain(goal.context.iter())
739            .filter_map(|expr| {
740                if let ProofExpr::Not(inner) = expr {
741                    Some((**inner).clone())
742                } else {
743                    None
744                }
745            })
746            .collect();
747
748        // For each implication P → Q
749        for impl_expr in &implications {
750            if let ProofExpr::Implies(antecedent, consequent) = impl_expr {
751                // Check if the antecedent P matches our inner goal (we want to prove ¬P)
752                if let Ok(subst) = unify_exprs(&inner_goal, antecedent) {
753                    // Apply substitution to the consequent Q
754                    let q = apply_subst_to_expr(consequent, &subst);
755
756                    // First, check if ¬Q is directly in our known facts
757                    for negated in &negations {
758                        if exprs_structurally_equal(negated, &q) {
759                            // We have P → Q and ¬Q, so we can derive ¬P
760                            let impl_leaf = DerivationTree::leaf(
761                                impl_expr.clone(),
762                                InferenceRule::PremiseMatch,
763                            );
764                            let neg_q_leaf = DerivationTree::leaf(
765                                ProofExpr::Not(Box::new(q.clone())),
766                                InferenceRule::PremiseMatch,
767                            );
768
769                            return Ok(Some(
770                                DerivationTree::new(
771                                    goal.target.clone(),
772                                    InferenceRule::ModusTollens,
773                                    vec![impl_leaf, neg_q_leaf],
774                                )
775                                .with_substitution(subst),
776                            ));
777                        }
778                    }
779
780                    // Second, try to prove ¬Q recursively (for chaining)
781                    let neg_q_goal = ProofGoal::with_context(
782                        ProofExpr::Not(Box::new(q.clone())),
783                        goal.context.clone(),
784                    );
785
786                    if let Ok(neg_q_proof) = self.prove_goal(neg_q_goal, depth + 1) {
787                        // We proved ¬Q, so we can derive ¬P
788                        let impl_leaf = DerivationTree::leaf(
789                            impl_expr.clone(),
790                            InferenceRule::PremiseMatch,
791                        );
792
793                        return Ok(Some(
794                            DerivationTree::new(
795                                goal.target.clone(),
796                                InferenceRule::ModusTollens,
797                                vec![impl_leaf, neg_q_proof],
798                            )
799                            .with_substitution(subst),
800                        ));
801                    }
802                }
803            }
804        }
805
806        Ok(None)
807    }

813    /// Try universal instantiation: if KB has ∀x.P(x), try to prove P(t) for some term t.
814    fn try_universal_inst(
815        &mut self,
816        goal: &ProofGoal,
817        depth: usize,
818    ) -> ProofResult<Option<DerivationTree>> {
819        // Look for universal quantifiers in KB
820        let universals: Vec<(usize, ProofExpr)> = self
821            .knowledge_base
822            .iter()
823            .enumerate()
824            .filter_map(|(idx, expr)| {
825                if let ProofExpr::ForAll { .. } = expr {
826                    Some((idx, expr.clone()))
827                } else {
828                    None
829                }
830            })
831            .collect();
832
833        for (idx, forall_expr) in universals {
834            if let ProofExpr::ForAll { variable, body } = &forall_expr {
835                // Rename to avoid capture
836                let renamed = self.rename_variables(&forall_expr);
837                if let ProofExpr::ForAll {
838                    variable: var,
839                    body: renamed_body,
840                } = renamed
841                {
842                    // Try to unify the body with our goal
843                    if let Ok(subst) = unify_exprs(&goal.target, &renamed_body) {
844                        // Check if we found a value for the quantified variable
845                        if let Some(witness) = subst.get(&var) {
846                            let witness_str = format!("{}", witness);
847
848                            // The universal premise
849                            let universal_leaf = DerivationTree::leaf(
850                                forall_expr.clone(),
851                                InferenceRule::PremiseMatch,
852                            );
853
854                            return Ok(Some(
855                                DerivationTree::new(
856                                    goal.target.clone(),
857                                    InferenceRule::UniversalInst(witness_str),
858                                    vec![universal_leaf],
859                                )
860                                .with_substitution(subst),
861                            ));
862                        }
863                    }
864
865                    // Also try: if the body is an implication (∀x(P(x) → Q(x))),
866                    // and our goal is Q(t), try to prove P(t)
867                    if let ProofExpr::Implies(ant, con) = &*renamed_body {
868                        if let Ok(subst) = unify_exprs(&goal.target, con) {
869                            // Found a match! Now prove the antecedent
870                            let new_antecedent = apply_subst_to_expr(ant, &subst);
871
872                            let ant_goal =
873                                ProofGoal::with_context(new_antecedent, goal.context.clone());
874
875                            if let Ok(ant_proof) = self.prove_goal(ant_goal, depth + 1) {
876                                // Get the witness from substitution
877                                let witness_str = subst
878                                    .get(&var)
879                                    .map(|t| format!("{}", t))
880                                    .unwrap_or_else(|| var.clone());
881
882                                // Build the proof tree
883                                let universal_leaf = DerivationTree::leaf(
884                                    forall_expr.clone(),
885                                    InferenceRule::PremiseMatch,
886                                );
887
888                                let inst_node = DerivationTree::new(
889                                    apply_subst_to_expr(&renamed_body, &subst),
890                                    InferenceRule::UniversalInst(witness_str),
891                                    vec![universal_leaf],
892                                );
893
894                                return Ok(Some(
895                                    DerivationTree::new(
896                                        goal.target.clone(),
897                                        InferenceRule::ModusPonens,
898                                        vec![inst_node, ant_proof],
899                                    )
900                                    .with_substitution(subst),
901                                ));
902                            }
903                        }
904                    }
905                }
906            }
907        }
908
909        Ok(None)
910    }

916    /// Try existential introduction: to prove ∃x.P(x), find a witness t and prove P(t).
917    fn try_existential_intro(
918        &mut self,
919        goal: &ProofGoal,
920        depth: usize,
921    ) -> ProofResult<Option<DerivationTree>> {
922        if let ProofExpr::Exists { variable, body } = &goal.target {
923            // We need to find a witness that makes the body true
924            // Try each constant/ground term in our KB as a potential witness
925            let witnesses = self.collect_witnesses();
926
927            for witness in witnesses {
928                // Create a substitution mapping the variable to the witness
929                let mut subst = Substitution::new();
930                subst.insert(variable.clone(), witness.clone());
931
932                // Apply substitution to get the instantiated body
933                let instantiated = apply_subst_to_expr(body, &subst);
934
935                // Try to prove the instantiated body
936                let inst_goal = ProofGoal::with_context(instantiated, goal.context.clone());
937
938                if let Ok(body_proof) = self.prove_goal(inst_goal, depth + 1) {
939                    let witness_str = format!("{}", witness);
940                    // Extract witness type from body if available, otherwise default to Nat
941                    let witness_type = extract_type_from_exists_body(body)
942                        .unwrap_or_else(|| "Nat".to_string());
943                    return Ok(Some(DerivationTree::new(
944                        goal.target.clone(),
945                        InferenceRule::ExistentialIntro {
946                            witness: witness_str,
947                            witness_type,
948                        },
949                        vec![body_proof],
950                    )));
951                }
952            }
953        }
954
955        Ok(None)
956    }

962    /// Try disjunction elimination: if KB has A ∨ B and ¬A, conclude B (and vice versa).
963    ///
964    /// Disjunctive syllogism:
965    /// - From A ∨ B and ¬A, derive B
966    /// - From A ∨ B and ¬B, derive A
967    fn try_disjunction_elimination(
968        &mut self,
969        goal: &ProofGoal,
970        _depth: usize,
971    ) -> ProofResult<Option<DerivationTree>> {
972        // Collect disjunctions from KB and context
973        let disjunctions: Vec<(ProofExpr, ProofExpr, ProofExpr)> = self
974            .knowledge_base
975            .iter()
976            .chain(goal.context.iter())
977            .filter_map(|expr| {
978                if let ProofExpr::Or(left, right) = expr {
979                    Some((expr.clone(), (**left).clone(), (**right).clone()))
980                } else {
981                    None
982                }
983            })
984            .collect();
985
986        // Collect negations from KB and context
987        let negations: Vec<ProofExpr> = self
988            .knowledge_base
989            .iter()
990            .chain(goal.context.iter())
991            .filter_map(|expr| {
992                if let ProofExpr::Not(inner) = expr {
993                    Some((**inner).clone())
994                } else {
995                    None
996                }
997            })
998            .collect();
999
1000        // For each disjunction A ∨ B
1001        for (disj_expr, left, right) in &disjunctions {
1002            // Check if ¬left is in KB (so right must be true)
1003            for negated in &negations {
1004                if exprs_structurally_equal(negated, left) {
1005                    // We have A ∨ B and ¬A, so B is true
1006                    // Check if B matches our goal
1007                    if let Ok(subst) = unify_exprs(&goal.target, right) {
1008                        let disj_leaf = DerivationTree::leaf(
1009                            disj_expr.clone(),
1010                            InferenceRule::PremiseMatch,
1011                        );
1012                        let neg_leaf = DerivationTree::leaf(
1013                            ProofExpr::Not(Box::new(left.clone())),
1014                            InferenceRule::PremiseMatch,
1015                        );
1016                        return Ok(Some(
1017                            DerivationTree::new(
1018                                goal.target.clone(),
1019                                InferenceRule::DisjunctionElim,
1020                                vec![disj_leaf, neg_leaf],
1021                            )
1022                            .with_substitution(subst),
1023                        ));
1024                    }
1025                }
1026
1027                // Check if ¬right is in KB (so left must be true)
1028                if exprs_structurally_equal(negated, right) {
1029                    // We have A ∨ B and ¬B, so A is true
1030                    // Check if A matches our goal
1031                    if let Ok(subst) = unify_exprs(&goal.target, left) {
1032                        let disj_leaf = DerivationTree::leaf(
1033                            disj_expr.clone(),
1034                            InferenceRule::PremiseMatch,
1035                        );
1036                        let neg_leaf = DerivationTree::leaf(
1037                            ProofExpr::Not(Box::new(right.clone())),
1038                            InferenceRule::PremiseMatch,
1039                        );
1040                        return Ok(Some(
1041                            DerivationTree::new(
1042                                goal.target.clone(),
1043                                InferenceRule::DisjunctionElim,
1044                                vec![disj_leaf, neg_leaf],
1045                            )
1046                            .with_substitution(subst),
1047                        ));
1048                    }
1049                }
1050            }
1051        }
1052
1053        Ok(None)
1054    }

1056    // =========================================================================
1057    // STRATEGY 5c: PROOF BY CONTRADICTION (REDUCTIO AD ABSURDUM)
1058    // =========================================================================
1059
1060    /// Try proof by contradiction: to prove ¬P, assume P and derive a contradiction.
1061    ///
1062    /// This implements reductio ad absurdum:
1063    /// 1. To prove ¬∃x P(x), assume ∃x P(x), derive contradiction, conclude ¬∃x P(x)
1064    /// 2. To prove ¬P, assume P, derive contradiction, conclude ¬P
1065    ///
1066    /// A contradiction is detected when both Q and ¬Q are derivable.
1067    fn try_reductio_ad_absurdum(
1068        &mut self,
1069        goal: &ProofGoal,
1070        depth: usize,
1071    ) -> ProofResult<Option<DerivationTree>> {
1072        // Only apply to negation goals
1073        let assumed = match &goal.target {
1074            ProofExpr::Not(inner) => (**inner).clone(),
1075            _ => return Ok(None),
1076        };
1077
1078        // Aggressive depth limit - reductio is expensive
1079        if depth > 5 {
1080            return Ok(None);
1081        }
1082
1083        // Special handling for existence negation goals: ¬∃x P(x)
1084        // This is crucial for paradoxes like the Barber Paradox
1085        if let ProofExpr::Exists { .. } = &assumed {
1086            return self.try_existence_negation_proof(&goal, &assumed, depth);
1087        }
1088
1089        // For non-existence goals, skip if they contain other quantifiers
1090        // (to avoid infinite loops with universal instantiation)
1091        if self.contains_quantifier(&assumed) {
1092            return Ok(None);
1093        }
1094
1095        // Create a temporary context with the assumption added
1096        let mut extended_context = goal.context.clone();
1097        extended_context.push(assumed.clone());
1098
1099        // Also Skolemize existentials from the assumption (but be careful)
1100        let skolemized = self.skolemize_existential(&assumed);
1101        for sk in &skolemized {
1102            extended_context.push(sk.clone());
1103        }
1104
1105        // Look for contradiction in the extended context + KB
1106        // Note: find_contradiction does NOT call prove_goal recursively
1107        if let Some(contradiction_proof) = self.find_contradiction(&extended_context, depth)? {
1108            // Found a contradiction! Build the reductio proof
1109            let assumption_leaf = DerivationTree::leaf(
1110                assumed.clone(),
1111                InferenceRule::PremiseMatch,
1112            );
1113
1114            return Ok(Some(DerivationTree::new(
1115                goal.target.clone(),
1116                InferenceRule::ReductioAdAbsurdum,
1117                vec![assumption_leaf, contradiction_proof],
1118            )));
1119        }
1120
1121        Ok(None)
1122    }

1124    /// Try to prove ¬∃x P(x) by assuming ∃x P(x) and deriving contradiction.
1125    ///
1126    /// This is the core strategy for existence paradoxes like the Barber Paradox.
1127    /// Steps:
1128    /// 1. Assume ∃x P(x)
1129    /// 2. Skolemize to get P(c) for fresh constant c
1130    /// 3. Skolemize KB existentials (definite descriptions) to extract inner structure
1131    /// 4. Abstract event semantics to simple predicates
1132    /// 5. Instantiate universal premises with the Skolem constant
1133    /// 6. Extract uniqueness constraints and derive equalities
1134    /// 7. Look for contradiction (possibly via case analysis)
1135    fn try_existence_negation_proof(
1136        &mut self,
1137        goal: &ProofGoal,
1138        assumed_existence: &ProofExpr,
1139        depth: usize,
1140    ) -> ProofResult<Option<DerivationTree>> {
1141        // Skolemize the assumed existence: ∃x P(x) → P(c)
1142        let witness_facts = self.skolemize_existential(assumed_existence);
1143
1144        if witness_facts.is_empty() {
1145            return Ok(None);
1146        }
1147
1148        // Build extended context with witness facts
1149        let mut extended_context = goal.context.clone();
1150        extended_context.push(assumed_existence.clone());
1151
1152        // Add witness facts, abstracting events
1153        for fact in &witness_facts {
1154            let abstracted = self.abstract_all_events(fact);
1155            if !extended_context.contains(&abstracted) {
1156                extended_context.push(abstracted);
1157            }
1158            if !extended_context.contains(fact) {
1159                extended_context.push(fact.clone());
1160            }
1161        }
1162
1163        // Extract any Skolem constants from the witness facts
1164        let mut skolem_constants = self.extract_skolem_constants(&witness_facts);
1165
1166        // CRITICAL: Skolemize KB existentials to extract definite description structure.
1167        // Natural language "The barber" creates:
1168        // ∃y ((barber(y) ∧ ∀z (barber(z) → z = y)) ∧ ∀x ...)
1169        // We need to Skolemize these to access the inner universals.
1170        let kb_skolemized = self.skolemize_kb_existentials();
1171        for fact in &kb_skolemized {
1172            let abstracted = self.abstract_all_events(fact);
1173            if !extended_context.contains(&abstracted) {
1174                extended_context.push(abstracted);
1175            }
1176            if !extended_context.contains(fact) {
1177                extended_context.push(fact.clone());
1178            }
1179        }
1180
1181        // Extract additional Skolem constants from KB
1182        let kb_skolems = self.extract_skolem_constants(&kb_skolemized);
1183        for sk in kb_skolems {
1184            if !skolem_constants.contains(&sk) {
1185                skolem_constants.push(sk);
1186            }
1187        }
1188
1189        // Also extract unified definite description constants (e.g., "the_barber")
1190        // These are created by unify_definite_descriptions and should be treated like Skolems
1191        for expr in &self.knowledge_base {
1192            self.collect_unified_constants(expr, &mut skolem_constants);
1193        }
1194
1195        // Instantiate universal premises with Skolem constants
1196        let instantiated = self.instantiate_universals_with_constants(
1197            &extended_context,
1198            &skolem_constants,
1199        );
1200        for inst in &instantiated {
1201            let abstracted = self.abstract_all_events(inst);
1202            if !extended_context.contains(&abstracted) {
1203                extended_context.push(abstracted);
1204            }
1205        }
1206
1207        // Also process KB universals
1208        let kb_instantiated = self.instantiate_kb_universals_with_constants(&skolem_constants);
1209        for inst in &kb_instantiated {
1210            let abstracted = self.abstract_all_events(inst);
1211            if !extended_context.contains(&abstracted) {
1212                extended_context.push(abstracted);
1213            }
1214        }
1215
1216        // CRITICAL: Extract uniqueness constraints from definite descriptions
1217        // and derive equalities between Skolem constants and KB witnesses.
1218        // This handles Russell's definite descriptions: "The barber" creates
1219        // ∃y ((barber(y) ∧ ∀z (barber(z) → z = y)) ∧ ...)
1220        let derived_equalities = self.derive_equalities_from_uniqueness_constraints(
1221            &extended_context,
1222            &skolem_constants,
1223        );
1224
1225        // Add derived equalities to context
1226        for eq in &derived_equalities {
1227            if !extended_context.contains(eq) {
1228                extended_context.push(eq.clone());
1229            }
1230        }
1231
1232        // Apply derived equalities to substitute terms throughout context
1233        // This unifies facts about different barbers (sk_0, y, v) into a single entity
1234        let unified_context = self.apply_equalities_to_context(&extended_context, &derived_equalities);
1235
1236        // Look for direct contradiction first (in unified context)
1237        if let Some(contradiction_proof) = self.find_contradiction(&unified_context, depth)? {
1238            let assumption_leaf = DerivationTree::leaf(
1239                assumed_existence.clone(),
1240                InferenceRule::PremiseMatch,
1241            );
1242
1243            return Ok(Some(DerivationTree::new(
1244                goal.target.clone(),
1245                InferenceRule::ReductioAdAbsurdum,
1246                vec![assumption_leaf, contradiction_proof],
1247            )));
1248        }
1249
1250        // Try case analysis for self-referential structures (like Barber Paradox)
1251        if let Some(case_proof) = self.try_case_analysis_contradiction(&unified_context, &skolem_constants, depth)? {
1252            let assumption_leaf = DerivationTree::leaf(
1253                assumed_existence.clone(),
1254                InferenceRule::PremiseMatch,
1255            );
1256
1257            return Ok(Some(DerivationTree::new(
1258                goal.target.clone(),
1259                InferenceRule::ReductioAdAbsurdum,
1260                vec![assumption_leaf, case_proof],
1261            )));
1262        }
1263
1264        Ok(None)
1265    }

1269    /// This is essential for definite descriptions from natural language.
1270    /// "The barber" creates `∃y ((barber(y) ∧ ∀z (barber(z) → z = y)) ∧ ...)`.

1290    ///
1291    /// Given facts like `barber(sk_0)` and uniqueness constraints like
1292    /// `∀z (barber(z) → z = y)`, derive `sk_0 = y`.
1293    ///
1294    /// This is essential for Russell's definite descriptions where
1295    /// "The barber" creates `∃y ((barber(y) ∧ ∀z (barber(z) → z = y)) ∧ ...)`.
1296    fn derive_equalities_from_uniqueness_constraints(
1297        &self,
1298        context: &[ProofExpr],
1299        skolem_constants: &[String],
1300    ) -> Vec<ProofExpr> {
1301        let mut equalities = Vec::new();
1302
1303        // Collect all uniqueness constraints from KB and context
1304        // Pattern: ∀z (P(z) → z = c) where c is a constant/variable
1305        let uniqueness_constraints = self.extract_uniqueness_constraints(context);
1306
1307        // For each Skolem constant, check if it satisfies predicates
1308        // with uniqueness constraints
1309        for skolem in skolem_constants {
1310            for (predicate_name, unique_entity) in &uniqueness_constraints {
1311                // Check if we have predicate(skolem) in context
1312                let skolem_term = ProofTerm::Constant(skolem.clone());
1313                let skolem_satisfies_predicate = context.iter().any(|expr| {
1314                    self.predicate_matches(expr, predicate_name, &skolem_term)
1315                });
1316
1317                if skolem_satisfies_predicate {
1318                    // Derive: skolem = unique_entity
1319                    let equality = ProofExpr::Identity(
1320                        skolem_term.clone(),
1321                        unique_entity.clone(),
1322                    );
1323                    if !equalities.contains(&equality) {
1324                        equalities.push(equality);
1325                    }
1326
1327                    // Also add the symmetric version for easier matching
1328                    let sym_equality = ProofExpr::Identity(
1329                        unique_entity.clone(),
1330                        skolem_term.clone(),
1331                    );
1332                    if !equalities.contains(&sym_equality) {
1333                        equalities.push(sym_equality);
1334                    }
1335                }
1336            }
1337        }
1338
1339        // Derive transitive equalities: if sk_0 = y and sk_0 = v, then y = v
1340        let mut transitive_equalities = Vec::new();
1341        for eq1 in &equalities {
1342            if let ProofExpr::Identity(t1, t2) = eq1 {
1343                for eq2 in &equalities {
1344                    if let ProofExpr::Identity(t3, t4) = eq2 {
1345                        // If t1 = t2 and t1 = t4, then t2 = t4
1346                        if t1 == t3 && t2 != t4 {
1347                            let trans_eq = ProofExpr::Identity(t2.clone(), t4.clone());
1348                            if !equalities.contains(&trans_eq) && !transitive_equalities.contains(&trans_eq) {
1349                                transitive_equalities.push(trans_eq);
1350                            }
1351                        }
1352                        // If t1 = t2 and t3 = t1, then t2 = t3
1353                        if t1 == t4 && t2 != t3 {
1354                            let trans_eq = ProofExpr::Identity(t2.clone(), t3.clone());
1355                            if !equalities.contains(&trans_eq) && !transitive_equalities.contains(&trans_eq) {
1356                                transitive_equalities.push(trans_eq);
1357                            }
1358                        }
1359                    }
1360                }
1361            }
1362        }
1363        equalities.extend(transitive_equalities);
1364
1365        equalities
1366    }

1382    /// Recursively extract uniqueness constraints from an expression.
1383    fn extract_uniqueness_from_expr(&self, expr: &ProofExpr, constraints: &mut Vec<(String, ProofTerm)>) {
1384        match expr {
1385            // Direct uniqueness pattern: ∀z (P(z) → z = c)
1386            ProofExpr::ForAll { variable, body } => {
1387                if let ProofExpr::Implies(ante, cons) = body.as_ref() {
1388                    if let ProofExpr::Identity(left, right) = cons.as_ref() {
1389                        // Check if it's "z = c" where z is the quantified variable
1390                        let var_term = ProofTerm::Variable(variable.clone());
1391                        if left == &var_term {
1392                            // Extract the predicate name from the antecedent
1393                            if let Some(pred_name) = self.extract_unary_predicate_name(ante, variable) {
1394                                // right is the unique entity
1395                                constraints.push((pred_name, right.clone()));
1396                            }
1397                        } else if right == &var_term {
1398                            // Check c = z form
1399                            if let Some(pred_name) = self.extract_unary_predicate_name(ante, variable) {
1400                                constraints.push((pred_name, left.clone()));
1401                            }
1402                        }
1403                    }
1404                }
1405                // Recurse into body for nested structures
1406                self.extract_uniqueness_from_expr(body, constraints);
1407            }
1408
1409            // Conjunction: extract from both sides
1410            ProofExpr::And(left, right) => {
1411                self.extract_uniqueness_from_expr(left, constraints);
1412                self.extract_uniqueness_from_expr(right, constraints);
1413            }
1414
1415            // Existential: extract from body (definite descriptions are wrapped in ∃)
1416            ProofExpr::Exists { body, .. } => {
1417                self.extract_uniqueness_from_expr(body, constraints);
1418            }
1419
1420            _ => {}
1421        }
1422    }

1453    /// Apply derived equalities to substitute terms throughout context.
1454    ///
1455    /// This unifies facts about different entities (sk_0, y, v) by replacing
1456    /// all occurrences with a canonical representative (the first Skolem constant).
1457    fn apply_equalities_to_context(
1458        &self,
1459        context: &[ProofExpr],
1460        equalities: &[ProofExpr],
1461    ) -> Vec<ProofExpr> {
1462        if equalities.is_empty() {
1463            return context.to_vec();
1464        }
1465
1466        // Build a substitution map from equalities
1467        // Use the first term as the canonical representative
1468        let mut substitutions: Vec<(&ProofTerm, &ProofTerm)> = Vec::new();
1469        for eq in equalities {
1470            if let ProofExpr::Identity(t1, t2) = eq {
1471                // Prefer Skolem constants as canonical (they're from our assumption)
1472                if let ProofTerm::Constant(c) = t1 {
1473                    if c.starts_with("sk_") {
1474                        substitutions.push((t2, t1)); // t2 → t1 (Skolem)
1475                        continue;
1476                    }
1477                }
1478                if let ProofTerm::Constant(c) = t2 {
1479                    if c.starts_with("sk_") {
1480                        substitutions.push((t1, t2)); // t1 → t2 (Skolem)
1481                        continue;
1482                    }
1483                }
1484                // Default: first term is canonical
1485                substitutions.push((t2, t1));
1486            }
1487        }
1488
1489        // Apply substitutions to each expression in context
1490        let mut unified_context = Vec::new();
1491        for expr in context {
1492            let mut unified = expr.clone();
1493            for (from, to) in &substitutions {
1494                unified = self.substitute_term_in_expr(&unified, from, to);
1495            }
1496            // Add abstracted version too
1497            let abstracted = self.abstract_all_events(&unified);
1498            if !unified_context.contains(&unified) {
1499                unified_context.push(unified);
1500            }
1501            if !unified_context.contains(&abstracted) {
1502                unified_context.push(abstracted);
1503            }
1504        }
1505
1506        // Also add implications with substituted terms
1507        // This ensures cyclic implications like P(sk,sk) → ¬P(sk,sk) are in context
1508        for expr in context {
1509            if let ProofExpr::ForAll { variable, body } = expr {
1510                if let ProofExpr::Implies(_, _) = body.as_ref() {
1511                    // Find any Skolem constants and instantiate
1512                    for (from, to) in &substitutions {
1513                        if let ProofTerm::Constant(c) = to {
1514                            if c.starts_with("sk_") {
1515                                // Instantiate this universal with the Skolem constant
1516                                let mut subst = Substitution::new();
1517                                subst.insert(variable.clone(), (*to).clone());
1518                                let instantiated = apply_subst_to_expr(body, &subst);
1519                                let abstracted = self.abstract_all_events(&instantiated);
1520                                if !unified_context.contains(&abstracted) {
1521                                    unified_context.push(abstracted);
1522                                }
1523                            }
1524                        }
1525                    }
1526                }
1527            }
1528        }
1529
1530        unified_context
1531    }

1674    /// Try case analysis to derive a contradiction.
1675    ///
1676    /// For self-referential structures like the Barber Paradox:
1677    /// - Split on a predicate P(c, c) where c is a Skolem constant
1678    /// - Case 1: Assume P(c, c), derive ¬P(c, c) → contradiction
1679    /// - Case 2: Assume ¬P(c, c), derive P(c, c) → contradiction
1680    /// Either way we get contradiction (law of excluded middle).
1681    fn try_case_analysis_contradiction(
1682        &mut self,
1683        context: &[ProofExpr],
1684        skolem_constants: &[String],
1685        depth: usize,
1686    ) -> ProofResult<Option<DerivationTree>> {
1687        // Find candidate predicates for case splitting
1688        // Look for self-referential predicates: P(c, c) where c is a Skolem constant
1689        let candidates = self.find_case_split_candidates(context, skolem_constants);
1690
1691        for candidate in candidates {
1692            // Case 1: Assume the candidate is true
1693            let mut context_with_pos = context.to_vec();
1694            if !context_with_pos.contains(&candidate) {
1695                context_with_pos.push(candidate.clone());
1696            }
1697
1698            // Case 2: Assume the candidate is false
1699            let negated = ProofExpr::Not(Box::new(candidate.clone()));
1700            let mut context_with_neg = context.to_vec();
1701            if !context_with_neg.contains(&negated) {
1702                context_with_neg.push(negated.clone());
1703            }
1704
1705            // Try to derive contradiction in both cases
1706            let case1_contradiction = self.find_contradiction(&context_with_pos, depth)?;
1707            let case2_contradiction = self.find_contradiction(&context_with_neg, depth)?;
1708
1709            // If both cases lead to contradiction, we have a proof
1710            if let (Some(case1_proof), Some(case2_proof)) = (case1_contradiction, case2_contradiction) {
1711                // Build the case analysis proof tree
1712                let case1_tree = DerivationTree::new(
1713                    ProofExpr::Atom("⊥".into()),
1714                    InferenceRule::PremiseMatch,
1715                    vec![case1_proof],
1716                );
1717                let case2_tree = DerivationTree::new(
1718                    ProofExpr::Atom("⊥".into()),
1719                    InferenceRule::PremiseMatch,
1720                    vec![case2_proof],
1721                );
1722
1723                return Ok(Some(DerivationTree::new(
1724                    ProofExpr::Atom("⊥".into()),
1725                    InferenceRule::CaseAnalysis {
1726                        case_formula: format!("{}", candidate),
1727                    },
1728                    vec![case1_tree, case2_tree],
1729                )));
1730            }
1731        }
1732
1733        Ok(None)
1734    }

1736    /// Find candidate predicates for case splitting.
1737    ///
1738    /// Looks for:
1739    /// 1. Self-referential predicates: P(c, c) where c is a Skolem constant
1740    /// 2. Predicates that appear in contradictory implications: P → ¬P and ¬P → P
1741    fn find_case_split_candidates(
1742        &self,
1743        context: &[ProofExpr],
1744        skolem_constants: &[String],
1745    ) -> Vec<ProofExpr> {
1746        let mut candidates = Vec::new();
1747
1748        // Strategy 1: Find self-referential predicates P(c, c)
1749        for expr in context {
1750            if let ProofExpr::Predicate { name, args, world } = expr {
1751                // Check if it's a binary predicate with the same Skolem constant twice
1752                if args.len() == 2 {
1753                    if let (ProofTerm::Constant(c1), ProofTerm::Constant(c2)) = (&args[0], &args[1]) {
1754                        if c1 == c2 && skolem_constants.contains(c1) {
1755                            candidates.push(expr.clone());
1756                        }
1757                    }
1758                }
1759            }
1760        }
1761
1762        // Strategy 2: Find predicates involved in cyclic implications
1763        // Look for patterns like: (P → ¬P) ∧ (¬P → P)
1764        let implications: Vec<(ProofExpr, ProofExpr)> = context.iter()
1765            .chain(self.knowledge_base.iter())
1766            .filter_map(|e| {
1767                if let ProofExpr::Implies(ante, cons) = e {
1768                    Some(((**ante).clone(), (**cons).clone()))
1769                } else {
1770                    None
1771                }
1772            })
1773            .collect();
1774
1775        for (ante, cons) in &implications {
1776            // Check for P → ¬P pattern
1777            if let ProofExpr::Not(inner) = cons {
1778                if exprs_structurally_equal(ante, inner) {
1779                    // Found P → ¬P, check if ¬P → P also exists
1780                    let neg_ante = ProofExpr::Not(Box::new(ante.clone()));
1781                    for (a2, c2) in &implications {
1782                        if exprs_structurally_equal(a2, &neg_ante) && exprs_structurally_equal(c2, ante) {
1783                            // Found the cyclic pair - ante is a good candidate
1784                            if !candidates.contains(ante) {
1785                                candidates.push(ante.clone());
1786                            }
1787                        }
1788                    }
1789                }
1790            }
1791        }
1792
1793        // Strategy 3: Generate self-referential predicates for Skolem constants
1794        // For each Skolem constant sk_N, look for predicates P and create P(sk_N, sk_N)
1795        for const_name in skolem_constants {
1796            // Look for action predicates in implications
1797            for expr in context.iter().chain(self.knowledge_base.iter()) {
1798                if let ProofExpr::Implies(ante, cons) = expr {
1799                    // Extract predicate names from consequences
1800                    self.extract_predicate_template(cons, const_name, &mut candidates);
1801                }
1802            }
1803        }
1804
1805        candidates
1806    }

1852    // STRATEGY 5d: EXISTENTIAL ELIMINATION
1853    // =========================================================================
1854
1855    /// Try to eliminate existential quantifiers from premises.
1856    ///
1857    /// For each ∃x P(x) in the KB or context:
1858    /// 1. Generate a fresh Skolem constant c
1859    /// 2. Add P(c) to the context
1860    /// 3. Abstract any event semantics to simple predicates
1861    /// 4. Try to prove the goal with the extended context
1862    fn try_existential_elimination(
1863        &mut self,
1864        goal: &ProofGoal,
1865        depth: usize,
1866    ) -> ProofResult<Option<DerivationTree>> {
1867        // Depth guard to prevent infinite loops
1868        if depth > 8 {
1869            return Ok(None);
1870        }
1871
1872        // Find existential expressions in KB and context
1873        let existentials: Vec<ProofExpr> = self.knowledge_base.iter()
1874            .chain(goal.context.iter())
1875            .filter(|e| matches!(e, ProofExpr::Exists { .. }))
1876            .cloned()
1877            .collect();
1878
1879        if existentials.is_empty() {
1880            return Ok(None);
1881        }
1882
1883        // Try eliminating each existential
1884        for exist_expr in existentials {
1885            // Skolemize to get witness facts
1886            let witness_facts = self.skolemize_existential(&exist_expr);
1887
1888            if witness_facts.is_empty() {
1889                continue;
1890            }
1891
1892            // Abstract event semantics in witness facts
1893            let abstracted_facts: Vec<ProofExpr> = witness_facts.iter()
1894                .map(|f| self.abstract_all_events(f))
1895                .collect();
1896
1897            // Build extended context with witness facts
1898            let mut extended_context = goal.context.clone();
1899            for fact in &abstracted_facts {
1900                if !extended_context.contains(fact) {
1901                    extended_context.push(fact.clone());
1902                }
1903            }
1904
1905            // Also add the original witness facts (in case abstraction changes things)
1906            for fact in &witness_facts {
1907                if !extended_context.contains(fact) {
1908                    extended_context.push(fact.clone());
1909                }
1910            }
1911
1912            // Try to prove the goal with the extended context
1913            let extended_goal = ProofGoal::with_context(goal.target.clone(), extended_context);
1914
1915            // Use a fresh engine to avoid polluting our KB
1916            // But we need to be careful about depth to prevent loops
1917            if let Ok(inner_proof) = self.prove_goal(extended_goal, depth + 1) {
1918                // Build proof tree with existential elimination
1919                let witness_name = if let ProofExpr::Exists { variable, .. } = &exist_expr {
1920                    variable.clone()
1921                } else {
1922                    "witness".to_string()
1923                };
1924
1925                return Ok(Some(DerivationTree::new(
1926                    goal.target.clone(),
1927                    InferenceRule::ExistentialElim { witness: witness_name },
1928                    vec![inner_proof],
1929                )));
1930            }
1931        }
1932
1933        Ok(None)
1934    }

1949    /// Skolemize an existential expression.
1950    ///
1951    /// Given ∃x P(x), introduce a fresh Skolem constant c and return P(c).
1952    /// For nested structures like ∃x((type(x) ∧ unique(x)) ∧ prop(x)),
1953    /// we extract the predicates with the Skolem constant.
1954    fn skolemize_existential(&mut self, expr: &ProofExpr) -> Vec<ProofExpr> {
1955        let mut results = Vec::new();
1956
1957        if let ProofExpr::Exists { variable, body } = expr {
1958            // Generate a fresh Skolem constant
1959            let skolem = format!("sk_{}", self.fresh_var());
1960
1961            // Apply substitution to the body
1962            let mut subst = Substitution::new();
1963            subst.insert(variable.clone(), ProofTerm::Constant(skolem.clone()));
1964
1965            let instantiated = apply_subst_to_expr(body, &subst);
1966
1967            // Flatten conjunctions into separate facts
1968            self.flatten_conjunction(&instantiated, &mut results);
1969
1970            // Handle nested existentials in the result
1971            let mut i = 0;
1972            while i < results.len() {
1973                if let ProofExpr::Exists { .. } = &results[i] {
1974                    let nested = results.remove(i);
1975                    let nested_skolem = self.skolemize_existential(&nested);
1976                    results.extend(nested_skolem);
1977                } else {
1978                    i += 1;
1979                }
1980            }
1981        }
1982
1983        results
1984    }

2071    /// Abstract Neo-Davidsonian event semantics to simple predicates.
2072    ///
2073    /// Converts: ∃e(Shave(e) ∧ Agent(e, x) ∧ Theme(e, y)) → shaves(x, y)
2074    ///
2075    /// This allows the proof engine to reason about events using simpler
2076    /// predicate logic, which is essential for paradoxes like the Barber Paradox.
2077    fn abstract_event_to_predicate(&self, expr: &ProofExpr) -> Option<ProofExpr> {
2078        match expr {
2079            // Direct NeoEvent abstraction
2080            ProofExpr::NeoEvent { verb, roles, .. } => {
2081                // Extract Agent and Theme/Patient roles
2082                let agent = roles.iter()
2083                    .find(|(role, _)| role == "Agent")
2084                    .map(|(_, term)| term.clone());
2085
2086                let theme = roles.iter()
2087                    .find(|(role, _)| role == "Theme" || role == "Patient")
2088                    .map(|(_, term)| term.clone());
2089
2090                // Build a simple predicate: verb(agent, theme) or verb(agent)
2091                let mut args = Vec::new();
2092                if let Some(a) = agent {
2093                    args.push(a);
2094                }
2095                if let Some(t) = theme {
2096                    args.push(t);
2097                }
2098
2099                // Lowercase the verb for predicate naming convention
2100                let pred_name = verb.to_lowercase();
2101
2102                Some(ProofExpr::Predicate {
2103                    name: pred_name,
2104                    args,
2105                    world: None,
2106                })
2107            }
2108
2109            // Handle Exists wrapping an event expression
2110            ProofExpr::Exists { variable, body } => {
2111                // Check if this is an event quantification
2112                if !self.is_event_variable(variable) {
2113                    return None;
2114                }
2115
2116                // Try direct NeoEvent abstraction
2117                if let Some(abstracted) = self.abstract_event_to_predicate(body) {
2118                    return Some(abstracted);
2119                }
2120
2121                // Try to parse conjunction of event predicates
2122                // Pattern: ∃e(Verb(e) ∧ Agent(e, x) ∧ Theme(e, y)) → verb(x, y)
2123                if let Some(abstracted) = self.abstract_event_conjunction(variable, body) {
2124                    return Some(abstracted);
2125                }
2126
2127                None
2128            }
2129
2130            _ => None,
2131        }
2132    }

2134    /// Abstract a conjunction of event predicates to a simple predicate.
2135    ///
2136    /// Handles: Verb(e) ∧ Agent(e, x) ∧ Theme(e, y) → verb(x, y)
2137    fn abstract_event_conjunction(&self, event_var: &str, body: &ProofExpr) -> Option<ProofExpr> {
2138        // Flatten the conjunction to get all components
2139        let mut components = Vec::new();
2140        self.flatten_conjunction(body, &mut components);
2141
2142        // Find verb predicate (single arg that matches event_var)
2143        let mut verb_name: Option<String> = None;
2144        let mut agent: Option<ProofTerm> = None;
2145        let mut theme: Option<ProofTerm> = None;
2146
2147        for comp in &components {
2148            if let ProofExpr::Predicate { name, args, .. } = comp {
2149                // Check if first arg is the event variable
2150                let first_is_event = args.first().map_or(false, |arg| {
2151                    matches!(arg, ProofTerm::Variable(v) | ProofTerm::BoundVarRef(v) if v == event_var)
2152                });
2153
2154                if !first_is_event && args.len() == 1 {
2155                    // Single arg predicate that's the event var
2156                    if let Some(ProofTerm::Variable(v) | ProofTerm::BoundVarRef(v)) = args.first() {
2157                        if v == event_var {
2158                            verb_name = Some(name.clone());
2159                            continue;
2160                        }
2161                    }
2162                }
2163
2164                if first_is_event {
2165                    match name.as_str() {
2166                        "Agent" if args.len() == 2 => {
2167                            agent = Some(args[1].clone());
2168                        }
2169                        "Theme" | "Patient" if args.len() == 2 => {
2170                            theme = Some(args[1].clone());
2171                        }
2172                        _ if args.len() == 1 && verb_name.is_none() => {
2173                            // This is probably the verb predicate: Verb(e)
2174                            verb_name = Some(name.clone());
2175                        }
2176                        _ => {}
2177                    }
2178                }
2179            }
2180        }
2181
2182        // If we found a verb, construct the simple predicate
2183        if let Some(verb) = verb_name {
2184            let mut args = Vec::new();
2185            if let Some(a) = agent {
2186                args.push(a);
2187            }
2188            if let Some(t) = theme {
2189                args.push(t);
2190            }
2191
2192            return Some(ProofExpr::Predicate {
2193                name: verb.to_lowercase(),
2194                args,
2195                world: None,
2196            });
2197        }
2198
2199        None
2200    }

2210    /// Recursively abstract all events in an expression.
2211    ///
2212    /// This transforms the entire expression tree, replacing event semantics
2213    /// with simple predicates wherever possible.
2214    fn abstract_all_events(&self, expr: &ProofExpr) -> ProofExpr {
2215        // First try direct abstraction
2216        if let Some(abstracted) = self.abstract_event_to_predicate(expr) {
2217            return abstracted;
2218        }
2219
2220        // Otherwise recurse into the structure
2221        match expr {
2222            ProofExpr::And(left, right) => ProofExpr::And(
2223                Box::new(self.abstract_all_events(left)),
2224                Box::new(self.abstract_all_events(right)),
2225            ),
2226            ProofExpr::Or(left, right) => ProofExpr::Or(
2227                Box::new(self.abstract_all_events(left)),
2228                Box::new(self.abstract_all_events(right)),
2229            ),
2230            ProofExpr::Implies(left, right) => ProofExpr::Implies(
2231                Box::new(self.abstract_all_events(left)),
2232                Box::new(self.abstract_all_events(right)),
2233            ),
2234            ProofExpr::Iff(left, right) => ProofExpr::Iff(
2235                Box::new(self.abstract_all_events(left)),
2236                Box::new(self.abstract_all_events(right)),
2237            ),
2238            ProofExpr::Not(inner) => {
2239                // Apply De Morgan for quantifiers: ¬∃x.P ≡ ∀x.¬P
2240                // This normalization is crucial for efficient proof search
2241                // (Converting negated existentials to universals helps the prover)
2242                if let ProofExpr::Exists { variable, body } = inner.as_ref() {
2243                    return ProofExpr::ForAll {
2244                        variable: variable.clone(),
2245                        body: Box::new(self.abstract_all_events(&ProofExpr::Not(body.clone()))),
2246                    };
2247                }
2248                // Note: We do NOT convert ¬∀x.P to ∃x.¬P because the prover
2249                // works better with universal quantifiers for backward chaining.
2250                ProofExpr::Not(Box::new(self.abstract_all_events(inner)))
2251            }
2252            ProofExpr::ForAll { variable, body } => {
2253                // Check for pattern: ∀x ¬(P ∧ Q) → ∀x (P → ¬Q)
2254                // This converts to implication form for better backward chaining
2255                if let ProofExpr::Not(inner) = body.as_ref() {
2256                    if let ProofExpr::And(left, right) = inner.as_ref() {
2257                        return ProofExpr::ForAll {
2258                            variable: variable.clone(),
2259                            body: Box::new(ProofExpr::Implies(
2260                                Box::new(self.abstract_all_events(left)),
2261                                Box::new(self.abstract_all_events(&ProofExpr::Not(right.clone()))),
2262                            )),
2263                        };
2264                    }
2265                }
2266                ProofExpr::ForAll {
2267                    variable: variable.clone(),
2268                    body: Box::new(self.abstract_all_events(body)),
2269                }
2270            }
2271            ProofExpr::Exists { variable, body } => {
2272                // Check if this is an event quantification that should be abstracted
2273                if self.is_event_variable(variable) {
2274                    if let Some(abstracted) = self.abstract_event_to_predicate(body) {
2275                        return abstracted;
2276                    }
2277                }
2278                // Otherwise keep the existential and recurse
2279                ProofExpr::Exists {
2280                    variable: variable.clone(),
2281                    body: Box::new(self.abstract_all_events(body)),
2282                }
2283            }
2284            // For other expressions, return as-is
2285            other => other.clone(),
2286        }
2287    }

2343    /// Look for a contradiction in the knowledge base and context.
2344    ///
2345    /// A contradiction exists when both P and ¬P are derivable.
2346    fn find_contradiction(
2347        &mut self,
2348        context: &[ProofExpr],
2349        depth: usize,
2350    ) -> ProofResult<Option<DerivationTree>> {
2351        // Collect all expressions from KB and context
2352        let all_exprs: Vec<ProofExpr> = self.knowledge_base.iter()
2353            .chain(context.iter())
2354            .cloned()
2355            .collect();
2356
2357        // Strategy 1: Look for direct P and ¬P pairs
2358        for expr in &all_exprs {
2359            if let ProofExpr::Not(inner) = expr {
2360                // We have ¬P, check if P exists directly
2361                for other in &all_exprs {
2362                    if exprs_structurally_equal(other, inner) {
2363                        // Found both P and ¬P directly
2364                        let pos_leaf = DerivationTree::leaf(
2365                            (**inner).clone(),
2366                            InferenceRule::PremiseMatch,
2367                        );
2368                        let neg_leaf = DerivationTree::leaf(
2369                            expr.clone(),
2370                            InferenceRule::PremiseMatch,
2371                        );
2372                        return Ok(Some(DerivationTree::new(
2373                            ProofExpr::Atom("⊥".into()),
2374                            InferenceRule::Contradiction,
2375                            vec![pos_leaf, neg_leaf],
2376                        )));
2377                    }
2378                }
2379            }
2380        }
2381
2382        // Strategy 2: Look for implications that derive contradictory results
2383        // Check if context fact P triggers P → ¬P (immediate contradiction)
2384        // Or if P triggers P → Q where ¬Q is also in context
2385        // Note: Extract implications from both top-level and inside ForAll quantifiers
2386        let mut implications: Vec<(ProofExpr, ProofExpr)> = Vec::new();
2387        for e in &all_exprs {
2388            if let ProofExpr::Implies(ante, cons) = e {
2389                implications.push(((**ante).clone(), (**cons).clone()));
2390            }
2391            // Also extract from inside ForAll (important for barber paradox!)
2392            if let ProofExpr::ForAll { body, .. } = e {
2393                if let ProofExpr::Implies(ante, cons) = body.as_ref() {
2394                    implications.push(((**ante).clone(), (**cons).clone()));
2395                }
2396            }
2397        }
2398
2399        // For each fact in the context, see if it triggers contradictory implications
2400        for fact in context {
2401            // Find all implications where fact matches the antecedent
2402            let mut derivable_consequences: Vec<ProofExpr> = Vec::new();
2403
2404            for (ante, cons) in &implications {
2405                // Try to unify the antecedent with the fact
2406                if let Ok(subst) = unify_exprs(fact, ante) {
2407                    let instantiated_cons = apply_subst_to_expr(cons, &subst);
2408                    derivable_consequences.push(instantiated_cons);
2409                }
2410
2411                // Also try matching conjunctive antecedents with multiple facts
2412                if let ProofExpr::And(left, right) = ante {
2413                    // Try to find facts matching both parts of the conjunction
2414                    if let Some(subst) = self.try_match_conjunction_antecedent(
2415                        left, right, &all_exprs
2416                    ) {
2417                        let instantiated_cons = apply_subst_to_expr(cons, &subst);
2418                        if !derivable_consequences.contains(&instantiated_cons) {
2419                            derivable_consequences.push(instantiated_cons);
2420                        }
2421                    }
2422                }
2423            }
2424
2425            // Check if any derived consequence contradicts the triggering fact
2426            for cons in &derivable_consequences {
2427                // Check if cons = ¬fact (the classic barber structure: P → ¬P)
2428                if let ProofExpr::Not(inner) = cons {
2429                    if exprs_structurally_equal(inner, fact) {
2430                        // fact triggered an implication that derives ¬fact
2431                        // This is a contradiction: fact ∧ ¬fact
2432                        let pos_leaf = DerivationTree::leaf(
2433                            fact.clone(),
2434                            InferenceRule::PremiseMatch,
2435                        );
2436                        let neg_leaf = DerivationTree::leaf(
2437                            cons.clone(),
2438                            InferenceRule::ModusPonens,
2439                        );
2440                        return Ok(Some(DerivationTree::new(
2441                            ProofExpr::Atom("⊥".into()),
2442                            InferenceRule::Contradiction,
2443                            vec![pos_leaf, neg_leaf],
2444                        )));
2445                    }
2446                }
2447
2448                // Check if cons contradicts any other fact in context
2449                for other in context {
2450                    if std::ptr::eq(fact as *const _, other as *const _) {
2451                        continue; // Skip the triggering fact itself
2452                    }
2453                    // Check if cons = ¬other
2454                    if let ProofExpr::Not(inner) = cons {
2455                        if exprs_structurally_equal(inner, other) {
2456                            let pos_leaf = DerivationTree::leaf(
2457                                other.clone(),
2458                                InferenceRule::PremiseMatch,
2459                            );
2460                            let neg_leaf = DerivationTree::leaf(
2461                                cons.clone(),
2462                                InferenceRule::ModusPonens,
2463                            );
2464                            return Ok(Some(DerivationTree::new(
2465                                ProofExpr::Atom("⊥".into()),
2466                                InferenceRule::Contradiction,
2467                                vec![pos_leaf, neg_leaf],
2468                            )));
2469                        }
2470                    }
2471                    // Check if other = ¬cons
2472                    if let ProofExpr::Not(inner_other) = other {
2473                        if exprs_structurally_equal(inner_other, cons) {
2474                            let pos_leaf = DerivationTree::leaf(
2475                                cons.clone(),
2476                                InferenceRule::ModusPonens,
2477                            );
2478                            let neg_leaf = DerivationTree::leaf(
2479                                other.clone(),
2480                                InferenceRule::PremiseMatch,
2481                            );
2482                            return Ok(Some(DerivationTree::new(
2483                                ProofExpr::Atom("⊥".into()),
2484                                InferenceRule::Contradiction,
2485                                vec![pos_leaf, neg_leaf],
2486                            )));
2487                        }
2488                    }
2489                }
2490            }
2491
2492            // Check if any pair of consequences contradicts each other
2493            for i in 0..derivable_consequences.len() {
2494                for j in (i + 1)..derivable_consequences.len() {
2495                    let cons1 = &derivable_consequences[i];
2496                    let cons2 = &derivable_consequences[j];
2497
2498                    // Check if cons1 = ¬cons2 or cons2 = ¬cons1
2499                    if let ProofExpr::Not(inner1) = cons1 {
2500                        if exprs_structurally_equal(inner1, cons2) {
2501                            // cons1 = ¬cons2, contradiction!
2502                            let pos_leaf = DerivationTree::leaf(
2503                                cons2.clone(),
2504                                InferenceRule::ModusPonens,
2505                            );
2506                            let neg_leaf = DerivationTree::leaf(
2507                                cons1.clone(),
2508                                InferenceRule::ModusPonens,
2509                            );
2510                            return Ok(Some(DerivationTree::new(
2511                                ProofExpr::Atom("⊥".into()),
2512                                InferenceRule::Contradiction,
2513                                vec![pos_leaf, neg_leaf],
2514                            )));
2515                        }
2516                    }
2517                    if let ProofExpr::Not(inner2) = cons2 {
2518                        if exprs_structurally_equal(inner2, cons1) {
2519                            // cons2 = ¬cons1, contradiction!
2520                            let pos_leaf = DerivationTree::leaf(
2521                                cons1.clone(),
2522                                InferenceRule::ModusPonens,
2523                            );
2524                            let neg_leaf = DerivationTree::leaf(
2525                                cons2.clone(),
2526                                InferenceRule::ModusPonens,
2527                            );
2528                            return Ok(Some(DerivationTree::new(
2529                                ProofExpr::Atom("⊥".into()),
2530                                InferenceRule::Contradiction,
2531                                vec![pos_leaf, neg_leaf],
2532                            )));
2533                        }
2534                    }
2535                }
2536            }
2537        }
2538
2539        // Strategy 3: Try to find self-referential contradictions (like Barber Paradox)
2540        if let Some(proof) = self.find_self_referential_contradiction(context, depth)? {
2541            return Ok(Some(proof));
2542        }
2543
2544        Ok(None)
2545    }

2547    /// Try to match a conjunctive antecedent with facts in the context.
2548    ///
2549    /// For an antecedent like (man(z) ∧ shave(z,z)), we need to find facts
2550    /// that match both parts with consistent variable bindings.
2551    fn try_match_conjunction_antecedent(
2552        &self,
2553        left: &ProofExpr,
2554        right: &ProofExpr,
2555        facts: &[ProofExpr],
2556    ) -> Option<Substitution> {
2557        // Try to find a fact that matches the left part
2558        for fact1 in facts {
2559            if let Ok(subst1) = unify_exprs(fact1, left) {
2560                // Apply this substitution to the right part
2561                let instantiated_right = apply_subst_to_expr(right, &subst1);
2562                // Now look for a fact that matches the instantiated right part
2563                for fact2 in facts {
2564                    if let Ok(subst2) = unify_exprs(fact2, &instantiated_right) {
2565                        // Combine substitutions
2566                        let mut combined = subst1.clone();
2567                        for (k, v) in subst2.iter() {
2568                            combined.insert(k.clone(), v.clone());
2569                        }
2570                        return Some(combined);
2571                    }
2572                }
2573            }
2574        }
2575        // Also try right then left
2576        for fact1 in facts {
2577            if let Ok(subst1) = unify_exprs(fact1, right) {
2578                let instantiated_left = apply_subst_to_expr(left, &subst1);
2579                for fact2 in facts {
2580                    if let Ok(subst2) = unify_exprs(fact2, &instantiated_left) {
2581                        let mut combined = subst1.clone();
2582                        for (k, v) in subst2.iter() {
2583                            combined.insert(k.clone(), v.clone());
2584                        }
2585                        return Some(combined);
2586                    }
2587                }
2588            }
2589        }
2590        None
2591    }

2593    /// Special case: find self-referential contradictions (like the Barber Paradox).
2594    ///
2595    /// Pattern: If we have ∀x(P(x) → Q(b, x)) and ∀x(P(x) → ¬Q(b, x)),
2596    /// then for x = b with P(b), we get Q(b, b) ∧ ¬Q(b, b).
2597    ///
2598    /// This uses direct pattern matching WITHOUT recursive prove_goal calls
2599    /// to avoid infinite recursion.
2600    fn find_self_referential_contradiction(
2601        &mut self,
2602        context: &[ProofExpr],
2603        _depth: usize,
2604    ) -> ProofResult<Option<DerivationTree>> {
2605        // Collect all expressions from KB and context
2606        let all_exprs: Vec<ProofExpr> = self.knowledge_base.iter()
2607            .chain(context.iter())
2608            .cloned()
2609            .collect();
2610
2611        // Look for pairs of universal implications with contradictory conclusions
2612        // that can be instantiated with the same witness
2613        for expr1 in &all_exprs {
2614            if let ProofExpr::ForAll { variable: var1, body: body1 } = expr1 {
2615                if let ProofExpr::Implies(ante1, cons1) = body1.as_ref() {
2616                    for expr2 in &all_exprs {
2617                        if std::ptr::eq(expr1, expr2) {
2618                            continue; // Skip same expression
2619                        }
2620                        if let ProofExpr::ForAll { variable: var2, body: body2 } = expr2 {
2621                            if let ProofExpr::Implies(ante2, cons2) = body2.as_ref() {
2622                                // Check if cons2 = ¬cons1 (structurally)
2623                                if let ProofExpr::Not(neg_cons2) = cons2.as_ref() {
2624                                    // Check if cons1 and neg_cons2 have matching structure
2625                                    // For barber: cons1 = shaves(barber, x), neg_cons2 = shaves(barber, x)
2626
2627                                    // Try instantiating with x = barber (the self-referential case)
2628                                    // We look for constant terms in cons1 that could be witnesses
2629                                    let witnesses = self.extract_constants_from_expr(cons1);
2630
2631                                    for witness_name in &witnesses {
2632                                        let witness = ProofTerm::Constant(witness_name.clone());
2633
2634                                        // Instantiate both antecedents and consequents with this witness
2635                                        let mut subst1 = Substitution::new();
2636                                        subst1.insert(var1.clone(), witness.clone());
2637                                        let ante1_inst = apply_subst_to_expr(ante1, &subst1);
2638                                        let cons1_inst = apply_subst_to_expr(cons1, &subst1);
2639
2640                                        let mut subst2 = Substitution::new();
2641                                        subst2.insert(var2.clone(), witness.clone());
2642                                        let ante2_inst = apply_subst_to_expr(ante2, &subst2);
2643                                        let cons2_inst = apply_subst_to_expr(cons2, &subst2);
2644
2645                                        // Check if cons1_inst and ¬cons2_inst contradict
2646                                        // cons2_inst should be ¬X where X = cons1_inst
2647                                        if let ProofExpr::Not(inner2) = &cons2_inst {
2648                                            if exprs_structurally_equal(&cons1_inst, inner2) {
2649                                                // Now check if both antecedents could hold
2650                                                // ante1 typically is ¬P(x,x) and ante2 is P(x,x)
2651                                                // These are complementary - one must hold
2652                                                // For the paradox, we consider BOTH cases
2653
2654                                                // If ante1 = ¬P(x,x) and ante2 = P(x,x), and x = witness,
2655                                                // we have a tertium non datur case:
2656                                                // - Either P(w,w) holds → cons2_inst = ¬cons1_inst
2657                                                // - Or ¬P(w,w) holds → cons1_inst
2658
2659                                                // Check if ante1 and ante2 are complements
2660                                                if self.are_complements(&ante1_inst, &ante2_inst) {
2661                                                    // By excluded middle, one antecedent holds
2662                                                    // If cons1_inst and cons2_inst = ¬cons1_inst,
2663                                                    // we have a contradiction
2664                                                    let pos_leaf = DerivationTree::leaf(
2665                                                        cons1_inst.clone(),
2666                                                        InferenceRule::ModusPonens,
2667                                                    );
2668                                                    let neg_leaf = DerivationTree::leaf(
2669                                                        cons2_inst,
2670                                                        InferenceRule::ModusPonens,
2671                                                    );
2672                                                    return Ok(Some(DerivationTree::new(
2673                                                        ProofExpr::Atom("⊥".into()),
2674                                                        InferenceRule::Contradiction,
2675                                                        vec![pos_leaf, neg_leaf],
2676                                                    )));
2677                                                }
2678                                            }
2679                                        }
2680                                    }
2681                                }
2682                            }
2683                        }
2684                    }
2685                }
2686            }
2687        }
2688
2689        Ok(None)
2690    }

2716    fn extract_constants_recursive(&self, expr: &ProofExpr, constants: &mut Vec<String>) {
2717        match expr {
2718            ProofExpr::Predicate { args, .. } => {
2719                for arg in args {
2720                    self.extract_constants_from_term_recursive(arg, constants);
2721                }
2722            }
2723            ProofExpr::Identity(l, r) => {
2724                self.extract_constants_from_term_recursive(l, constants);
2725                self.extract_constants_from_term_recursive(r, constants);
2726            }
2727            ProofExpr::And(l, r)
2728            | ProofExpr::Or(l, r)
2729            | ProofExpr::Implies(l, r)
2730            | ProofExpr::Iff(l, r) => {
2731                self.extract_constants_recursive(l, constants);
2732                self.extract_constants_recursive(r, constants);

2734            ProofExpr::Not(inner) => {
2735                self.extract_constants_recursive(inner, constants);
2736            }
2737            ProofExpr::ForAll { body, .. } | ProofExpr::Exists { body, .. } => {
2738                self.extract_constants_recursive(body, constants);
2739            }

2741        }

2769    /// Try rewriting using equalities in the knowledge base.
2770    ///
2771    /// Leibniz's Law: If a = b and P(a), then P(b).
2772    /// Also handles symmetry (a = b ⊢ b = a) and transitivity (a = b, b = c ⊢ a = c).
2773    fn try_equality_rewrite(
2774        &mut self,
2775        goal: &ProofGoal,
2776        depth: usize,
2777    ) -> ProofResult<Option<DerivationTree>> {
2778        // Collect equalities from KB and context
2779        let equalities: Vec<(ProofTerm, ProofTerm)> = self
2780            .knowledge_base
2781            .iter()
2782            .chain(goal.context.iter())
2783            .filter_map(|expr| {
2784                if let ProofExpr::Identity(l, r) = expr {
2785                    Some((l.clone(), r.clone()))
2786                } else {
2787                    None
2788                }
2789            })
2790            .collect();
2791
2792        if equalities.is_empty() {
2793            return Ok(None);
2794        }
2795
2796        // Handle special case: goal is itself an equality (symmetry/transitivity)
2797        if let ProofExpr::Identity(goal_l, goal_r) = &goal.target {
2798            // Try symmetry: a = b ⊢ b = a
2799            if let Some(tree) = self.try_equality_symmetry(goal_l, goal_r, &equalities, depth)? {
2800                return Ok(Some(tree));
2801            }
2802
2803            // Try transitivity: a = b, b = c ⊢ a = c
2804            if let Some(tree) = self.try_equality_transitivity(goal_l, goal_r, &equalities, depth)? {
2805                return Ok(Some(tree));
2806            }
2807
2808            // Try equational rewriting: use axioms to rewrite LHS step by step
2809            // Only if we have depth budget remaining (prevents infinite recursion)
2810            if depth + 3 < self.max_depth {
2811                if let Some(tree) = self.try_equational_identity_rewrite(goal, goal_l, goal_r, depth)? {
2812                    return Ok(Some(tree));
2813                }
2814            }
2815
2816            return Ok(None);
2817        }
2818
2819        // Try rewriting: substitute one term for another (for non-Identity goals)
2820        for (eq_from, eq_to) in &equalities {
2821            // Try forward: a = b, P(a) ⊢ P(b)
2822            if let Some(tree) = self.try_rewrite_with_equality(
2823                goal, eq_from, eq_to, depth,
2824            )? {
2825                return Ok(Some(tree));
2826            }
2827
2828            // Try backward: a = b, P(b) ⊢ P(a)
2829            if let Some(tree) = self.try_rewrite_with_equality(
2830                goal, eq_to, eq_from, depth,
2831            )? {
2832                return Ok(Some(tree));
2833            }
2834        }
2835
2836        Ok(None)
2837    }

2839    /// Try to prove goal by substituting `from` with `to` in some known fact.
2840    fn try_rewrite_with_equality(
2841        &mut self,
2842        goal: &ProofGoal,
2843        from: &ProofTerm,
2844        to: &ProofTerm,
2845        depth: usize,
2846    ) -> ProofResult<Option<DerivationTree>> {
2847        // Create the "source" expression by substituting `to` with `from` in the goal
2848        // If goal is P(b) and we have a = b, we want to find P(a)
2849        let source_goal = self.substitute_term_in_expr(&goal.target, to, from);
2850
2851        // Check if source_goal differs from the goal (substitution had effect)
2852        if source_goal == goal.target {
2853            return Ok(None);
2854        }
2855
2856        // Try to prove the source goal
2857        let source_proof_goal = ProofGoal::with_context(source_goal.clone(), goal.context.clone());
2858        if let Ok(source_proof) = self.prove_goal(source_proof_goal, depth + 1) {
2859            // Also need a proof of the equality
2860            let equality = ProofExpr::Identity(from.clone(), to.clone());
2861            let eq_proof_goal = ProofGoal::with_context(equality.clone(), goal.context.clone());
2862
2863            if let Ok(eq_proof) = self.prove_goal(eq_proof_goal, depth + 1) {
2864                return Ok(Some(DerivationTree::new(
2865                    goal.target.clone(),
2866                    InferenceRule::Rewrite {
2867                        from: from.clone(),
2868                        to: to.clone(),
2869                    },
2870                    vec![eq_proof, source_proof],
2871                )));
2872            }
2873        }
2874
2875        Ok(None)
2876    }

2878    /// Try equality symmetry: a = b ⊢ b = a
2879    fn try_equality_symmetry(
2880        &mut self,
2881        goal_l: &ProofTerm,
2882        goal_r: &ProofTerm,
2883        equalities: &[(ProofTerm, ProofTerm)],
2884        _depth: usize,

2887        for (eq_l, eq_r) in equalities {
2888            if eq_l == goal_r && eq_r == goal_l {
2889                // Found r = l, can derive l = r by symmetry
2890                let source = ProofExpr::Identity(goal_r.clone(), goal_l.clone());
2891                return Ok(Some(DerivationTree::new(
2892                    ProofExpr::Identity(goal_l.clone(), goal_r.clone()),
2893                    InferenceRule::EqualitySymmetry,

2895                )));
2896            }

2901    /// Try equality transitivity: a = b, b = c ⊢ a = c
2902    fn try_equality_transitivity(
2903        &mut self,
2904        goal_l: &ProofTerm,
2905        goal_r: &ProofTerm,
2906        equalities: &[(ProofTerm, ProofTerm)],
2907        _depth: usize,
2908    ) -> ProofResult<Option<DerivationTree>> {
2909        // Look for a = b and b = c where we want a = c
2910        for (eq1_l, eq1_r) in equalities {
2911            if eq1_l == goal_l {
2912                // Found a = b, now look for b = c
2913                for (eq2_l, eq2_r) in equalities {
2914                    if eq2_l == eq1_r && eq2_r == goal_r {
2915                        // Found a = b and b = c, derive a = c
2916                        let premise1 = ProofExpr::Identity(eq1_l.clone(), eq1_r.clone());
2917                        let premise2 = ProofExpr::Identity(eq2_l.clone(), eq2_r.clone());
2918                        return Ok(Some(DerivationTree::new(
2919                            ProofExpr::Identity(goal_l.clone(), goal_r.clone()),
2920                            InferenceRule::EqualityTransitivity,
2921                            vec![
2922                                DerivationTree::leaf(premise1, InferenceRule::PremiseMatch),
2923                                DerivationTree::leaf(premise2, InferenceRule::PremiseMatch),
2924                            ],
2925                        )));
2926                    }
2927                }
2928            }
2929        }
2930        Ok(None)
2931    }

2933    /// Try equational rewriting for Identity goals.
2934    ///
2935    /// For a goal `f(a) = b`, find an axiom `f(x) = g(x)` that matches,
2936    /// rewrite to get `g(a) = b`, and recursively prove that.
2937    fn try_equational_identity_rewrite(
2938        &mut self,
2939        goal: &ProofGoal,
2940        goal_l: &ProofTerm,
2941        goal_r: &ProofTerm,
2942        depth: usize,
2943    ) -> ProofResult<Option<DerivationTree>> {
2944        // First, try congruence: if both sides have the same outermost function/ctor,
2945        // recursively prove the arguments are equal.
2946        if let (
2947            ProofTerm::Function(name_l, args_l),
2948            ProofTerm::Function(name_r, args_r),
2949        ) = (goal_l, goal_r)
2950        {
2951            if name_l == name_r && args_l.len() == args_r.len() {
2952                // All arguments must be equal
2953                let mut arg_proofs = Vec::new();
2954                let mut all_ok = true;
2955                for (arg_l, arg_r) in args_l.iter().zip(args_r.iter()) {
2956                    let arg_goal_expr = ProofExpr::Identity(arg_l.clone(), arg_r.clone());
2957                    let arg_goal = ProofGoal::with_context(arg_goal_expr, goal.context.clone());
2958                    match self.prove_goal(arg_goal, depth + 1) {
2959                        Ok(proof) => arg_proofs.push(proof),
2960                        Err(_) => {
2961                            all_ok = false;
2962                            break;
2963                        }
2964                    }
2965                }
2966                if all_ok {
2967                    // All arguments are equal, so the functions are equal by congruence
2968                    return Ok(Some(DerivationTree::new(
2969                        goal.target.clone(),
2970                        InferenceRule::Reflexivity, // Using Reflexivity for congruence
2971                        arg_proofs,
2972                    )));
2973                }
2974            }
2975        }
2976        // Collect Identity axioms from KB
2977        let axioms: Vec<(ProofTerm, ProofTerm)> = self
2978            .knowledge_base
2979            .iter()
2980            .filter_map(|e| {
2981                if let ProofExpr::Identity(l, r) = e {
2982                    Some((l.clone(), r.clone()))
2983                } else {
2984                    None
2985                }
2986            })
2987            .collect();
2988
2989        // Try each axiom to rewrite the goal's LHS
2990        for (axiom_l, axiom_r) in &axioms {
2991            // Rename variables in axiom to avoid capture (use same map for both sides!)
2992            let mut var_map = std::collections::HashMap::new();
2993            let renamed_l = self.rename_term_vars_with_map(axiom_l, &mut var_map);
2994            let renamed_r = self.rename_term_vars_with_map(axiom_r, &mut var_map);
2995
2996            // Try to unify axiom LHS with goal LHS
2997            // e.g., unify(Add(Succ(k), n), Add(Succ(Zero), Succ(Zero)))
2998            //       => {k: Zero, n: Succ(Zero)}
2999            if let Ok(subst) = unify_terms(&renamed_l, goal_l) {
3000                // Apply substitution to axiom RHS to get the rewritten term
3001                let rewritten = self.apply_subst_to_term(&renamed_r, &subst);
3002
3003                // First check: does rewritten equal goal_r directly?
3004                if terms_structurally_equal(&rewritten, goal_r) {
3005                    // Direct match! Build the proof
3006                    let axiom_expr = ProofExpr::Identity(axiom_l.clone(), axiom_r.clone());
3007                    return Ok(Some(DerivationTree::new(
3008                        goal.target.clone(),
3009                        InferenceRule::Rewrite {
3010                            from: goal_l.clone(),
3011                            to: rewritten,
3012                        },
3013                        vec![DerivationTree::leaf(axiom_expr, InferenceRule::PremiseMatch)],
3014                    )));
3015                }
3016
3017                // Otherwise, create a new goal with the rewritten LHS
3018                let new_goal_expr = ProofExpr::Identity(rewritten.clone(), goal_r.clone());
3019                let new_goal = ProofGoal::with_context(new_goal_expr.clone(), goal.context.clone());
3020
3021                // Recursively try to prove the new goal
3022                if let Ok(sub_proof) = self.prove_goal(new_goal, depth + 1) {
3023                    // Success! Build the full proof
3024                    let axiom_expr = ProofExpr::Identity(axiom_l.clone(), axiom_r.clone());
3025                    return Ok(Some(DerivationTree::new(
3026                        goal.target.clone(),
3027                        InferenceRule::Rewrite {
3028                            from: goal_l.clone(),
3029                            to: rewritten,
3030                        },
3031                        vec![
3032                            DerivationTree::leaf(axiom_expr, InferenceRule::PremiseMatch),
3033                            sub_proof,
3034                        ],
3035                    )));
3036                }
3037            }
3038        }
3039
3040        Ok(None)
3041    }

3246        let body = self.convert_typed_var_to_variable(goal, var_name);

3248        // Unify: ?Motive(#n) = body
3249        match unify_pattern(&pattern, &body) {

3251            Err(_) => None,
3252        }

3539        // First, try direct proof (might work for simple cases)
3540        if let Ok(proof) = self.prove_goal(goal.clone(), depth + 1) {
3541            return Ok(proof);
3542        }

3545        if let ProofExpr::Identity(lhs, rhs) = &goal.target {

3555    /// Try equational reasoning: rewrite LHS to match RHS using axioms and IH.
3556    ///
3557    /// For the step case of induction, we need to:
3558    /// 1. Find an axiom that matches the goal's LHS pattern
3559    /// 2. Use the axiom to rewrite LHS
3560    /// 3. Apply the induction hypothesis to simplify
3561    /// 4. Check if the result equals RHS
3562    fn try_equational_proof(
3563        &mut self,
3564        goal: &ProofGoal,
3565        lhs: &ProofTerm,
3566        rhs: &ProofTerm,
3567        ih: &ProofExpr,
3568        _depth: usize,
3569    ) -> ProofResult<Option<DerivationTree>> {
3570        // Find applicable equations from KB (Identity axioms)
3571        let equations: Vec<ProofExpr> = self
3572            .knowledge_base
3573            .iter()
3574            .filter(|e| matches!(e, ProofExpr::Identity(_, _)))
3575            .cloned()
3576            .collect();
3577
3578        // Try each equation to rewrite LHS
3579        for eq_axiom in &equations {
3580            if let ProofExpr::Identity(_, _) = &eq_axiom {
3581                // Rename variables in the axiom to avoid capture
3582                let renamed_axiom = self.rename_variables(&eq_axiom);
3583                if let ProofExpr::Identity(renamed_lhs, renamed_rhs) = renamed_axiom {
3584                    // Unify axiom LHS with goal LHS
3585                    // This binds axiom variables to goal terms
3586                    // e.g., unify(Add(Succ(x), m), Add(Succ(k), Zero)) gives {x->k, m->Zero}
3587                    if let Ok(subst) = unify_terms(&renamed_lhs, lhs) {
3588                        // Apply the substitution to the axiom's RHS
3589                        // This gives us what LHS rewrites to
3590                        let rewritten = self.apply_subst_to_term_with(&renamed_rhs, &subst);
3591
3592                        // Now check if rewritten equals RHS (possibly using IH)
3593                        if self.terms_equal_with_ih(&rewritten, rhs, ih) {
3594                            // Success! Build proof using the axiom and IH
3595                            let axiom_leaf =
3596                                DerivationTree::leaf(eq_axiom.clone(), InferenceRule::PremiseMatch);
3597
3598                            let ih_leaf =
3599                                DerivationTree::leaf(ih.clone(), InferenceRule::PremiseMatch);
3600
3601                            return Ok(Some(DerivationTree::new(
3602                                goal.target.clone(),
3603                                InferenceRule::PremiseMatch, // Equational step
3604                                vec![axiom_leaf, ih_leaf],
3605                            )));
3606                        }
3607                    }
3608                }
3609            }
3610        }
3611
3612        Ok(None)
3613    }

3615    /// Check if two terms are equal, potentially using the induction hypothesis.
3616    fn terms_equal_with_ih(&self, t1: &ProofTerm, t2: &ProofTerm, ih: &ProofExpr) -> bool {

3618        if t1 == t2 {
3619            return true;
3620        }
3621
3622        // Try using IH: if IH is `x = y`, and t1 contains x, replace with y
3623        if let ProofExpr::Identity(ih_lhs, ih_rhs) = ih {
3624            // Check if t1 can be transformed to t2 using IH

3626            if &t1_with_ih == t2 {
3627                return true;
3628            }

3630            // Also try the other direction
3631            let t2_with_ih = self.rewrite_term_with_equation(t2, ih_rhs, ih_lhs);
3632            if t1 == &t2_with_ih {
3633                return true;
3634            }
3635        }

3756    /// Substitute a TypedVar with a given expression throughout the goal.
3757    fn substitute_typed_var(
3758        &self,
3759        expr: &ProofExpr,
3760        var_name: &str,
3761        replacement: &ProofExpr,
3762    ) -> ProofExpr {
3763        match expr {
3764            ProofExpr::TypedVar { name, .. } if name == var_name => replacement.clone(),
3765            ProofExpr::Identity(l, r) => {
3766                let new_l = self.substitute_typed_var_in_term(l, var_name, replacement);
3767                let new_r = self.substitute_typed_var_in_term(r, var_name, replacement);
3768                ProofExpr::Identity(new_l, new_r)
3769            }
3770            ProofExpr::Predicate { name, args, world } => {
3771                let new_args: Vec<ProofTerm> = args
3772                    .iter()
3773                    .map(|a| self.substitute_typed_var_in_term(a, var_name, replacement))
3774                    .collect();
3775                ProofExpr::Predicate {
3776                    name: name.clone(),
3777                    args: new_args,
3778                    world: world.clone(),
3779                }
3780            }
3781            ProofExpr::And(l, r) => ProofExpr::And(
3782                Box::new(self.substitute_typed_var(l, var_name, replacement)),
3783                Box::new(self.substitute_typed_var(r, var_name, replacement)),
3784            ),
3785            ProofExpr::Or(l, r) => ProofExpr::Or(
3786                Box::new(self.substitute_typed_var(l, var_name, replacement)),
3787                Box::new(self.substitute_typed_var(r, var_name, replacement)),
3788            ),
3789            ProofExpr::Implies(l, r) => ProofExpr::Implies(
3790                Box::new(self.substitute_typed_var(l, var_name, replacement)),
3791                Box::new(self.substitute_typed_var(r, var_name, replacement)),
3792            ),
3793            ProofExpr::Iff(l, r) => ProofExpr::Iff(
3794                Box::new(self.substitute_typed_var(l, var_name, replacement)),
3795                Box::new(self.substitute_typed_var(r, var_name, replacement)),
3796            ),
3797            ProofExpr::Not(inner) => {
3798                ProofExpr::Not(Box::new(self.substitute_typed_var(inner, var_name, replacement)))
3799            }
3800            ProofExpr::ForAll { variable, body } => ProofExpr::ForAll {
3801                variable: variable.clone(),
3802                body: Box::new(self.substitute_typed_var(body, var_name, replacement)),
3803            },
3804            ProofExpr::Exists { variable, body } => ProofExpr::Exists {
3805                variable: variable.clone(),
3806                body: Box::new(self.substitute_typed_var(body, var_name, replacement)),
3807            },
3808            _ => expr.clone(),
3809        }
3810    }
